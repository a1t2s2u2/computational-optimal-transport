\section{基本的な定義}
\label{sec:prerequisites}

本節では，距離空間，測度，確率測度，コスト関数などの基本概念を定義し，
以降の議論で用いる記法を定める．

%% --- 2.1.1 基本的な記法と概念 ---
\subsection{基本的な記法と概念}
\label{subsec:basic-notation}

\begin{definition}{非負実数の空間}{nonneg-reals}
  $\R_+ \defeq \{x \in \R \mid x \geq 0\}$ と記す．
  $\R_+^n$ は $n$ 次元ベクトルで全成分が非負であるものの集合を表す．
\end{definition}

\begin{definition}{凸集合}{convex-set}
  $\R^n$ の部分集合 $S$ が\textbf{凸}（convex）であるとは，
  任意の $x, y \in S$ と $t \in [0, 1]$ に対して
  $tx + (1-t)y \in S$ が成り立つことをいう．
  すなわち，$S$ の任意の2点を結ぶ線分が $S$ に含まれる．
\end{definition}

\begin{definition}{凸包}{convex-hull}
  集合 $A \subset \R^n$ の\textbf{凸包}（convex hull）とは，
  $A$ を含むすべての凸集合の共通部分をいい，$\operatorname{conv}(A)$ と表す：
  \[
    \operatorname{conv}(A) \defeq \bigcap \{ S \subset \R^n \mid A \subset S,\; S \text{ は凸} \}.
  \]
\end{definition}

\begin{proposition}{凸包の特徴づけ}{convex-hull-char}
  集合 $A \subset \R^n$ に対して，
  \[
    \operatorname{conv}(A) = \left\{
    \sum_{i=1}^k \lambda_i x_i \;\middle|\;
    k \in \N,\; x_i \in A,\; \lambda_i \geq 0,\; \sum_{i=1}^k \lambda_i = 1
    \right\}.
  \]
  すなわち，凸包は $A$ の有限個の点の凸結合全体に一致する．
\end{proposition}

\begin{definition}{端点}{extreme-point}
  凸集合 $S$ の点 $x \in S$ が\textbf{端点}（extreme point）であるとは，
  $x = \frac{y + z}{2}$ かつ $y, z \in S$ ならば $x = y = z$ が成り立つことをいう．
  すなわち，$x$ は $S$ の他の2点の中点としては表せない．
\end{definition}

\begin{definition}{ほとんどいたるところ}{almost-everywhere}
  測度 $\mu$ に関して，性質 $P(x)$ が
  \textbf{ほとんどいたるところ}（almost everywhere; a.e.）成り立つとは，
  $P(x)$ が成り立たない点の集合 $\{x \mid \lnot P(x)\}$ が $\mu$-零集合
  （$\mu(\{x \mid \lnot P(x)\}) = 0$）であることをいう．
  $\mu$ がルベーグ測度の場合は単に「a.e.」と書く．
\end{definition}

\begin{definition}{上極限と下極限}{limsup-liminf}
  実数列 $(a_n)_{n \geq 1}$ に対して，
  \textbf{上極限}（limit superior）と\textbf{下極限}（limit inferior）を
  \[
    \limsup_{n \to \infty} a_n \defeq \lim_{n \to \infty} \sup_{k \geq n} a_k, \qquad
    \liminf_{n \to \infty} a_n \defeq \lim_{n \to \infty} \inf_{k \geq n} a_k
  \]
  で定義する．
  $\liminf a_n \leq \limsup a_n$ が常に成り立ち，
  両者が一致するとき通常の極限 $\lim a_n$ が存在してその値に等しい．
\end{definition}

\begin{definition}{下半連続}{lower-semicontinuous}
  距離空間 $(X, d)$ 上の関数 $f \colon X \to \R \cup \{+\infty\}$ が
  \textbf{下半連続}（lower semicontinuous）であるとは，
  任意の $x \in X$ と $x_n \to x$ なる任意の列に対して
  $f(x) \leq \liminf_{n \to \infty} f(x_n)$ が成り立つことをいう．
  連続関数は下半連続であるが，逆は成り立たない．
\end{definition}

%% --- 2.1.2 距離空間 ---
\subsection{距離空間}
\label{subsec:metric-space}

\begin{definition}{距離空間}{metric-space}
  集合 $X$ と関数 $d \colon X \times X \to [0, \infty)$ の組 $(X, d)$ が
  \textbf{距離空間}（metric space）であるとは，
  以下の3条件がすべての $x, y, z \in X$ に対して成り立つことをいう：
  \begin{enumerate}
    \item \textbf{同一性}：$d(x, y) = 0 \iff x = y$．
    \item \textbf{対称性}：$d(x, y) = d(y, x)$．
    \item \textbf{三角不等式}：$d(x, z) \leq d(x, y) + d(y, z)$．
  \end{enumerate}
  このとき $d$ を $X$ 上の\textbf{距離関数}（metric）と呼ぶ．
\end{definition}

\begin{example}{$\ell^p$ 距離空間}{metric-lp}
  実数 $p \geq 1$ に対して，$\R^n$ 上の $\ell^p$ 距離を
  \[
    d_p(x, y) = \norm{x - y}_p
    = \left(\sum_{i=1}^n \abs{x_i - y_i}^p\right)^{1/p}
  \]
  で定義する．特に $p = 2$ のときユークリッド距離となる．
  同一性と対称性は定義から直ちに従い，
  三角不等式はミンコフスキーの不等式（Minkowski's inequality）から成り立つ．
\end{example}

\begin{definition}{コーシー列}{cauchy-sequence}
  距離空間 $(X, d)$ における列 $(x_n)_{n \geq 1} \subset X$ が
  \textbf{コーシー列}（Cauchy sequence）であるとは，
  任意の $\varepsilon > 0$ に対してある $N \in \N$ が存在し，
  $n, m \geq N$ ならば $d(x_n, x_m) < \varepsilon$ が成り立つことをいう．
\end{definition}

\begin{definition}{完備距離空間}{complete-metric-space}
  距離空間 $(X, d)$ が\textbf{完備}（complete）であるとは，
  $X$ 上のすべてのコーシー列が $X$ の点に収束することをいう．
  すなわち，$(x_n)_{n \geq 1}$ が $X$ のコーシー列ならば，
  ある $x \in X$ が存在して $\lim_{n \to \infty} d(x_n, x) = 0$ となる．
\end{definition}

\begin{remark}{完備でない距離空間の例}{not-complete}
  有理数全体 $\Q$ にユークリッド距離 $d(x,y) = \abs{x - y}$ を入れた空間 $(\Q, d)$ は完備でない．
  例えば，$\sqrt{2}$ の十進展開を途中で打ち切った有理数列
  $x_1 = 1,\; x_2 = 1.4,\; x_3 = 1.41,\; x_4 = 1.414,\; \ldots$
  は $\Q$ 上のコーシー列であるが，極限 $\sqrt{2} \notin \Q$ であるから
  $\Q$ の中では収束しない．
  $\R$ は $\Q$ をこのような「穴」がないように完備化した空間と見なせる．
\end{remark}

\begin{definition}{コンパクト集合}{compact}
  距離空間 $(X, d)$ の部分集合 $K \subset X$ が\textbf{コンパクト}（compact）であるとは，
  $K$ の任意の点列が $K$ の点に収束する部分列を持つことをいう．
\end{definition}

\begin{example}{$\R^d$ におけるコンパクト集合}{compact-example}
  $\R^d$ の部分集合がコンパクトであることは，
  その集合が\textbf{有界}（ある開球 $B(0, R)$ に含まれる）かつ
  \textbf{閉}であることと同値である（Heine--Borel の定理）．
  例えば閉区間 $[a, b] \subset \R$ はコンパクトであるが，
  開区間 $(a, b)$ は閉でないためコンパクトでなく，
  $\R$ 自身は有界でないためコンパクトでない．
\end{example}

\begin{definition}{可算集合}{countable}
  集合 $A$ が\textbf{可算}（countable）であるとは，
  $A$ から自然数全体 $\N$ への単射が存在することをいう．
  すなわち，$A$ の元を $a_1, a_2, a_3, \ldots$ と番号付けできる．
  有限集合は可算であり，$\N$，$\Z$，$\Q$ は可算である．
  一方，$\R$ は可算でない（非可算である）．
\end{definition}

\begin{definition}{稠密部分集合と可分性}{separable}
  距離空間 $(X, d)$ の部分集合 $D \subset X$ が\textbf{稠密}（dense）であるとは，
  任意の $x \in X$ と任意の $\varepsilon > 0$ に対して，
  ある $y \in D$ が存在して $d(x, y) < \varepsilon$ が成り立つことをいう．
  $X$ が\textbf{可算}な稠密部分集合を持つとき，$X$ は\textbf{可分}（separable）であるという．
\end{definition}

\begin{example}{$\R$ の可分性}{separable-example}
  $\Q$ は $\R$ の稠密部分集合である．
  実際，任意の $x \in \R$ と任意の $\varepsilon > 0$ に対して，
  $\R$ のアルキメデスの性質
  （任意の $\varepsilon > 0$ に対して $n\varepsilon > 1$ となる $n \in \N$ が存在する）
  から $1/n < \varepsilon$ となる $n$ が取れる．
  すると $\abs{x - m/n} < 1/n < \varepsilon$ を満たす整数 $m$ が存在し，
  $q = m/n \in \Q$ が所望の有理数を与える．
  $\Q$ は可算集合であるから（定義~\ref{def:countable}），$\R$ は可分である．

  一方，$\Z$ は $\R$ の稠密部分集合ではない．
  例えば $x = 0.5$，$\varepsilon = 0.1$ に対して
  $\abs{0.5 - n} < 0.1$ を満たす $n \in \Z$ は存在しない．
\end{example}

\begin{definition}{ポーランド空間}{polish-space}
  完備かつ可分な距離空間を\textbf{ポーランド空間}（Polish space）と呼ぶ．
\end{definition}

\begin{example}{$\R^d$ はポーランド空間}{rd-polish}
  ユークリッド距離を備えた $\R^d$ はポーランド空間である．
  完備性は解析学の基本定理であり，
  可分性は $\Q^d$ が可算稠密部分集合であることから従う
  （例~\ref{ex:separable-example}）．
\end{example}

ポーランド空間を導入する理由は，この空間上で確率測度の集合に良いコンパクト性が成り立ち，
最適輸送問題の解の存在が保証されるためである．
詳細は確率測度を定義した後，\ref{subsec:prokhorov}~節で述べる．

\subsection{可測空間とボレル集合族}
\label{subsec:measurable-space}

測度を定義するためには，まず「測定可能な」集合の族を定める必要がある．
その準備として，距離空間における開集合と閉集合を定義する．

\begin{definition}{開球}{open-ball}
  距離空間 $(X, d)$ において，点 $x \in X$ と実数 $r > 0$ に対して，
  \textbf{開球}（open ball）を
  $B(x, r) \defeq \{y \in X \mid d(x, y) < r\}$
  で定義する．
\end{definition}

\begin{definition}{開集合と閉集合}{open-closed}
  距離空間 $(X, d)$ の部分集合 $U \subset X$ が\textbf{開集合}（open set）であるとは，
  任意の $x \in U$ に対してある $r > 0$ が存在して
  $B(x, r) \subset U$ が成り立つことをいう．
  すなわち，$U$ の各点のまわりに $U$ に含まれる開球が取れる．

  部分集合 $F \subset X$ が\textbf{閉集合}（closed set）であるとは，
  その補集合 $F^c = X \setminus F$ が開集合であることをいう．
  同値な特徴づけとして，$F$ が閉集合であることは，
  $F$ の点列の極限が常に $F$ に属すること
  （すなわち $x_n \in F$，$x_n \to x$ ならば $x \in F$）と同値である．
\end{definition}

\begin{definition}{$\sigma$-代数}{sigma-algebra}
  集合 $X$ の部分集合族 $\Bb \subset 2^X$ が\textbf{$\sigma$-代数}
  （$\sigma$-algebra）であるとは，以下の3条件を満たすことをいう：
  \begin{enumerate}
    \item $X \in \Bb$．
    \item $A \in \Bb$ ならば $A^c = X \setminus A \in \Bb$（補集合で閉じている）．
    \item $A_1, A_2, \ldots \in \Bb$ ならば
      $\bigcup_{k=1}^{\infty} A_k \in \Bb$（可算合併で閉じている）．
  \end{enumerate}
  組 $(X, \Bb)$ を\textbf{可測空間}（measurable space）と呼ぶ．
\end{definition}

\begin{definition}{ボレル $\sigma$-代数}{borel}
  距離空間 $(X, d)$ に対して，$X$ の\textbf{すべての開集合}を含む最小の $\sigma$-代数を
  \textbf{ボレル $\sigma$-代数}（Borel $\sigma$-algebra）と呼び，$\Bb(X)$ と記す．
  $\Bb(X)$ の元を\textbf{ボレル集合}（Borel set）と呼ぶ．
\end{definition}

直感的には，ボレル $\sigma$-代数は「開集合と閉集合を基本ブロックとして，
可算回の集合演算で構成できるすべての集合」の族と理解できる．
確率や統計，最適輸送で自然に現れる集合はほぼすべてボレル集合であり，
以降，距離空間上の測度は常にボレル $\sigma$-代数に関して定義されるものとする．

%% --- 2.1.4 測度と確率測度 ---
\subsection{測度と確率測度}
\label{subsec:measures}

\begin{definition}{測度}{measure}
  可測空間 $(X, \Bb)$ 上の\textbf{測度}（measure）とは，
  関数 $\mu \colon \Bb \to [0, \infty]$ であって，以下を満たすものをいう：
  \begin{enumerate}
    \item $\mu(\emptyset) = 0$．
    \item \textbf{可算加法性}：互いに素な列 $A_1, A_2, \ldots \in \Bb$
      に対して $\mu\!\left(\bigcup_{k=1}^{\infty} A_k\right) = \sum_{k=1}^{\infty} \mu(A_k)$．
  \end{enumerate}
\end{definition}

\begin{definition}{確率測度}{probability-measure}
  距離空間 $X$ 上の\textbf{確率測度}（probability measure）とは，
  ボレル $\sigma$-代数 $\Bb(X)$ 上の測度 $\mu$ であって $\mu(X) = 1$ を満たすものをいう．
  $X$ 上のすべての確率測度の集合を
  \[
    \Mm_+^1(X) \defeq \left\{ \mu \in \Mm_+(X) \;\middle|\; \mu(X) = 1 \right\}
  \]
  と記す．ここで $\Mm_+(X)$ は $X$ 上のすべての非負ボレル測度の集合である．
\end{definition}

\begin{remark}{測度の直感的理解}{measure-intuition}
  測度とは，集合の各部分に「大きさ」や「質量」を割り当てる関数である．
  確率測度は全体の質量が1に正規化されたものであり，「確率分布」と同義に用いられる．
  砂山の比喩を使えば，確率測度は「各領域にどれだけの砂があるか」を記述する．
\end{remark}

\begin{definition}{測度の台}{support}
  距離空間 $X$ 上の測度 $\mu$ の\textbf{台}（support）とは，
  \[
    \supp(\mu) \defeq \left\{ x \in X \;\middle|\;
    \text{$x$ を含むすべての開集合 $U$ に対して } \mu(U) > 0 \right\}
  \]
  で定義される $X$ の部分集合をいう．
\end{definition}

$\supp(\mu)$ は閉集合である．
実際，補集合
\[
  \supp(\mu)^c = \{x \in X \mid x \in U,\; \mu(U) = 0 \text{ なる開集合 } U \text{ が存在する}\}
\]
が開集合であることを示せばよい．
$x \in \supp(\mu)^c$ ならば $x \in U$，$\mu(U) = 0$ なる開集合 $U$ が存在する．
任意の $y \in U$ に対して，$U$ は $y$ の開近傍でもあり $\mu(U) = 0$ であるから，
$y \in \supp(\mu)^c$ となる．よって $U \subset \supp(\mu)^c$ であり，
$\supp(\mu)^c$ は開集合である．

\begin{definition}{Dirac測度}{dirac}
  距離空間 $X$ の点 $x \in X$ に対して，\textbf{Dirac測度}（Dirac measure）
  $\delta_x$ を
  \[
    \delta_x(A) \defeq
    \begin{cases}
      1 & x \in A, \\
      0 & x \notin A
    \end{cases}
    \qquad (A \in \Bb(X))
  \]
  で定義する．すなわち，$\delta_x$ は点 $x$ に全質量を集中させた確率測度である．
  Dirac測度に対する積分は
  $\int_X f(y) \, \d\delta_x(y) = f(x)$
  で与えられる．
\end{definition}

%% --- 2.1.5 確率測度のコンパクト性と Prokhorov の定理 ---
\subsection{確率測度のコンパクト性と Prokhorov の定理}
\label{subsec:prokhorov}

最適輸送の Kantorovich 問題（\ref{sec:kantorovich-overview}~節で定式化）では，
カップリング測度の集合の上でコストの下限を取る．
この下限が実際に達成される（最適解が存在する）ためには，
カップリングの集合がある種のコンパクト性を持つ必要がある．

ポーランド空間上では，確率測度の集合のコンパクト性を
「緊密性」という条件で特徴づけることができる．
これが Prokhorov の定理であり，
ポーランド空間を導入する根本的な理由である．
完備性がなければ極限測度が空間上の測度とならない可能性があり，
可分性がなければ以下の同値関係が成り立たない．

まず，確率測度の列が「収束する」とはどういう意味かを定める．
直感的には，測度 $\mu_n$ が $\mu$ に近づくとは，
任意の連続関数 $f$ に対して $f$ の $\mu_n$ による積分が $f$ の $\mu$ による積分に近づくことをいう．

\begin{definition}{有界関数}{bounded-function}
  距離空間 $X$ 上の関数 $f \colon X \to \R$ が\textbf{有界}（bounded）であるとは，
  ある定数 $M \geq 0$ が存在して，すべての $x \in X$ に対して $\abs{f(x)} \leq M$
  が成り立つことをいう．
\end{definition}

\begin{definition}{確率測度の弱収束}{weak-convergence}
  ポーランド空間 $X$ 上の確率測度の列 $(\mu_n)_{n \geq 1}$ が
  確率測度 $\mu$ に\textbf{弱収束}（weakly converge）するとは，
  すべての有界連続関数 $f \colon X \to \R$ に対して
  \[
    \lim_{n \to \infty} \int_X f \, \d\mu_n = \int_X f \, \d\mu
  \]
  が成り立つことをいう．
\end{definition}

弱収束に関する基本的な特徴づけとして，以下の定理がある．

\begin{theorem}{Portmanteau の定理}{portmanteau}
  ポーランド空間 $X$ 上の確率測度の列 $(\mu_n)$ と確率測度 $\mu$ に対して，
  以下は同値である：
  \begin{enumerate}
    \item $\mu_n$ が $\mu$ に弱収束する．
    \item すべての閉集合 $F \subset X$ に対して
      $\limsup_{n \to \infty} \mu_n(F) \leq \mu(F)$．
    \item すべての開集合 $U \subset X$ に対して
      $\liminf_{n \to \infty} \mu_n(U) \geq \mu(U)$．
  \end{enumerate}
\end{theorem}

\begin{remark}{}{portmanteau-remark}
  (2) と (3) は互いに補集合を取れば同値であることがわかる．
  Portmanteau の定理は，弱収束が「閉集合上では測度が増えない」
  「開集合上では測度が減らない」という直感的な性質と同値であることを述べている．
\end{remark}

弱収束する部分列を取るためには，
測度の質量が無限遠に「逃げない」ことが必要である．
例えば，$\R$ 上のDirac測度の列 $\mu_n = \delta_n$
（点 $n$ に全質量を集中させた測度）を考える．

収束する部分列を取ることができる．
緊密性はこのような状況を排除する条件である．

\begin{definition}{緊密性}{tightness}
  ポーランド空間 $X$ 上の確率測度の集合 $\Pi \subset \Mm_+^1(X)$ が
  \textbf{緊密}（tight）であるとは，
  任意の $\varepsilon > 0$ に対してコンパクト集合 $K \subset X$ が存在し，
  すべての $\mu \in \Pi$ に対して $\mu(K) \geq 1 - \varepsilon$ が成り立つことをいう．
  すなわち，$\Pi$ に属するすべての測度の質量が，共通のコンパクト集合にほぼ集中している．
\end{definition}

\begin{definition}{相対コンパクト性}{relatively-compact}
  距離空間の部分集合 $S$ が\textbf{相対コンパクト}（relatively compact）であるとは，
  $S$ の任意の列が収束する部分列を持つことをいう．
\end{definition}

コンパクト集合の定義（定義~\ref{def:compact}）では
部分列の極限が元の集合に属することを要求するが，
相対コンパクトでは極限が $S$ の外にあってもよい．
$\R^d$ において「有界な列は収束部分列を持つ」（Bolzano--Weierstrass の定理）は，
有界集合が相対コンパクトであることの主張に他ならない．

\begin{theorem}{Prokhorov の定理}{prokhorov}
  ポーランド空間 $X$ 上の確率測度の集合 $\Pi \subset \Mm_+^1(X)$ に対して，
  以下は同値である：
  \begin{enumerate}
    \item $\Pi$ は緊密である．
    \item $\Pi$ は弱収束に関して相対コンパクトである．
  \end{enumerate}
\end{theorem}

証明は Billingsley, \textit{Convergence of Probability Measures} (1999) 等の
測度論の教科書を参照されたい．
直感的には，(1) $\Rightarrow$ (2) は対角線論法と Riesz の表現定理により
弱収束する部分列を構成し，緊密性から極限が確率測度であることを示す．
(2) $\Rightarrow$ (1) は対偶を取り，緊密でなければ
弱収束する部分列が取れないことを示す．

%% --- 2.1.6 確率単体 ---
\subsection{確率単体}
\label{subsec:simplex}

有限集合 $\{1, \ldots, n\}$ 上の確率測度は，
非負の重みベクトル $\mathbf{a} = (a_1, \ldots, a_n)$ で $\sum_i a_i = 1$ を満たすものと
同一視できる．このようなベクトルの全体が確率単体である．

\begin{definition}{確率単体}{simplex}
  $n$ 次元の\textbf{確率単体}（probability simplex）とは，
  $\R^n$ の部分集合
  \[
    \simplex_n \defeq
    \left\{ \mathbf{a} \in \R_+^n \;\middle|\; \sum_{i=1}^n a_i = 1 \right\}
  \]
  をいう．$\simplex_n$ の元を\textbf{確率ベクトル}（probability vector）
  または\textbf{ヒストグラム}（histogram）と呼ぶ．
\end{definition}

\begin{remark}{}{simplex-properties}
  $\simplex_n$ は $\R^n$ の\textbf{凸}かつ\textbf{コンパクト}な部分集合である．
  その端点（extreme points）は標準基底ベクトル $e_1, \ldots, e_n$ であり，
  $\simplex_n$ はこれらの凸包に等しい：
  $\simplex_n = \operatorname{conv}(e_1, \ldots, e_n)$．
\end{remark}

\begin{example}{低次元の確率単体}{simplex-low-dim}
  \begin{itemize}
    \item $n = 2$ のとき，$\simplex_2 = \{(a_1, a_2) \in \R_+^2 \mid a_1 + a_2 = 1\}$
      は $(1, 0)$ と $(0, 1)$ を結ぶ線分である．
    \item $n = 3$ のとき，$\simplex_3$ は $\R^3$ 内の三角形であり，
      頂点 $(1,0,0)$, $(0,1,0)$, $(0,0,1)$ を持つ．
  \end{itemize}

  \begin{center}

    \coordinate (B) at (2, 0);

\end{example}

%% --- 2.1.7 離散測度 ---
\subsection{離散測度}
\label{subsec:discrete-measures}

Dirac測度（定義~\ref{def:dirac}）を用いて，有限個の点に質量を持つ測度を構成できる．

\begin{remark}{}{dirac-integration}
  離散測度に対する積分を和に帰着させる際の基本は，
  Dirac測度に対する積分の性質
  $\int_X f(y) \, \d\delta_x(y) = f(x)$（定義~\ref{def:dirac}）である．
\end{remark}

\begin{definition}{離散測度}{discrete-measure}
  距離空間 $X$ 上の\textbf{離散測度}（discrete measure）とは，
  有限個の点 $x_1, \ldots, x_n \in X$ と非負の重み $a_1, \ldots, a_n > 0$ に対して
  \[
    \alpha = \sum_{i=1}^n a_i \, \delta_{x_i}
  \]
  の形で表される測度をいう．
  さらに $\mathbf{a} = (a_1, \ldots, a_n) \in \simplex_n$ が成り立つとき，
  $\alpha$ は\textbf{離散確率測度}（discrete probability measure）と呼ばれる．
\end{definition}

\begin{remark}{ヒストグラムと離散測度の関係}{histogram-vs-discrete}
  ヒストグラム $\mathbf{a} \in \simplex_n$ は，ビンの位置が固定された場合の確率ベクトルである．
  離散測度 $\alpha = \sum_i a_i \delta_{x_i}$ は，
  確率ベクトル $\mathbf{a}$ に加えて，各ビンの\textbf{位置} $(x_1, \ldots, x_n)$ の情報も持つ．
  ヒストグラムは「重みのみ」を記述し，離散測度は「重みと位置の両方」を記述する．
  この区別は，特に異なる台を持つ離散測度を比較する際に本質的な意味を持つ．
\end{remark}

\begin{example}{経験分布}{empirical}
  データ点 $x_1, \ldots, x_n \in X$ が何らかの確率分布から独立にサンプルされたとき，
  \textbf{経験分布}（empirical distribution）
  \[
    \hat{\mu}_n = \frac{1}{n} \sum_{i=1}^n \delta_{x_i}
  \]
  は，データの分布を近似する離散確率測度である．
  大数の法則により，$n \to \infty$ で $\hat{\mu}_n$ は真の分布に弱収束する．
  経験分布は最適輸送の応用（特に統計的推定）において重要な対象である．
\end{example}

%% --- 2.1.6 測度の密度 ---
\subsection{測度の密度}
\label{subsec:density}

離散測度に対する対極として，ルベーグ測度に関して連続的な密度を持つ測度がある．

\begin{definition}{ルベーグ測度に関する密度}{lebesgue-density}
  $X = \R^d$ 上の測度 $\alpha$ が
  ルベーグ測度に関して\textbf{密度}（density）$\rho_\alpha$ を持つとは，
  非負可測関数 $\rho_\alpha \colon \R^d \to [0, \infty)$ が存在して，
  すべてのボレル集合 $A \in \Bb(\R^d)$ に対して
  \[
    \alpha(A) = \int_A \rho_\alpha(x) \, \d x
  \]
  が成り立つことをいう．ここで右辺はルベーグ積分である．
  このとき $\rho_\alpha = \frac{\d\alpha}{\d x}$ と記す．
  $\alpha$ が確率測度であるためには，$\int_{\R^d} \rho_\alpha(x) \, \d x = 1$ が必要十分である．
\end{definition}

\begin{definition}{ラドン--ニコディム導関数}{radon-nikodym}
  より一般に，測度 $\alpha$ が測度 $\beta$ に関して\textbf{絶対連続}
  （absolutely continuous; $\alpha \ll \beta$ と記す）であるとは，
  $\beta(A) = 0$ ならば $\alpha(A) = 0$ が成り立つことをいう．
  このとき，ラドン--ニコディムの定理により，
  $\beta$-可測関数 $\frac{\d\alpha}{\d\beta} \geq 0$ が $\beta$-a.e.\ に一意的に存在して，
  \[
    \alpha(A) = \int_A \frac{\d\alpha}{\d\beta}(x) \, \d\beta(x)
    \qquad (\forall\, A \in \Bb(X))
  \]
  が成り立つ．この関数 $\frac{\d\alpha}{\d\beta}$ を
  $\alpha$ の $\beta$ に関する\textbf{ラドン--ニコディム導関数}
  （Radon--Nikodym derivative）または\textbf{相対密度}と呼ぶ．
\end{definition}

\begin{remark}{測度の3つの類型}{three-types}
  最適輸送の理論では，主に以下の3種類の測度を扱う：
  \begin{enumerate}
    \item \textbf{ヒストグラム}：$\mathbf{a} \in \simplex_n$（有限次元の確率ベクトル）．
      ビンの位置は固定されており，最も単純な設定である．
    \item \textbf{離散測度}：$\alpha = \sum_i a_i \delta_{x_i}$（有限個の点に台を持つ測度）．
      重みと位置の両方が自由度を持つ．
    \item \textbf{連続測度}：$\d\alpha(x) = \rho_\alpha(x) \, \d x$
      （ルベーグ測度に関して密度を持つ測度）．
      最も一般的であるが，計算上は最も扱いが難しい．
  \end{enumerate}
  本書では，主にヒストグラムと離散測度を扱い，
  必要に応じて連続測度への拡張にも触れる．
\end{remark}

%% --- 2.1.7 コスト関数 ---
\subsection{コスト関数}
\label{subsec:cost-function}

最適輸送の目的関数を定めるのがコスト関数である．

\begin{definition}{コスト関数}{cost-function}
  空間 $\X$ と $\Y$ に対して，\textbf{コスト関数}（cost function）とは，
  可測関数
  \[
    c \colon \X \times \Y \to \R_+ \cup \{+\infty\}
  \]
  をいう．$c(x, y)$ は，点 $x \in \X$ から点 $y \in \Y$ へ単位質量を輸送するコストを表す．
\end{definition}

\begin{example}{典型的なコスト関数}{cost-examples}
  $(X, d)$ を距離空間とし，$\X = \Y = X$ とする．
  \begin{enumerate}
    \item \textbf{距離コスト}：$c(x, y) = d(x, y)$．
      $p$-Wasserstein 距離（$p=1$）に対応する．
    \item \textbf{二乗距離コスト}：$c(x, y) = d(x, y)^2$．
      最もよく研究されているケースであり，Brenier の定理の仮定となる．
      特に $X = \R^d$ では $c(x,y) = \norm{x-y}_2^2$ である．
    \item \textbf{$p$ 乗距離コスト}：$c(x, y) = d(x, y)^p$（$p \geq 1$）．
      $p$-Wasserstein 距離を導く．
    \item \textbf{離散コスト行列}：
      点集合 $x_1, \ldots, x_n \in \X$ と $y_1, \ldots, y_m \in \Y$ が固定されているとき，
      $n \times m$ 行列 $\mathbf{C}$ を $C_{i,j} = c(x_i, y_j)$ で定義する．
      これを\textbf{コスト行列}（cost matrix）と呼ぶ．
  \end{enumerate}
\end{example}

%% --- 2.2.1 最適割当問題 ---
\subsection{最適割当問題}
\label{subsec:assignment}

Monge 問題を定式化する前に，その特殊な場合である最適割当問題を導入する．

\begin{definition}{置換}{permutation}
  有限集合 $\range{n} = \{1, \ldots, n\}$ 上の\textbf{置換}（permutation）とは，
  $\range{n}$ から $\range{n}$ への全単射 $\sigma \colon \range{n} \to \range{n}$ をいう．
  $\range{n}$ 上のすべての置換の集合を $\Perm(n)$ と記す．
  $\Perm(n)$ の元の個数は $n!$ である．
\end{definition}

\begin{definition}{置換行列}{permutation-matrix}
  置換 $\sigma \in \Perm(n)$ に対して，\textbf{置換行列} $\mathbf{P}_\sigma \in \{0, 1\}^{n \times n}$ を
  \[
    (\mathbf{P}_\sigma)_{i,j} =
    \begin{cases}
      1 & j = \sigma(i), \\
      0 & j \neq \sigma(i)
    \end{cases}
  \]
  で定義する．置換行列は各行・各列にちょうど1つの $1$ を持つ行列であり，
  $\mathbf{P}_\sigma \ones_n = \ones_n$ かつ $\mathbf{P}_\sigma^\top \ones_n = \ones_n$ を満たす．
\end{definition}

\begin{setting}{最適割当問題}{assignment}
  コスト行列 $\mathbf{C} \in \R^{n \times n}$ が与えられたとき，
  \textbf{最適割当問題}（optimal assignment problem）とは，
  \[
    \min_{\sigma \in \Perm(n)} \frac{1}{n} \sum_{i=1}^n C_{i, \sigma(i)}
  \]
  を求める問題である．
  これは $n$ 個のソースと $n$ 個のターゲットの間に
  1対1対応を見つけ，総コストを最小化する問題である．
\end{setting}

\begin{remark}{}{assignment-complexity}
  置換の全探索は $n!$ 通りの列挙を要し，$n = 70$ でも $n! > 10^{100}$ となるため
  実用上不可能である．
  効率的な解法（オークションアルゴリズムやハンガリアン法など）は第3章で扱う．
\end{remark}

最適割当問題は，ソースとターゲットの個数が等しく（$n = m$），
すべての重みが均一（$a_i = b_j = 1/n$）という特殊な設定であった．
次に扱う Monge 問題は，個数が異なる場合（$n \neq m$）や
重みが不均一な場合を含む一般化である．
割当問題では写像が全単射（置換）であったのに対し，
Monge 問題では複数のソース点が同じターゲット点に送られることも許される．

\subsection{Monge 問題}
\label{subsec:monge}

Monge (1781) は，ある分布の質量を別の分布へ写像によって移すとき，
総コストを最小にする問題を定式化した．
この問題を正確に述べるためには，写像による測度の「移動」を定義する必要がある．

\begin{definition}{可測写像}{measurable-map}
  可測空間 $(X, \Bb(X))$ と $(Y, \Bb(Y))$ の間の写像 $T \colon X \to Y$ が
  \textbf{可測}（measurable）であるとは，
  任意の $B \in \Bb(Y)$ に対して $T^{-1}(B) \in \Bb(X)$ が成り立つことをいう．
\end{definition}

\begin{proposition}{連続写像は可測}{continuous-is-measurable}
  距離空間 $(X, d_X)$ と $(Y, d_Y)$ の間の連続写像 $T \colon X \to Y$ は，
  ボレル $\sigma$-代数に関して可測である．
\end{proposition}

\begin{proof}
  $U \subset Y$ を開集合とする．$T$ は連続であるから，$T^{-1}(U)$ は $X$ の開集合であり，
  $T^{-1}(U) \in \Bb(X)$ が成り立つ．
  $\Bb(Y)$ は $Y$ のすべての開集合を含む最小の $\sigma$-代数であるから，
  $\{B \in \Bb(Y) \mid T^{-1}(B) \in \Bb(X)\}$ が $\sigma$-代数であること
  （$T^{-1}$ が合併・補集合と可換であることから従う）と，
  この $\sigma$-代数がすべての開集合を含むことを合わせると，
  $\Bb(Y) \subset \{B \in \Bb(Y) \mid T^{-1}(B) \in \Bb(X)\}$ が得られる．
\end{proof}

\begin{definition}{押し出し（push-forward）}{pushforward}
  $(X, \Bb(X))$ と $(Y, \Bb(Y))$ を可測空間とし，
  $T \colon X \to Y$ を可測写像とする．
  測度 $\alpha \in \Mm_+(X)$ の $T$ による\textbf{押し出し}
  $T\pushforward \alpha \in \Mm_+(Y)$ を
  \[
    (T\pushforward \alpha)(B) \defeq \alpha(T^{-1}(B))
    \qquad (\forall\, B \in \Bb(Y))
  \]
  で定義する．ここで $T^{-1}(B) = \{x \in X \mid T(x) \in B\}$ は $B$ の逆像である．
\end{definition}

\begin{remark}{}{pushforward-integration}
  押し出しの定義を積分の言葉で述べると，
  すべての可測関数 $h \colon Y \to \R$ に対して
  \[
    \int_Y h(y) \, \d(T\pushforward \alpha)(y) = \int_X h(T(x)) \, \d\alpha(x)
  \]
  が成り立つ．離散測度 $\alpha = \sum_i a_i \delta_{x_i}$ に対しては，
  \[
    T\pushforward \alpha = \sum_{i=1}^n a_i \, \delta_{T(x_i)}
  \]
  となる．すなわち，押し出しは各点 $x_i$ の位置を $T(x_i)$ に移し，重み $a_i$ は保存する操作である．
\end{remark}

以上を用いて，Monge 問題を定式化する．

\begin{setting}{Monge 問題}{monge}
  ソース空間 $\X$ 上の確率測度 $\alpha \in \Mm_+^1(\X)$，
  ターゲット空間 $\Y$ 上の確率測度 $\beta \in \Mm_+^1(\Y)$，
  およびコスト関数 $c \colon \X \times \Y \to \R_+$ が与えられたとき，
  \textbf{Monge 問題}とは次の最適化問題をいう：
  \[
    \inf_{\substack{T \colon \X \to \Y \\ T\pushforward \alpha = \beta}}
    \int_{\X} c(x, T(x)) \, \d\alpha(x).
  \]
  ここで最適化は，$T\pushforward \alpha = \beta$ を満たすすべての可測写像 $T$ にわたって行う．
  条件 $T\pushforward \alpha = \beta$ は，写像 $T$ が
  ソースの質量分布をターゲットの質量分布に正確に移すことを要求する．
\end{setting}

Monge 問題には，以下の本質的な困難がある．

\begin{remark}{Monge 問題の困難}{monge-difficulty}
  \begin{enumerate}
    \item \textbf{非凸性}：
      実行可能集合 $\{T \colon T\pushforward \alpha = \beta\}$ は一般に\textbf{凸でない}．
      すなわち，2つの実行可能な写像 $T_1, T_2$ に対して，
      その凸結合 $(1-t)T_1 + tT_2$ は一般に実行可能でない
      （押し出し条件は写像に関して非線形だからである）．
      このため，標準的な凸最適化の手法が直接は適用できない．

    \item \textbf{解の非存在}：
      以下の例~\ref{ex:monge-nonexistence} に示すように，
      実行可能な写像 $T$ が\textbf{存在しない}場合がある．
      特に，ソース測度がターゲット測度より「粗い」
      （少数の点に質量が集中している）場合に，質量を分割できないという制約が障害となる．
  \end{enumerate}
\end{remark}

\begin{example}{Monge 写像が存在しない場合}{monge-nonexistence}
  $\X = \Y = \R$ 上で，ソース測度 $\alpha = \delta_0$（原点に全質量を集中），
  ターゲット測度 $\beta = \frac{1}{2}\delta_{-1} + \frac{1}{2}\delta_1$
  （$-1$ と $1$ に半分ずつ質量を配置）を考える．

  $T\pushforward \alpha = \beta$ を満たす写像 $T \colon \R \to \R$ が存在すると仮定する．
  $\alpha = \delta_0$ であるから，
  $T\pushforward \delta_0 = \delta_{T(0)}$ となる．
  しかし $\delta_{T(0)}$ は1点に集中した測度であり，
  $\beta = \frac{1}{2}\delta_{-1} + \frac{1}{2}\delta_1$ は2点に質量を持つ．
  よって $\delta_{T(0)} \neq \beta$ であり，矛盾する．

\section{Kantorovich 緩和の概要}
\label{sec:kantorovich-overview}

Kantorovich (1942) の革命的なアイデアは，
Monge の決定論的な輸送写像 $T$ を\textbf{確率的な輸送計画}に置き換えることである．
すなわち，各点 $x$ の質量を\textbf{1つの行き先に集中させる}代わりに，
\textbf{複数の行き先に分割して}送ることを許す．
この確率的な輸送計画は，$\X \times \Y$ 上の結合測度（カップリング）として定式化される．

\begin{definition}{カップリング}{coupling}
  確率測度 $\alpha \in \Mm_+^1(\X)$ と $\beta \in \Mm_+^1(\Y)$ に対して，
  $\alpha$ と $\beta$ の\textbf{カップリング}（coupling）の集合を
  \[
    \Couplings(\alpha, \beta) \defeq
    \left\{ \pi \in \Mm_+^1(\X \times \Y) \;\middle|\;
    (P_\X)\pushforward \pi = \alpha, \;
    (P_\Y)\pushforward \pi = \beta \right\}
  \]
  で定義する．ここで $P_\X(x, y) = x$，$P_\Y(x, y) = y$ はそれぞれ
  第1成分・第2成分への射影である．
  すなわち，$\pi$ の\textbf{周辺分布}（marginals）がそれぞれ $\alpha$, $\beta$ に一致することを要求する．
\end{definition}

\begin{remark}{周辺分布の条件}{marginal-condition}
  周辺分布の条件を積分で書くと，任意の可測関数 $\varphi, \psi$ に対して
  \begin{align*}
    \int_{\X \times \Y} \varphi(x) \, \d\pi(x, y) &= \int_\X \varphi(x) \, \d\alpha(x), \\
    \int_{\X \times \Y} \psi(y) \, \d\pi(x, y) &= \int_\Y \psi(y) \, \d\beta(y)
  \end{align*}
  が成り立つことと同値である．
  直感的には，$\pi(A \times \Y) = \alpha(A)$（$A \subset \X$）および
  $\pi(\X \times B) = \beta(B)$（$B \subset \Y$）が要求されている．
\end{remark}

\begin{remark}{離散版カップリング}{discrete-coupling}
  ヒストグラム $\mathbf{a} \in \simplex_n$，$\mathbf{b} \in \simplex_m$ に対する
  離散的なカップリングは $n \times m$ の非負行列 $\mathbf{P}$ で表され，
  \[
    \CouplingsD(\mathbf{a}, \mathbf{b}) \defeq
    \left\{ \mathbf{P} \in \R_+^{n \times m} \;\middle|\;
    \mathbf{P} \ones_m = \mathbf{a}, \;
    \mathbf{P}^\top \ones_n = \mathbf{b} \right\}
  \]
  で定義される．ここで $\ones_k$ は成分がすべて1の $k$ 次元ベクトルである．
  $P_{i,j}$ は「ソースのビン $i$ からターゲットのビン $j$ へ輸送される質量」を表す．
  行和の条件 $\mathbf{P} \ones_m = \mathbf{a}$ は供給制約，
  列和の条件 $\mathbf{P}^\top \ones_n = \mathbf{b}$ は需要制約に対応する．
\end{remark}

\begin{definition}{Birkhoff 多面体}{birkhoff}
  $n = m$ かつ $\mathbf{a} = \mathbf{b} = \ones_n / n$（均一重み）のとき，
  $\CouplingsD(\ones_n/n, \ones_n/n)$ を\textbf{Birkhoff 多面体}（Birkhoff polytope）と呼ぶ．
  これは二重確率行列（doubly stochastic matrix;
  すべての行和・列和が $1/n$ の非負行列）の集合に等しい．
\end{definition}

\begin{theorem}{Birkhoff--von Neumann の定理}{birkhoff-von-neumann}
  Birkhoff 多面体の\textbf{端点}（extreme point）は
  $\frac{1}{n}\mathbf{P}_\sigma$（$\sigma \in \Perm(n)$）の形の置換行列に限る．
  すなわち，任意の二重確率行列は置換行列の凸結合として表される．
\end{theorem}

\begin{remark}{}{birkhoff-consequence}
  線形計画問題の最適解は多面体の端点で達成されるから，
  均一重みの場合の Kantorovich 問題は置換行列で最適解を持つ．
  これは最適割当問題（設定~\ref{set:assignment}）の最適値と一致し，
  Kantorovich 緩和が割当問題のタイトな緩和であることを意味する．
\end{remark}

\begin{remark}{カップリングの存在}{coupling-existence}
  $\Couplings(\alpha, \beta)$ は常に\textbf{空でない}．
  実際，\textbf{積測度}（product measure）$\alpha \otimes \beta$ はカップリングの一つである：
  \[
    (\alpha \otimes \beta)(A \times B) = \alpha(A) \cdot \beta(B)
    \qquad (\forall\, A \in \Bb(\X), \; B \in \Bb(\Y)).
  \]
  これは「ソースとターゲットを独立に結合する」最も情報を持たないカップリングであり，
  一般に最適ではないが，実行可能性を保証する．
  離散版では $\mathbf{P} = \mathbf{a} \mathbf{b}^\top$（外積）が対応する．
\end{remark}

以上を用いて，Kantorovich 問題を定式化する．

\begin{setting}{Kantorovich 問題}{kantorovich}
  確率測度 $\alpha \in \Mm_+^1(\X)$，$\beta \in \Mm_+^1(\Y)$，
  およびコスト関数 $c \colon \X \times \Y \to \R_+$ に対して，
  \textbf{Kantorovich 問題}とは次の最適化問題をいう：
  \[
    \MK_c(\alpha, \beta) \defeq
    \inf_{\pi \in \Couplings(\alpha, \beta)}
    \int_{\X \times \Y} c(x, y) \, \d\pi(x, y).
  \]
  ヒストグラム $\mathbf{a} \in \simplex_n$，$\mathbf{b} \in \simplex_m$ と
  コスト行列 $\mathbf{C} \in \R_+^{n \times m}$ に対する離散版は
  \[
    \MKD_{\mathbf{C}}(\mathbf{a}, \mathbf{b}) \defeq
    \min_{\mathbf{P} \in \CouplingsD(\mathbf{a}, \mathbf{b})}
    \sum_{i,j} C_{i,j} P_{i,j}
    = \min_{\mathbf{P} \in \CouplingsD(\mathbf{a}, \mathbf{b})}
    \inner{\mathbf{C}}{\mathbf{P}}
  \]
  で与えられる．ここで $\inner{\mathbf{C}}{\mathbf{P}} = \tr(\mathbf{C}^\top \mathbf{P})$
  はフロベニウス内積である．
\end{setting}

\begin{remark}{確率論的解釈}{probabilistic-interpretation}
  Kantorovich 問題は確率論の言葉で次のように述べられる．
  確率変数 $X \sim \alpha$，$Y \sim \beta$ の同時分布 $(X, Y)$ のうち，
  周辺分布がそれぞれ $\alpha$, $\beta$ に一致するものを考えて，
  \[
    \MK_c(\alpha, \beta) = \inf_{\substack{(X, Y) \\ X \sim \alpha,\; Y \sim \beta}}
    \E\bigl[c(X, Y)\bigr].
  \]
  すなわち，周辺分布を固定したもとでの輸送コストの期待値を最小化する問題である．
  カップリング $\pi$ は同時分布に対応し，
  周辺分布の制約は $X \sim \alpha$，$Y \sim \beta$ に対応する．
\end{remark}

Kantorovich 緩和が Monge の定式化より優れている理由を整理する．

\begin{remark}{Kantorovich 緩和の利点}{kantorovich-advantages}
  \begin{enumerate}
    \item \textbf{凸性}：$\Couplings(\alpha, \beta)$ は凸集合である
      （2つのカップリングの凸結合は再びカップリング）．
      目的関数 $\pi \mapsto \int c \, \d\pi$ は $\pi$ に関して線形であるから，
      Kantorovich 問題は\textbf{凸最適化問題}である．
    \item \textbf{線形計画}：離散版は有限次元の\textbf{線形計画問題}（LP）であり，
      単体法や内点法などの効率的なアルゴリズムで解ける．
    \item \textbf{解の存在}：$\X, \Y$ がポーランド空間かつ $c$ が下半連続ならば，
      $\Couplings(\alpha, \beta)$ は緊密かつ弱位相で閉であることが示される．
      したがって Prokhorov の定理（定理~\ref{thm:prokhorov}）から
      $\Couplings(\alpha, \beta)$ は弱収束に関してコンパクトとなり，
      目的関数は弱収束に関して下半連続であるから，
      最適なカップリング $\pi^*$ が\textbf{常に存在}する．
      離散版でも，$\CouplingsD(\mathbf{a}, \mathbf{b})$ はコンパクト凸集合であるから，
      最適な輸送計画は常に存在する．
  \end{enumerate}
\end{remark}

Monge 問題と Kantorovich 問題の関係を明確にしておく．

\begin{proposition}{Monge 問題との関係}{monge-kantorovich-relation}
  $T\pushforward \alpha = \beta$ を満たす可測写像 $T \colon \X \to \Y$ に対して，
  \[
    \pi_T \defeq (\operatorname{Id}, T)\pushforward \alpha
  \]
  で定義される測度 $\pi_T \in \Mm_+^1(\X \times \Y)$ は
  $\Couplings(\alpha, \beta)$ の元であり，
  \[
    \int_{\X \times \Y} c(x, y) \, \d\pi_T(x, y)
    = \int_\X c(x, T(x)) \, \d\alpha(x)
  \]
  が成り立つ．ここで $(\operatorname{Id}, T)(x) = (x, T(x))$ である．
  したがって，
  \[
    \MK_c(\alpha, \beta) \leq
    \inf_{\substack{T \colon \X \to \Y \\ T\pushforward \alpha = \beta}}
    \int_\X c(x, T(x)) \, \d\alpha(x).
  \]
  すなわち，Kantorovich 問題は Monge 問題の\textbf{緩和}である．
\end{proposition}

\begin{proof}
  $\pi_T = (\operatorname{Id}, T)\pushforward \alpha$ の周辺分布を確認する．
  任意のボレル集合 $A \subset \X$ に対して，
  \[
    \pi_T(A \times \Y)
    = \alpha\bigl(\{x \in \X \mid (x, T(x)) \in A \times \Y\}\bigr)
    = \alpha(A),
  \]
  また任意のボレル集合 $B \subset \Y$ に対して，
  \[
    \pi_T(\X \times B)
    = \alpha\bigl(\{x \in \X \mid T(x) \in B\}\bigr)
    = \alpha(T^{-1}(B))
    = (T\pushforward \alpha)(B)
    = \beta(B).
  \]
  よって $\pi_T \in \Couplings(\alpha, \beta)$ である．

  コストの等式は，押し出しの定義から直ちに従う：
  \[
    \int_{\X \times \Y} c(x, y) \, \d\pi_T(x, y)
    = \int_\X c(x, T(x)) \, \d\alpha(x).
  \]
  Kantorovich 問題は $\Couplings(\alpha, \beta)$ 全体にわたる下限であるから，
  特に $\pi_T$ に対するコスト以下であり，不等式が得られる．
\end{proof}

\begin{remark}{}{monge-kantorovich-comparison}
  上の命題は，Monge 写像 $T$ による輸送がカップリングの特殊な場合

  であることを示している．

  \end{center}

  左の Monge 写像では各点 $x_i$ は1つの行き先にのみ送られるが，
  右の Kantorovich カップリングでは $x_2$ の質量が $y_1$ と $y_2$ に分割されている．
\end{remark}

\section{Wasserstein 距離の概要}
\label{sec:wasserstein-overview}

コスト関数が距離から導かれる場合，最適輸送コストは確率測度の空間に
自然な距離構造を導入する．これが Wasserstein 距離である．

\begin{definition}{$p$-Wasserstein 距離}{wasserstein}
  $(X, d)$ を距離空間とし，$p \geq 1$ とする．
  確率測度 $\alpha, \beta \in \Mm_+^1(X)$ に対して，
  \textbf{$p$-Wasserstein 距離}（$p$-Wasserstein distance）を
  \[
    \Wass_p(\alpha, \beta) \defeq
    \left(
    \inf_{\pi \in \Couplings(\alpha, \beta)}
    \int_{X \times X} d(x, y)^p \, \d\pi(x, y)
    \right)^{1/p}
  \]
  で定義する．すなわち，$\Wass_p(\alpha, \beta) = \MK_{d^p}(\alpha, \beta)^{1/p}$ である．

  ヒストグラム $\mathbf{a}, \mathbf{b}$ と距離行列 $\mathbf{D}$（$D_{i,j} = d(x_i, y_j)$）に対する
  離散版は
  \[
    \WassD_p(\mathbf{a}, \mathbf{b})
    = \MKD_{\mathbf{D}^{\circ p}}(\mathbf{a}, \mathbf{b})^{1/p}
  \]
  で与えられる．ここで $\mathbf{D}^{\circ p}$ は $D_{i,j}^p$ を成分とする行列である．
\end{definition}

\begin{proposition}{$\Wass_p$ は距離である}{wasserstein-metric}
  $(X, d)$ を距離空間，$p \geq 1$ とする．
  有限 $p$ 次モーメントを持つ確率測度の空間
  \[
    \Mm_+^{1,p}(X) \defeq
    \left\{ \alpha \in \Mm_+^1(X) \;\middle|\;
    \int_X d(x, x_0)^p \, \d\alpha(x) < \infty \right\}
  \]
  （ここで $x_0 \in X$ は任意の固定点）において，
  $\Wass_p$ は距離の公理を満たす：
  \begin{enumerate}
    \item $\Wass_p(\alpha, \beta) = 0 \iff \alpha = \beta$．
    \item $\Wass_p(\alpha, \beta) = \Wass_p(\beta, \alpha)$．
    \item $\Wass_p(\alpha, \gamma) \leq \Wass_p(\alpha, \beta) + \Wass_p(\beta, \gamma)$．
  \end{enumerate}
\end{proposition}

三角不等式の証明には以下の補題が必要である．

\begin{proposition}{糊付け補題}{gluing-lemma}
  ポーランド空間 $X, Y, Z$ 上の確率測度に対して，
  $\pi_1 \in \Couplings(\alpha, \beta)$（$X \times Y$ 上）と
  $\pi_2 \in \Couplings(\beta, \gamma)$（$Y \times Z$ 上）が与えられたとき，
  $X \times Y \times Z$ 上の確率測度 $\mu$ であって，
  $(X \times Y)$ への周辺分布が $\pi_1$ に，
  $(Y \times Z)$ への周辺分布が $\pi_2$ に一致するものが存在する．
\end{proposition}

\begin{proof}
  $\pi_1$ の $Y$ 上の条件付き分布を $\pi_1(\cdot \mid y)$，
  $\pi_2$ の $Y$ 上の条件付き分布を $\pi_2(\cdot \mid y)$ と書くと，
  $\mu$ を
  \[
    \d\mu(x, y, z) = \d\pi_1(x \mid y) \, \d\beta(y) \, \d\pi_2(z \mid y)
  \]
  で定義すればよい．
  $(X \times Y)$ への周辺は $\d\pi_1(x \mid y) \d\beta(y) = \d\pi_1(x, y)$，
  $(Y \times Z)$ への周辺は $\d\beta(y) \d\pi_2(z \mid y) = \d\pi_2(y, z)$
  となり，所望の性質を持つ．
\end{proof}

\begin{proof}[命題~\ref{prop:wasserstein-metric} の証明]
  \textbf{(1) 同一性}：
  $\alpha = \beta$ ならば $\pi = (\operatorname{Id}, \operatorname{Id})\pushforward \alpha$
  （対角線上に集中したカップリング）が $\Couplings(\alpha, \alpha)$ の元であり，
  $\int d(x,y)^p \, \d\pi = \int d(x,x)^p \, \d\alpha = 0$ であるから $\Wass_p(\alpha, \alpha) = 0$．
  逆に $\Wass_p(\alpha, \beta) = 0$ ならば，最適カップリング $\pi^*$ に対して
  $\int d(x,y)^p \, \d\pi^* = 0$ であるから，$\pi^*$ は $\{(x,y) \mid x = y\}$ 上に集中し，
  $\alpha = (P_\X)\pushforward \pi^* = (P_\Y)\pushforward \pi^* = \beta$ が従う．

  \textbf{(2) 対称性}：
  写像 $S(x, y) = (y, x)$ による押し出しは $\Couplings(\alpha, \beta)$ と $\Couplings(\beta, \alpha)$
  の間の全単射を与え，$d(x,y) = d(y,x)$ から $\Wass_p(\alpha, \beta) = \Wass_p(\beta, \alpha)$ が従う．

  \textbf{(3) 三角不等式}：
  $\pi_1 \in \Couplings(\alpha, \beta)$ と $\pi_2 \in \Couplings(\beta, \gamma)$ を
  それぞれ $\Wass_p(\alpha, \beta)$ と $\Wass_p(\beta, \gamma)$ の最適カップリングとする．
  \textbf{糊付け補題}（gluing lemma）により，
  $\X \times \Y \times \X$ 上の測度 $\mu$ であって，
  $(\X \times \Y)$ への周辺が $\pi_1$ に，$(\Y \times \X)$ への周辺が $\pi_2$ に
  一致するものが存在する．
  $(P_1, P_3)\pushforward \mu \in \Couplings(\alpha, \gamma)$
  （$P_1, P_3$ はそれぞれ第1・第3成分への射影）を
  $\alpha$ と $\gamma$ のカップリングとして用いると，
  ミンコフスキーの不等式から
  \begin{align*}
    \Wass_p(\alpha, \gamma)
    &\leq \left( \int d(x, z)^p \, \d\mu(x, y, z) \right)^{1/p} \\
    &\leq \left( \int d(x, y)^p \, \d\mu(x, y, z) \right)^{1/p}
    + \left( \int d(y, z)^p \, \d\mu(x, y, z) \right)^{1/p} \\
    &= \Wass_p(\alpha, \beta) + \Wass_p(\beta, \gamma)
  \end{align*}
  が得られる．2行目では距離の三角不等式 $d(x,z) \leq d(x,y) + d(y,z)$ と
  $L^p$ 空間のミンコフスキーの不等式を用いた．
\end{proof}

\begin{remark}{Wasserstein 距離の意義}{wasserstein-significance}
  Wasserstein 距離は，確率測度の空間に\textbf{基底空間の幾何学を反映した}距離を与える．
  この性質は，\textbf{KL ダイバージェンス}（Kullback--Leibler divergence）
  \[
    \KL(\alpha \| \beta) \defeq \int \log\frac{\d\alpha}{\d\beta} \, \d\alpha
    \qquad (\alpha \ll \beta \text{ のとき；そうでなければ } +\infty)
  \]
  など他の分布間の乖離度とは対照的である．
  ここで $\frac{\d\alpha}{\d\beta}$ はラドン--ニコディム導関数
  （定義~\ref{def:radon-nikodym}）である．
  KL ダイバージェンスは基底空間の距離構造を一切考慮しないが，
  Wasserstein 距離は点間の距離 $d(x, y)$ を通じてこれを取り込む．

  また，KL ダイバージェンスは台が重ならない分布間では $+\infty$ となるが，
  Wasserstein 距離は台が異なる場合でも有限の値を取り得る．
  このため，Wasserstein 距離は機械学習（特に生成モデル）において
  分布間の類似度を測る指標として広く用いられている．
\end{remark}

\begin{example}{Dirac測度間の Wasserstein 距離}{wasserstein-dirac}
  2つのDirac測度 $\delta_x$ と $\delta_y$（$x, y \in X$）に対して，
  唯一のカップリングは $\pi = \delta_{(x,y)}$ であるから，
  \[
    \Wass_p(\delta_x, \delta_y)
    = \left( \int_{X \times X} d(u, v)^p \, \d\delta_{(x,y)}(u,v) \right)^{1/p}
    = d(x, y).
  \]
  すなわち，Wasserstein 距離は\textbf{点間の距離を確率測度の距離に自然に持ち上げる}．
\end{example}

\begin{proposition}{Wasserstein 距離と弱収束}{wasserstein-weak-convergence}
  $(X, d)$ をポーランド空間，$p \geq 1$ とする．
  確率測度の列 $(\alpha_k)_{k \geq 1} \subset \Mm_+^{1,p}(X)$ と
  $\alpha \in \Mm_+^{1,p}(X)$ に対して，以下は同値である：
  \begin{enumerate}
    \item $\Wass_p(\alpha_k, \alpha) \to 0$．
    \item $\alpha_k$ が $\alpha$ に弱収束し，
      かつ $p$ 次モーメントが収束する：
      $\int_X d(x, x_0)^p \, \d\alpha_k(x) \to \int_X d(x, x_0)^p \, \d\alpha(x)$．
  \end{enumerate}
  特に $(X, d)$ がコンパクトならば，$\Wass_p$ による収束と弱収束は同値である．
\end{proposition}

\begin{remark}{$\Wass_\infty$ 距離}{wasserstein-infinity}
  $p \to \infty$ の極限として，\textbf{$\Wass_\infty$ 距離}を
  \[
    \Wass_\infty(\alpha, \beta) \defeq
    \inf_{\pi \in \Couplings(\alpha, \beta)}
    \sup_{(x,y) \in \supp(\pi)} d(x, y)
  \]
  で定義する（より正確には $\pi$ に関する本質的上限を取る）．
  $\Wass_\infty$ は台の間の Hausdorff 距離と関連するが，
  有限 $p$ の場合と異なり非凸最適化問題となる．
\end{remark}

\begin{remark}{平行移動の分解}{translation-decomposition}
  $X = \R^d$，$c(x,y) = \norm{x - y}_2^2$ のとき，
  確率測度 $\alpha$ の\textbf{平均}を
  $\bar{x}_\alpha \defeq \int_{\R^d} x \, \d\alpha(x)$
  と定め，$\tilde{\alpha}$ を $\alpha$ を平均が原点になるよう平行移動した測度とする．
  すなわち，$\tilde{\alpha} \defeq (x \mapsto x - \bar{x}_\alpha)\pushforward \alpha$ とすると，
  \[
    \Wass_2(\alpha, \beta)^2
    = \Wass_2(\tilde{\alpha}, \tilde{\beta})^2
    + \norm{\bar{x}_\alpha - \bar{x}_\beta}_2^2
  \]
  が成り立つ．Wasserstein 距離は「平均の差」と「平均を揃えた後の分布形状の差」に
  直交分解される．
  この分解はガウス分布の場合（定理~\ref{thm:wasserstein-gaussian}）の
  平均項と Bures 距離項への分解の一般化である．
\end{remark}

\section{双対問題}
\label{sec:dual}

Kantorovich 問題は線形計画問題であるから，線形計画の双対理論が適用できる．
双対問題は最適輸送の理論的理解と計算手法の両面で重要な役割を果たす．
ここではまず必要な最適化の概念を導入し，その後で双対問題を定式化する．

%% --- 2.5.1 線形計画と双対理論 ---
\subsection{線形計画と双対理論}
\label{subsec:lp-duality}

\begin{definition}{線形計画問題}{linear-program}
  ベクトル $\mathbf{c} \in \R^N$，行列 $\mathbf{A} \in \R^{M \times N}$，
  ベクトル $\mathbf{b} \in \R^M$ に対して，\textbf{線形計画問題}（linear program; LP）とは
  \[
    \min_{\mathbf{x} \in \R^N} \left\{
    \inner{\mathbf{c}}{\mathbf{x}} \;\middle|\;
    \mathbf{A}\mathbf{x} = \mathbf{b}, \; \mathbf{x} \geq 0
    \right\}
  \]
  の形の最適化問題をいう．
  目的関数 $\inner{\mathbf{c}}{\mathbf{x}}$ と制約がいずれも $\mathbf{x}$ に関して線形である．
\end{definition}

\begin{remark}{Kantorovich 問題は線形計画である}{kantorovich-as-lp}
  離散版の Kantorovich 問題
  $\min_{\mathbf{P} \in \CouplingsD(\mathbf{a}, \mathbf{b})} \inner{\mathbf{C}}{\mathbf{P}}$
  は，行列 $\mathbf{P}$ をベクトルに並べ直すことで
  上の形式の線形計画に帰着される．
  制約 $\mathbf{P}\ones_m = \mathbf{a}$，$\mathbf{P}^\top \ones_n = \mathbf{b}$，
  $\mathbf{P} \geq 0$ はいずれも線形である．
\end{remark}

\begin{definition}{ラグランジュ関数}{lagrangian}
  等式制約付き最適化問題 $\min_{\mathbf{x}} \{f(\mathbf{x}) \mid h_k(\mathbf{x}) = 0,\; k = 1, \ldots, M\}$
  に対して，\textbf{ラグランジュ関数}（Lagrangian）を
  \[
    L(\mathbf{x}, \boldsymbol{\lambda})
    \defeq f(\mathbf{x}) + \sum_{k=1}^M \lambda_k h_k(\mathbf{x})
  \]
  で定義する．ベクトル $\boldsymbol{\lambda} = (\lambda_1, \ldots, \lambda_M) \in \R^M$ を
  \textbf{ラグランジュ乗数}（Lagrange multiplier）と呼ぶ．
\end{definition}

\begin{definition}{双対問題}{dual-problem}
  主問題（primal problem）$\min_{\mathbf{x}} \{f(\mathbf{x}) \mid \mathbf{x} \in \mathcal{C}\}$
  に対して，ラグランジュ関数を用いた
  \[
    \max_{\boldsymbol{\lambda}} \inf_{\mathbf{x}} L(\mathbf{x}, \boldsymbol{\lambda})
  \]
  を\textbf{双対問題}（dual problem）と呼ぶ．
  一般に $\inf_{\mathbf{x}} L \leq f(\mathbf{x}^*)$ であるから，
  双対問題の最適値は主問題の最適値以下である（\textbf{弱双対性}; weak duality）．
  等号が成り立つとき\textbf{強双対性}（strong duality）が成り立つという．
  線形計画問題では，主問題が実行可能解を持つならば強双対性が常に成り立つ．
\end{definition}

%% --- 2.5.2 離散版の双対 ---
\subsection{離散版の双対問題}
\label{subsec:discrete-dual}

\begin{definition}{双対実行可能集合}{dual-feasible}
  コスト行列 $\mathbf{C} \in \R^{n \times m}$ に対して，
  \textbf{Kantorovich 双対ポテンシャル}の実行可能集合を
  \[
    \PotentialsD(\mathbf{C}) \defeq
    \left\{ (\mathbf{f}, \mathbf{g}) \in \R^n \times \R^m \;\middle|\;
    \forall\, (i, j), \; f_i + g_j \leq C_{i,j} \right\}
  \]
  で定義する．ベクトル $\mathbf{f}$, $\mathbf{g}$ を\textbf{双対変数}
  または\textbf{Kantorovich ポテンシャル}と呼ぶ．
\end{definition}

\begin{theorem}{Kantorovich 双対定理（離散版）}{kantorovich-duality-discrete}
  ヒストグラム $\mathbf{a} \in \simplex_n$，$\mathbf{b} \in \simplex_m$ と
  コスト行列 $\mathbf{C} \in \R^{n \times m}$ に対して，
  \[
    \MKD_{\mathbf{C}}(\mathbf{a}, \mathbf{b})
    = \max_{(\mathbf{f}, \mathbf{g}) \in \PotentialsD(\mathbf{C})}
    \inner{\mathbf{f}}{\mathbf{a}} + \inner{\mathbf{g}}{\mathbf{b}}.
  \]
  すなわち，Kantorovich 問題（主問題）の最適値は双対問題の最適値に等しい．
\end{theorem}

\begin{proof}
  Kantorovich 問題の主問題は
  \[
    \min_{\mathbf{P} \geq 0} \left\{
    \inner{\mathbf{C}}{\mathbf{P}} \;\middle|\;
    \mathbf{P} \ones_m = \mathbf{a}, \;
    \mathbf{P}^\top \ones_n = \mathbf{b}
    \right\}
  \]
  である．等式制約 $\mathbf{P} \ones_m = \mathbf{a}$，
  $\mathbf{P}^\top \ones_n = \mathbf{b}$ に対するラグランジュ乗数を
  $\mathbf{f} \in \R^n$，$\mathbf{g} \in \R^m$ とおくと，ラグランジュ関数は
  \[
    L(\mathbf{P}, \mathbf{f}, \mathbf{g})
    = \inner{\mathbf{C}}{\mathbf{P}}
    + \inner{\mathbf{f}}{\mathbf{a} - \mathbf{P} \ones_m}
    + \inner{\mathbf{g}}{\mathbf{b} - \mathbf{P}^\top \ones_n}
  \]
  となる．これを整理すると，
  \[
    L(\mathbf{P}, \mathbf{f}, \mathbf{g})
    = \inner{\mathbf{f}}{\mathbf{a}} + \inner{\mathbf{g}}{\mathbf{b}}
    + \inner{\mathbf{C} - \mathbf{f} \ones_m^\top - \ones_n \mathbf{g}^\top}{\mathbf{P}}.
  \]
  $\mathbf{P} \geq 0$ のもとで $L$ を $\mathbf{P}$ について最小化すると，
  $C_{i,j} - f_i - g_j \geq 0$（すなわち $f_i + g_j \leq C_{i,j}$）のとき
  最小値は $\inner{\mathbf{f}}{\mathbf{a}} + \inner{\mathbf{g}}{\mathbf{b}}$ であり，
  そうでなければ $-\infty$ となる．
  したがって双対問題は
  \[
    \max_{(\mathbf{f}, \mathbf{g}) \in \PotentialsD(\mathbf{C})}
    \inner{\mathbf{f}}{\mathbf{a}} + \inner{\mathbf{g}}{\mathbf{b}}
  \]
  となる．主問題は有界な実行可能領域を持つ線形計画であるから，
  強双対性が成り立ち，主問題と双対問題の最適値は一致する．
\end{proof}

\begin{proposition}{主双対最適性条件}{primal-dual-optimality}
  $\mathbf{P}^*$ と $(\mathbf{f}^*, \mathbf{g}^*)$ がそれぞれ主問題と双対問題の最適解ならば，
  \textbf{相補性条件}（complementary slackness）
  \[
    P^*_{i,j} > 0 \implies f^*_i + g^*_j = C_{i,j}
  \]
  が成り立つ．すなわち，最適輸送計画で正の質量が輸送される経路 $(i, j)$ では，
  双対ポテンシャルの和がコストに一致する．
\end{proposition}

\begin{proof}
  強双対性から $\inner{\mathbf{C}}{\mathbf{P}^*}
  = \inner{\mathbf{f}^*}{\mathbf{a}} + \inner{\mathbf{g}^*}{\mathbf{b}}$ が成り立つ．
  主問題の制約から
  $\inner{\mathbf{f}^*}{\mathbf{a}} + \inner{\mathbf{g}^*}{\mathbf{b}}
  = \inner{\mathbf{f}^* \ones_m^\top + \ones_n {\mathbf{g}^*}^\top}{\mathbf{P}^*}$ であるから，
  \[
    \inner{\mathbf{C} - \mathbf{f}^* \ones_m^\top - \ones_n {\mathbf{g}^*}^\top}{\mathbf{P}^*} = 0.
  \]
  $\mathbf{P}^* \geq 0$ かつ $C_{i,j} - f^*_i - g^*_j \geq 0$ であるから，
  各成分の積が非負でその和が $0$ となるためには，
  $P^*_{i,j} (C_{i,j} - f^*_i - g^*_j) = 0$ が
  すべての $(i, j)$ で成り立たなければならない．
\end{proof}

\begin{remark}{双対問題の経済学的解釈}{dual-economic}
  双対問題は次のように解釈できる．
  輸送業者が，倉庫 $i$ での集荷に $f_i$，工場 $j$ への配送に $g_j$ の料金を設定し，
  元の輸送会社に代わってすべての輸送を引き受けるとする．
  輸送業者の収入は $\inner{\mathbf{f}}{\mathbf{a}} + \inner{\mathbf{g}}{\mathbf{b}}$ である．

  輸送会社がこの契約を受け入れるためには，
  どの経路 $(i, j)$ についても輸送業者の料金が自前の輸送コスト以下でなければならない：
  $f_i + g_j \leq C_{i,j}$．
  この条件のもとで輸送業者が収入を最大化する問題が双対問題であり，
  その最適値が最適輸送コストに一致する．
\end{remark}

%% --- 2.5.3 連続版の双対 ---
\subsection{連続版の双対問題}
\label{subsec:continuous-dual}

離散版の双対問題では，制約 $f_i + g_j \leq C_{i,j}$ を満たすベクトルの組
$(\mathbf{f}, \mathbf{g})$ を最適化した．
連続版では，これが関数の組 $(f, g)$ に対する制約 $f(x) + g(y) \leq c(x, y)$ に置き換わる．
このとき，$f$ が与えられたときに制約を満たす最良の $g$ は
$g(y) = \inf_x [c(x, y) - f(x)]$ で定まる．
この操作を $c$-変換と呼ぶ．
$c$-変換により，双対問題を2つの関数 $(f, g)$ ではなく
1つの関数 $f$ のみで記述できるようになる．

\begin{definition}{$c$-変換}{c-transform}
  コスト関数 $c \colon \X \times \Y \to \R \cup \{+\infty\}$ と
  関数 $f \colon \X \to \R \cup \{-\infty\}$ に対して，
  $f$ の\textbf{$c$-変換}（$c$-transform）$f^c \colon \Y \to \R \cup \{-\infty\}$ を
  \[
    f^c(y) \defeq \inf_{x \in \X} \left[ c(x, y) - f(x) \right]
  \]
  で定義する．同様に，$g \colon \Y \to \R \cup \{-\infty\}$ の
  $\bar{c}$-変換 $g^{\bar{c}} \colon \X \to \R \cup \{-\infty\}$ を
  \[
    g^{\bar{c}}(x) \defeq \inf_{y \in \Y} \left[ c(x, y) - g(y) \right]
  \]
  で定義する．関数 $f$ が $f = g^{\bar{c}}$ なる $g$ をもつとき，
  $f$ は\textbf{$c$-凸}（$c$-convex）であるという．
\end{definition}

\begin{remark}{$c$-変換と凸共役}{c-transform-legendre}
  $c(x, y) = \inner{x}{y}$（$\X = \Y = \R^d$）のとき，
  $f^c(y) = \inf_x [\inner{x}{y} - f(x)]= -\sup_x [f(x) - \inner{x}{y}] = -f^*(y)$
  となる．ここで $f^*(y) = \sup_x [\inner{x}{y} - f(x)]$ は凸解析における
  ルジャンドル--フェンシェル変換（Legendre--Fenchel transform）である．
  すなわち，$c$-変換はルジャンドル変換のコスト関数 $c$ への一般化である．
\end{remark}

\begin{theorem}{Kantorovich 双対定理（連続版）}{kantorovich-duality-continuous}
  ポーランド空間 $\X, \Y$ 上の確率測度 $\alpha, \beta$ と
  下半連続なコスト関数 $c \colon \X \times \Y \to \R_+ \cup \{+\infty\}$ に対して，
  \[
    \MK_c(\alpha, \beta)
    = \sup_{\substack{f \in L^1(\alpha),\; g \in L^1(\beta) \\ f(x) + g(y) \leq c(x,y)}}
    \int_\X f \, \d\alpha + \int_\Y g \, \d\beta.
  \]
  さらに，双対問題の最適解 $(f^*, g^*)$ が存在し，$g^* = {f^*}^c$ と取れる．
\end{theorem}

\begin{remark}{制約なし双対}{unconstrained-dual}
  $c$-変換を用いると，双対問題を $f$ のみの制約なし最適化として書き直せる：
  \[
    \MK_c(\alpha, \beta)
    = \sup_{f}
    \left\{
    \int_\X f \, \d\alpha + \int_\Y f^c \, \d\beta
    \right\}.
  \]
  制約 $f(x) + g(y) \leq c(x, y)$ は $g = f^c$ と置くことで自動的に満たされる．
  この定式化はエントロピー正則化（第4章）や半離散最適輸送の計算において重要な役割を果たす．
\end{remark}

\section{特殊な場合}
\label{sec:special-cases}

%% --- 2.6.1 全変動距離との関係 ---
\subsection{全変動距離との関係}
\label{subsec:total-variation}

コスト関数の選び方によって，最適輸送コストは既知の確率分布間の距離と一致する．

\begin{remark}{Kronecker コスト関数と全変動}{kronecker-tv}
  有限集合 $X = \{x_1, \ldots, x_n\}$ 上の2つのヒストグラム
  $\mathbf{a}, \mathbf{b} \in \simplex_n$ を考える．
  コスト行列を $C_{i,j} = 1 - \delta_{i,j}$
  （$i = j$ のとき $0$，$i \neq j$ のとき $1$）と取ると，
  \[
    \MKD_{\mathbf{C}}(\mathbf{a}, \mathbf{b})
    = \frac{1}{2}\norm{\mathbf{a} - \mathbf{b}}_1
    = \frac{1}{2}\sum_{i=1}^n \abs{a_i - b_i}.
  \]
  右辺は\textbf{全変動距離}（total variation distance）に等しい．
  すなわち，移動先を区別しないコスト（同じ場所なら $0$，異なれば $1$）のもとでの
  最適輸送コストは，どれだけの質量を「動かす必要があるか」のみを測り，
  基底空間の幾何学的構造は反映しない．
\end{remark}

%% --- 2.6.2 1次元の場合 ---
\subsection{1次元の場合}
\label{subsec:one-dimensional}

$\X = \Y = \R$ の場合，最適輸送には美しい閉じた形の解が存在する．
この結果は累積分布関数と分位関数を用いて述べられる．

\begin{definition}{累積分布関数}{cdf}
  $\R$ 上の確率測度 $\alpha$ に対して，
  \textbf{累積分布関数}（cumulative distribution function; CDF）を
  \[
    F_\alpha(x) \defeq \alpha((-\infty, x]) = \int_{-\infty}^x \d\alpha
  \]
  で定義する．$F_\alpha$ は右連続かつ単調非減少であり，
  $\lim_{x \to -\infty} F_\alpha(x) = 0$，$\lim_{x \to +\infty} F_\alpha(x) = 1$ を満たす．
\end{definition}

\begin{definition}{分位関数}{quantile}
  累積分布関数 $F_\alpha$ に対して，
  \textbf{分位関数}（quantile function）または\textbf{一般化逆関数}を
  \[
    F_\alpha^{-1}(r) \defeq \inf \left\{ x \in \R \;\middle|\; F_\alpha(x) \geq r \right\}
    \qquad (r \in (0, 1])
  \]
  で定義する．$F_\alpha$ が狭義単調増加であるとき，$F_\alpha^{-1}$ は通常の逆関数に一致する．
\end{definition}

\begin{theorem}{1次元の最適輸送}{ot-one-dimensional}
  $\R$ 上の確率測度 $\alpha, \beta$ と実数 $p \geq 1$ に対して，
  \[
    \Wass_p(\alpha, \beta)^p
    = \int_0^1 \abs{F_\alpha^{-1}(r) - F_\beta^{-1}(r)}^p \, \d r.
  \]
  さらに，最適な Monge 写像は $T = F_\beta^{-1} \circ F_\alpha$ で与えられる．
\end{theorem}

\begin{proof}
  $T = F_\beta^{-1} \circ F_\alpha$ が $\alpha$ から $\beta$ への Monge 写像であることを確認する．
  $r \in (0, 1)$ に対して，分位関数の定義から
  $F_\alpha(F_\alpha^{-1}(r)) \geq r$ かつ $F_\beta^{-1}(F_\beta(y)) \leq y$ が成り立つ．
  任意のボレル集合 $B \subset \R$ に対して，
  \begin{align*}
    (T\pushforward \alpha)(B)
    &= \alpha(T^{-1}(B))
    = \alpha(\{x \mid F_\beta^{-1}(F_\alpha(x)) \in B\}) \\
    &= \alpha(\{x \mid F_\alpha(x) \in F_\beta(B)\}).
  \end{align*}
  ここで，$[0, 1]$ 上の一様分布を $\lambda$ とすると，
  $F_\alpha^{-1}$ が $\lambda$ から $\alpha$ への Monge 写像であること
  （すなわち $(F_\alpha^{-1})\pushforward \lambda = \alpha$）から，
  $F_\alpha$ は $\alpha$ を $[0,1]$ 上の一様分布に移す．
  同様に $F_\beta^{-1}$ は $\lambda$ を $\beta$ に移すから，
  合成 $T = F_\beta^{-1} \circ F_\alpha$ は $\alpha$ を $\beta$ に移す：
  $T\pushforward \alpha = \beta$．

  この $T$ による輸送コストは
  \[
    \int_\R \abs{x - T(x)}^p \, \d\alpha(x)
    = \int_0^1 \abs{F_\alpha^{-1}(r) - F_\beta^{-1}(r)}^p \, \d r
  \]
  となる（$x = F_\alpha^{-1}(r)$ と変数変換）．
  この値が Kantorovich 問題の最適値以下であること，
  および Kantorovich 問題の最適値がこの値以上であること
  （双対問題の実行可能解を構成すれば示せる）を合わせると，
  $T$ が最適な Monge 写像であり，
  上の積分が $\Wass_p(\alpha, \beta)^p$ に一致する．
\end{proof}

\begin{remark}{}{one-dim-remark}
  1次元では，Wasserstein 距離は分位関数の $L^p$ ノルムの差として表される．
  この等長性は高次元では成り立たず，1次元特有の性質である．
  最適な Monge 写像は「$\alpha$ の $r$-分位点を $\beta$ の $r$-分位点に送る」
  という自然な対応である．
\end{remark}

\begin{example}{1次元の離散的な場合}{one-dim-discrete}
  $\alpha = \frac{1}{n}\sum_{i=1}^n \delta_{x_i}$，
  $\beta = \frac{1}{n}\sum_{i=1}^n \delta_{y_i}$
  （$x_1 \leq \cdots \leq x_n$，$y_1 \leq \cdots \leq y_n$ と整列済み）のとき，
  最適な割当は $\sigma(i) = i$（順序を保つ対応）であり，
  \[
    \Wass_p(\alpha, \beta)^p = \frac{1}{n} \sum_{i=1}^n \abs{x_i - y_i}^p.
  \]
\end{example}

%% --- 2.6.2 ガウス分布の場合 ---
\subsection{ガウス分布の場合}
\label{subsec:gaussian}

多次元ガウス分布間の $\Wass_2$ 距離には閉じた形の公式が存在する．
まず必要な線形代数の概念を導入する．

\begin{definition}{正定値対称行列}{positive-definite}
  対称行列 $\mathbf{\Sigma} \in \R^{d \times d}$（$\mathbf{\Sigma}^\top = \mathbf{\Sigma}$）が
  \textbf{正定値}（positive definite）であるとは，
  すべての $\mathbf{0} \neq \mathbf{v} \in \R^d$ に対して
  $\mathbf{v}^\top \mathbf{\Sigma} \mathbf{v} > 0$ が成り立つことをいう．
  正定値対称行列のすべての固有値は正である．
\end{definition}

\begin{definition}{行列の平方根}{matrix-square-root}
  正定値対称行列 $\mathbf{\Sigma}$ に対して，
  その\textbf{平方根} $\mathbf{\Sigma}^{1/2}$ とは，
  $\mathbf{\Sigma}^{1/2} \mathbf{\Sigma}^{1/2} = \mathbf{\Sigma}$ かつ
  $\mathbf{\Sigma}^{1/2}$ 自身も正定値対称であるような一意的な行列をいう．
  固有値分解 $\mathbf{\Sigma} = \mathbf{U} \diag(\lambda_1, \ldots, \lambda_d) \mathbf{U}^\top$
  を用いると，$\mathbf{\Sigma}^{1/2} = \mathbf{U} \diag(\sqrt{\lambda_1}, \ldots, \sqrt{\lambda_d}) \mathbf{U}^\top$ で与えられる．
  $\mathbf{\Sigma}^{-1/2}$ は $\mathbf{\Sigma}^{1/2}$ の逆行列を表す．
\end{definition}

\begin{definition}{多次元ガウス分布}{gaussian}
  $\R^d$ 上の\textbf{ガウス分布}（Gaussian distribution）
  $\mathcal{N}(\mathbf{m}, \mathbf{\Sigma})$ は，
  平均ベクトル $\mathbf{m} \in \R^d$ と
  正定値対称行列である共分散行列 $\mathbf{\Sigma} \in \R^{d \times d}$ によって定まる確率分布であり，
  密度関数
  \[
    \rho(x) = \frac{1}{(2\pi)^{d/2} \det(\mathbf{\Sigma})^{1/2}}
    \exp\!\left( -\frac{1}{2} (x - \mathbf{m})^\top \mathbf{\Sigma}^{-1} (x - \mathbf{m}) \right)
  \]
  を持つ．
\end{definition}

\begin{theorem}{ガウス分布間の Wasserstein 距離}{wasserstein-gaussian}
  $\alpha = \mathcal{N}(\mathbf{m}_\alpha, \mathbf{\Sigma}_\alpha)$，
  $\beta = \mathcal{N}(\mathbf{m}_\beta, \mathbf{\Sigma}_\beta)$ に対して，
  \[
    \Wass_2(\alpha, \beta)^2
    = \norm{\mathbf{m}_\alpha - \mathbf{m}_\beta}_2^2
    + \mathcal{B}(\mathbf{\Sigma}_\alpha, \mathbf{\Sigma}_\beta)^2
  \]
  が成り立つ．ここで $\mathcal{B}$ は\textbf{Bures 距離}であり，
  \[
    \mathcal{B}(\mathbf{\Sigma}_\alpha, \mathbf{\Sigma}_\beta)^2
    \defeq \tr\!\left(
    \mathbf{\Sigma}_\alpha + \mathbf{\Sigma}_\beta
    - 2 \left( \mathbf{\Sigma}_\alpha^{1/2} \mathbf{\Sigma}_\beta \mathbf{\Sigma}_\alpha^{1/2} \right)^{1/2}
    \right)
  \]
  で定義される．また，最適な Monge 写像は線形写像
  \[
    T(x) = \mathbf{m}_\beta + \mathbf{A}(x - \mathbf{m}_\alpha), \quad
    \mathbf{A} = \mathbf{\Sigma}_\alpha^{-1/2}
    \left( \mathbf{\Sigma}_\alpha^{1/2} \mathbf{\Sigma}_\beta \mathbf{\Sigma}_\alpha^{1/2} \right)^{1/2}
    \mathbf{\Sigma}_\alpha^{-1/2}
  \]
  で与えられる．
\end{theorem}

\begin{remark}{}{gaussian-remark}
  ガウス分布間の Wasserstein 距離は，平均の差のユークリッドノルムと
  共分散行列の差を測る Bures 距離に分解される．
  1次元（$d = 1$）では $\mathbf{\Sigma}_\alpha = \sigma_\alpha^2$，
  $\mathbf{\Sigma}_\beta = \sigma_\beta^2$ であるから，
  \[
    \Wass_2(\alpha, \beta)^2 = (m_\alpha - m_\beta)^2 + (\sigma_\alpha - \sigma_\beta)^2
  \]
  となり，平均-標準偏差平面上のユークリッド距離に一致する．
\end{remark}

\section{Brenier の定理}
\label{sec:brenier}

Monge 問題と Kantorovich 問題の関係について，
二次コストの場合に両者が同値であるという深い結果を述べる．

\begin{definition}{凸関数}{convex-function}
  $\R^d$ 上の関数 $\varphi \colon \R^d \to \R \cup \{+\infty\}$ が\textbf{凸}（convex）であるとは，
  任意の $x, y \in \R^d$ と $t \in [0, 1]$ に対して
  \[
    \varphi(tx + (1-t)y) \leq t\varphi(x) + (1-t)\varphi(y)
  \]
  が成り立つことをいう．
\end{definition}

\begin{definition}{凸関数の勾配}{gradient-convex}
  凸関数 $\varphi \colon \R^d \to \R$ が点 $x$ で微分可能であるとき，
  その\textbf{勾配}（gradient）$\nabla\varphi(x) \in \R^d$ は
  \[
    \varphi(y) \geq \varphi(x) + \inner{\nabla\varphi(x)}{y - x}
    \qquad (\forall\, y \in \R^d)
  \]
  を満たす．凸関数はほとんどいたるところ微分可能である（Rademacher の定理）．
\end{definition}

\begin{theorem}{Brenier の定理}{brenier}
  $\alpha, \beta \in \Mm_+^1(\R^d)$ をコスト $c(x, y) = \norm{x - y}_2^2$ に対する
  最適輸送問題の入力とする．
  $\alpha$ がルベーグ測度に関して密度を持つ（すなわち $\alpha \ll \d x$）と仮定する．
  このとき，以下が成り立つ：
  \begin{enumerate}
    \item 最適なカップリング $\pi^*$ は\textbf{一意的}であり，
      Monge 写像によって誘導される：$\pi^* = (\operatorname{Id}, T)\pushforward \alpha$．
    \item 最適な Monge 写像 $T$ は凸関数 $\varphi \colon \R^d \to \R$ の勾配として表される：
      $T(x) = \nabla\varphi(x)$（$\alpha$-a.e.）．
    \item この $\varphi$ は $\alpha$-a.e.\ の意味で一意的であり，
      $(\nabla\varphi)\pushforward \alpha = \beta$ を満たす．
  \end{enumerate}
\end{theorem}

\begin{remark}{}{brenier-significance}
  Brenier の定理の意義は以下の点にある：
  \begin{itemize}
    \item Kantorovich 緩和が\textbf{タイト}である：
      最適カップリングは Monge 写像のグラフ上に集中しており，
      質量の分割は起こらない．
    \item 最適写像が凸関数の勾配で特徴づけられるため，
      最適輸送と\textbf{凸解析}の間に深い関係が成立する．
    \item ソース測度 $\alpha$ が密度を持つという条件は本質的である．
      $\alpha$ が離散測度の場合，最適な Monge 写像が存在するとは限らない
      （\ref{subsec:monge}~節の例~\ref{ex:monge-nonexistence} を参照）．
  \end{itemize}
\end{remark}

\begin{remark}{Monge--Amp\`{e}re 方程式}{monge-ampere}
  $\alpha, \beta$ がともに密度 $\rho_\alpha, \rho_\beta$ を持つとき，
  Brenier の定理における凸関数 $\varphi$ は
  \textbf{Monge--Amp\`{e}re 方程式}
  \[
    \det(\partial^2 \varphi(x)) \, \rho_\beta(\nabla\varphi(x)) = \rho_\alpha(x)
  \]
  を満たす．ここで $\partial^2 \varphi(x) \in \R^{d \times d}$ は $\varphi$ のヘッセ行列である．
  これは変位 $\varphi = \frac{\norm{\cdot}^2}{2} + \varepsilon\psi$ が小さいとき
  $\det(\partial^2\varphi) \approx 1 + \varepsilon\Delta\psi$ と線形化されることから，
  Monge--Amp\`{e}re 作用素は非線形の退化ラプラシアンとみなせる．
  一般の測度に対しては古典的な意味の解が存在しないため，
  Alexandrov 解の枠組みが必要となる．
\end{remark}
