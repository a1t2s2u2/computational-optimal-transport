\chapter{Optimal Transport の基礎理論}
\label{ch:seminar-01}

\begin{remark}{本章の設定}{sem-ch2-setup}
  本章では $\X, \Y$ を\textbf{完備可分な距離空間}（Polish 空間）とする．
  距離関数を $d$ と書く．$\X$ にはボレル $\sigma$-代数 $\Bb(\X)$ を入れて
  可測空間 $(\X, \Bb(\X))$ とみなし，
  $\X$ 上の確率測度全体を $\Mm_+^1(\X)$ と記す．$\Y$ についても同様．

  距離は本章で次のように使う：
  \begin{enumerate}
    \item[(i)] 距離から $\X, \Y$ の開集合・ボレル $\sigma$-代数・可測関数を定める．
    \item[(ii)] Wasserstein 距離の輸送コスト
      $c(x, y) = d(x, y)^p$ に\textbf{ground 距離} $d$ を用いる．
  \end{enumerate}
  離散版では離散測度の表示に現れる点集合
  $S \defeq \{x_1, \ldots, x_n\} \subset \X$ が
  有限集合となる．$S$ にどの距離 $d$ を入れた距離空間 $(S, d)$ も
  自動で Polish 空間になる：
  \begin{itemize}
    \item[] \textbf{完備性．} $n = 1$ なら任意の列は定値列ゆえ自明に収束する．
      以下 $n \geq 2$ とする．相異なる 2 点間の距離の最小値を
      $\delta \defeq \min_{i \neq j} d(x_i, x_j)$ とおく．
      これは有限個の正数の最小値なので $\delta > 0$．
      任意の Cauchy 列 $(y_k)_{k \geq 1} \subset S$ に対し，
      Cauchy 条件で $\varepsilon \defeq \delta / 2$ を取ると，
      ある $N \in \N$ が存在して
      $k, l \geq N \Longrightarrow d(y_k, y_l) < \delta / 2 < \delta$．
      一方 $S$ の相異なる 2 点の距離は $\delta$ 以上だから，
      $d(y_k, y_l) < \delta$ は $y_k = y_l$ を強制する．
      ゆえに列 $(y_k)$ は $N$ 以降すべて同一の点となり，
      その点に収束する．よって完備．
    \item[] \textbf{可分性．} $S$ 自身が有限（特に可算）集合であり，
      自身の中で稠密だから可分．
  \end{itemize}
  これは Cuturi のように $\X, \Y$ を単なる距離空間とする設定でも，
  離散版では本章の Polish 設定が自動成立することの確認である．
  本章の Polish 強化は連続側で必要となる一方，
  離散版には追加制約を課さない．
\end{remark}

\section{最適割当問題}
\label{sec:sem-assignment}

\begin{definition}{最適割当問題}{sem-assignment}
  $n \in \N$，置換 $\sigma \in \Perm(n)$，
  および行列 $\mathbf{C} \in \R_+^{n \times n}$ に対して，
  \[
    \min_{\sigma \in \Perm(n)} \frac{1}{n} \sum_{i=1}^{n} C_{i, \sigma(i)}
  \]
  を達成する $\sigma$ を求める問題を\textbf{最適割当問題}という．
\end{definition}

添字 $i$ から $j$ への\textbf{輸送コスト}を $C_{i,j}$ と解釈すれば，
$\sigma(i) = j$ は $i$ を $j$ に割り当てることを意味し，
目的関数 $\frac{1}{n}\sum_i C_{i, \sigma(i)}$ はその割当の平均輸送コストを表す．

\begin{example}{最適割当}{sem-assignment-cost-example}
  $n = 2$ とし，置換 $\sigma \in \Perm(2)$ と行列
  \[
    \mathbf{C} =
    \begin{pmatrix}
      1 & 4 \\
      2 & 1
    \end{pmatrix} \in \R_+^{2 \times 2}
  \]
  を考える．$\Perm(2) = \{(1,2),\, (2,1)\}$ の $2! = 2$ 通りの割当を以下に図示する．

  \begin{center}
  \begin{tikzpicture}[
    wk/.style={circle, draw=blue!60, fill=blue!10,
               minimum size=18pt, inner sep=1pt, font=\small},
    tk/.style={rectangle, rounded corners=1.5pt,
               draw=red!60, fill=red!10,
               minimum size=18pt, inner sep=1pt, font=\small},
    optarr/.style={-{Stealth[length=5pt]}, thick, green!50!black},
    arr/.style={-{Stealth[length=5pt]}, semithick, black!60, dashed},
    costlbl/.style={fill=white, inner sep=1pt, font=\footnotesize},
  ]
    \node[wk] (w1) at (0, 1.2) {1};
    \node[wk] (w2) at (3, 1.2) {2};
    \node[tk] (t1) at (1, -1.2) {1};
    \node[tk] (t2) at (4, -1.2) {2};

    % σ=(1,2): 1→1, 2→2（最適）
    \draw[optarr] (w1) -- node[costlbl, midway] {(a)\ $1$} (t1);
    \draw[optarr] (w2) -- node[costlbl, midway] {(a)\ $1$} (t2);
    % σ=(2,1): 1→2, 2→1
    \draw[arr] (w1) -- node[costlbl, pos=0.25] {(b)\ $4$} (t2);
    \draw[arr] (w2) -- node[costlbl, pos=0.25] {(b)\ $2$} (t1);
  \end{tikzpicture}
  \end{center}

  各割当のコストをまとめると，
  \begin{center}
  \begin{tabular}{ccc}
     & $\sigma$ & コスト $\frac{1}{n}\sum_i C_{i,\sigma(i)}$ \\\hline
    (a) & $(1,2)$ & $\frac{1}{2}(1 + 1) = \mathbf{1}$ \\
    (b) & $(2,1)$ & $\frac{1}{2}(4 + 2) = 3$ \\
  \end{tabular}
  \end{center}
  よって $\sigma = (1, 2)$ が最小コスト $1$ を達成し，最適である．
\end{example}

\begin{claim}{最適解の存在}{sem-assignment-existence}
  任意の $n \in \N$ と任意の $\mathbf{C} \in \R_+^{n \times n}$ に対して，
  最適割当問題の最小値は達成される．
\end{claim}

\begin{proof}
  $\Perm(n)$ は $|\Perm(n)| = n!$ の有限集合なので，集合
  \[
    \left\{ \frac{1}{n}\sum_{i=1}^{n} C_{i, \sigma(i)} \;\middle|\; \sigma \in \Perm(n) \right\}
  \]
  は高々 $n!$ 個の実数からなる空でない有限集合である．
  $\R$ の空でない有限部分集合は最小元を持つ（主張~\ref{clm:sem-finite-min}）
  ので，それを達成する $\sigma^* \in \Perm(n)$ が存在する．
\end{proof}

\begin{claim}{最適解が一意でない場合の存在}{sem-uniqueness}
  任意の $n \geq 2$ に対して，
  異なる $\sigma_1, \sigma_2 \in \Perm(n)$ がともに最小値を達成するような
  $\mathbf{C} \in \R_+^{n \times n}$ が存在する．
\end{claim}

\begin{proof}
  $C_{i,j} \defeq 1$（$i, j = 1, \ldots, n$）と定めると，任意の $\sigma \in \Perm(n)$ に対して
  $\frac{1}{n}\sum_{i=1}^{n} C_{i, \sigma(i)} = 1$．
  異なる $\sigma_1, \sigma_2 \in \Perm(n)$ を取れば，
  いずれも最小値 $1$ を達成する．
\end{proof}

\section{Monge 問題}
\label{sec:sem-monge}

\begin{definition}{Monge 問題}{sem-monge}
  $\X, \Y$ 上の確率測度 $\alpha \in \Mm_+^1(\X)$, $\beta \in \Mm_+^1(\Y)$ と
  可測関数 $c \colon \X \times \Y \to \R_+$ に対して，
  \textbf{Monge 問題}とは
  \[
    \inf_{T}
    \left\{
    \int_{\X} c(x, T(x)) \, \d\alpha(x)
    \;\middle|\;
    T \colon \X \to \Y \text{ は可測}, \;
    T\pushforward \alpha = \beta
    \right\}
  \]
  を求める問題である．
\end{definition}

$c(x, y)$ を $x$ から $y$ への\textbf{輸送コスト}と解釈すれば，
写像 $T \colon \X \to \Y$ は $x$ にある質量を $T(x)$ へ送ることを意味し，
目的関数 $\int_\X c(x, T(x)) \, \d\alpha(x)$ はその輸送の総コストを表す．

$\alpha, \beta$ がともに離散測度（定義~\ref{def:sem-discrete-measure}）の場合を考える．
定義により，
ある相異なる $x_1, \ldots, x_n \in \X$，相異なる $y_1, \ldots, y_m \in \Y$，
$\mathbf{a} = (a_1, \ldots, a_n) \in \R_{++}^n$，$\mathbf{b} = (b_1, \ldots, b_m) \in \R_{++}^m$
（$\sum_{i=1}^n a_i = \sum_{j=1}^m b_j = 1$）を用いて
\[
  \alpha = \sum_{i=1}^n a_i\, \delta_{x_i},
  \qquad
  \beta = \sum_{j=1}^m b_j\, \delta_{y_j}
\]
と書ける．行列 $\mathbf{C} \in \R_+^{n \times m}$ の成分を $C_{i,j} \defeq c(x_i, y_j)$ とおく．

\begin{claim}{離散 Monge 問題}{sem-discrete-monge}
  $\alpha, \beta$ を上記の表示の離散測度とする．
  $\{x_1, \ldots, x_n\}$ 上の写像 $T \colon \{x_1, \ldots, x_n\} \to \Y$ に対する押し出しを
  $T\pushforward \alpha \defeq \sum_{i=1}^{n} a_i\, \delta_{T(x_i)}$ と定め，
  \textbf{離散 Monge 問題}を
  \[
    \inf_{T}
    \left\{
    \sum_{i=1}^{n} a_i \, c(x_i, T(x_i))
    \;\middle|\;
    T \colon \{x_1, \ldots, x_n\} \to \Y, \;
    T\pushforward \alpha = \beta
    \right\}
  \]
  により定める（実行可能集合が空のときは下限を $+\infty$ と約束する）．
  このとき離散 Monge 問題と連続版 Monge 問題（定義~\ref{def:sem-monge}）の
  下限は一致する：連続版の可測写像 $T \colon \X \to \Y$ から制限
  $T|_{\{x_1, \ldots, x_n\}}$ により離散版の写像が得られ，
  この対応の下で目的関数値・押し出し制約が保存される．
  逆に離散版の任意の写像はある可測写像の制限として得られる．
\end{claim}

\begin{proof}
  目的関数 (a) と押し出し制約 (b) が $T$ の値 $T(x_1), \ldots, T(x_n)$
  のみに依存することを示し，さらに任意の値の組
  $(y_1, \ldots, y_n) \in \Y^n$ がある可測写像 $\widetilde T \colon \X \to \Y$ の
  $x_1, \ldots, x_n$ における値として実現されること (c) を示す．

  \textbf{(a) 目的関数の書き換え．}
  $\alpha = \sum_i a_i\, \delta_{x_i}$ を代入し，
  測度に関する積分の線形性（命題~\ref{prop:sem-measure-linearity}）と
  Dirac 測度に対する積分（主張~\ref{clm:sem-dirac-integral}）を順に適用すると
  \begin{align*}
    \int_\X c(x, T(x)) \, \d\alpha(x)
    &= \int_\X c(x, T(x)) \, \d\!\Bigl(\textstyle\sum_{i=1}^{n} a_i\, \delta_{x_i}\Bigr)(x)
      && \text{（$\alpha$ の表示）} \\
    &= \sum_{i=1}^{n} a_i \int_\X c(x, T(x)) \, \d\delta_{x_i}(x)
      && \text{（命題~\ref{prop:sem-measure-linearity}）} \\
    &= \sum_{i=1}^{n} a_i \, c(x_i, T(x_i))
      && \text{（主張~\ref{clm:sem-dirac-integral}）}.
  \end{align*}

  \textbf{(b) 押し出し制約の書き換え．}
  離散測度の押し出し（命題~\ref{prop:sem-dirac-pushforward} (ii)）より
  \[
    T\pushforward \alpha
    = T\pushforward \Bigl(\sum_{i=1}^n a_i\, \delta_{x_i}\Bigr)
    = \sum_{i=1}^n a_i\, \delta_{T(x_i)}
  \]
  であり，等式 $T\pushforward \alpha = \beta$ もまた
  $T(x_1), \ldots, T(x_n)$ のみに依存する．

  以上 (a)(b) より制限写像 $T \mapsto T|_{\{x_1, \ldots, x_n\}}$ は，
  連続版の実行可能写像を離散版の実行可能写像に目的関数値を保存して写す．
  残るはこの制限写像が全射であること，
  すなわち $x_1, \ldots, x_n$ における任意の値の組が
  可測写像から実現されることである．

  \textbf{(c) 可測延長の存在．}
  任意の値の組 $(y_1, \ldots, y_n) \in \Y^n$ に対し，
  $y_\ast \in \Y$ を一つ任意に固定して
  \[
    \widetilde T(x) \defeq
    \begin{cases}
      y_i & (x = x_i,\; i = 1, \ldots, n), \\
      y_\ast & (x \notin \{x_1, \ldots, x_n\})
    \end{cases}
  \]
  と定める．$\X$ は Polish ゆえ Hausdorff
  （注釈~\ref{rem:sem-ch2-setup}, 主張~\ref{clm:sem-metric-hausdorff}）であり，
  単点集合 $\{x_i\}$ は閉集合，したがって $\Bb(\X)$ に属する．
  $\widetilde T$ の値域は有限集合 $V \defeq \{y_\ast, y_1, \ldots, y_n\}$ であり，
  各 $y \in V$ に対し $\widetilde T^{-1}(\{y\})$ は
  $\{x_i \mid y_i = y\} \in \Bb(\X)$ あるいはそれと
  $\X \setminus \{x_1, \ldots, x_n\} \in \Bb(\X)$ との和集合に一致するから，
  ともにボレル集合．任意の $B \in \Bb(\Y)$ に対し
  $\widetilde T^{-1}(B) = \bigcup_{y \in B \cap V} \widetilde T^{-1}(\{y\})$
  はボレル集合の有限和ゆえボレル．
  したがって $\widetilde T$ は可測で $\widetilde T(x_i) = y_i$ をみたす．
\end{proof}

\paragraph{最適割当問題への帰着．}
特に $n = m$ かつ $a_i = b_j = 1/n$（すべての砂山・行き先の質量が等しい）とする．
このとき離散測度の押し出し（命題~\ref{prop:sem-dirac-pushforward} (ii)）より
\[
  T\pushforward \alpha
  = \frac{1}{n} \sum_{i=1}^{n} \delta_{T(x_i)}.
\]
これが $\beta = \frac{1}{n}\sum_{j=1}^{n} \delta_{y_j}$ に一致するには，
多重集合として
$\{T(x_1), \ldots, T(x_n)\} = \{y_1, \ldots, y_n\}$
でなければならない．したがって $T$ は $\{x_i\}$ から $\{y_j\}$ への全単射，
すなわちある置換 $\sigma \in \Perm(n)$ により
$T(x_i) = y_{\sigma(i)}$ と書け，離散 Monge 問題は最適割当問題に帰着する．

\subsection{Monge 問題の本質的困難}

\subsubsection{困難1：実行可能集合の非凸性}

$\X = \Y = \R$ とする．
このとき可測写像 $T \colon \R \to \R$ の全体は，点ごとの和・スカラー倍
$(tT_1 + (1-t)T_2)(x) \defeq t\, T_1(x) + (1-t)\, T_2(x)$
によりベクトル空間をなす．この空間の部分集合としての
Monge 問題の実行可能集合
\[
  \mathcal{T}(\alpha, \beta)
  \defeq \left\{ T \colon \X \to \Y \;\middle|\;
  T \text{ は可測},\; T\pushforward \alpha = \beta \right\}
\]
は一般に\textbf{凸でない}．

\begin{example}{実行可能集合の非凸性}{sem-nonconvex}
  $\alpha = \beta = \tfrac{1}{2}\delta_{-1} + \tfrac{1}{2}\delta_1$ とし，
  \[
    T_1(x) = x,\qquad T_2(x) = -x,\qquad t = \tfrac{1}{2}
  \]
  をとる．$\alpha$ の定義を代入し，命題~\ref{prop:sem-dirac-pushforward} (ii) を適用すると
  \begin{align*}
    {T_1}\pushforward \alpha
    &= {T_1}\pushforward \bigl(\tfrac{1}{2}\delta_{-1} + \tfrac{1}{2}\delta_{1}\bigr)
      && \text{（$\alpha$ の定義）} \\
    &= \tfrac{1}{2}\delta_{T_1(-1)} + \tfrac{1}{2}\delta_{T_1(1)}
      && \text{（命題~\ref{prop:sem-dirac-pushforward} (ii)）} \\
    &= \tfrac{1}{2}\delta_{-1} + \tfrac{1}{2}\delta_{1} = \beta
      && \text{（$T_1(x) = x$ より）}.
  \end{align*}
  同様に
  \begin{align*}
    {T_2}\pushforward \alpha
    &= \tfrac{1}{2}\delta_{T_2(-1)} + \tfrac{1}{2}\delta_{T_2(1)}
    = \tfrac{1}{2}\delta_{1} + \tfrac{1}{2}\delta_{-1}
    = \beta
    && \text{（$T_2(x) = -x$ より）}.
  \end{align*}
  よって $T_1, T_2 \in \mathcal{T}(\alpha, \beta)$ である．

  一方，$\bar{T} = \tfrac{1}{2}T_1 + \tfrac{1}{2}T_2$ は
  $\bar{T}(x) = \tfrac{1}{2}x + \tfrac{1}{2}(-x) = 0$
  すなわち $0$ を値にとる定値写像であり，
  $\bar{T}\pushforward\alpha = \delta_0 \neq \beta$．
  よって $\bar{T} \notin \mathcal{T}(\alpha, \beta)$ であり，
  $\mathcal{T}(\alpha, \beta)$ は凸でない．
\end{example}

\subsubsection{困難2：解の非存在}

\begin{example}{Monge 写像が存在しない場合}{sem-monge-noexist}
  $\alpha = \delta_0$，$\beta = \tfrac{1}{2}\delta_{-1} + \tfrac{1}{2}\delta_1$
  を考える．

  任意の可測写像 $T \colon \X \to \Y$ に対して，
  命題~\ref{prop:sem-dirac-pushforward} (i) より
  $T\pushforward \delta_0 = \delta_{T(0)}$ である．

  したがって $T \in \mathcal{T}(\delta_0, \beta)$ となるには
  $\delta_{T(0)} = \beta$ が必要だが，
  $A = \{-1\}$ で $\delta_{T(0)}(\{-1\}) \in \{0, 1\}$ に対し
  $\beta(\{-1\}) = \tfrac{1}{2}$ なので，どんな $T(0)$ でも一致しない．
  よって条件を満たす $T$ は存在せず $\mathcal{T}(\delta_0, \beta) = \emptyset$ である（下図）．

\section{Kantorovich 緩和}
\label{sec:sem-kantorovich}

\subsection{カップリング}

\begin{definition}{カップリング}{sem-coupling}
  $\X, \Y$ 上の確率測度 $\alpha \in \Mm_+^1(\X)$, $\beta \in \Mm_+^1(\Y)$
  に対して，$\X \times \Y$ 上の確率測度のうち周辺測度がそれぞれ
  $\alpha, \beta$ に一致するものの集合
  \[
    \Couplings(\alpha, \beta) \defeq
    \left\{ \pi \in \Mm_+^1(\X \times \Y) \;\middle|\;
    (P_\X)\pushforward \pi = \alpha, \;
    (P_\Y)\pushforward \pi = \beta \right\}
  \]
  を $\alpha$ と $\beta$ の\textbf{カップリング}の集合という．
  ここで $P_\X \colon \X \times \Y \to \X$, $P_\Y \colon \X \times \Y \to \Y$
  はそれぞれ第 1・第 2 成分への射影である．
\end{definition}

離散測度
$\alpha = \sum_i a_i \delta_{x_i}$，
$\beta = \sum_j b_j \delta_{y_j}$
の場合，カップリングは
$\pi = \sum_{i,j} P_{i,j} \delta_{(x_i, y_j)}$
の形をとる．
周辺条件 $(P_\X)\pushforward \pi = \alpha$ は
$\sum_j P_{i,j} = a_i$，すなわち $\mathbf{P}\ones_m = \mathbf{a}$ に帰着し，
$(P_\Y)\pushforward \pi = \beta$ は $\mathbf{P}^\top \ones_n = \mathbf{b}$ に帰着する．
したがって\textbf{離散カップリング}の集合は
\[
  \CouplingsD(\mathbf{a}, \mathbf{b}) \defeq
  \left\{ \mathbf{P} \in \R_+^{n \times m} \;\middle|\;
  \mathbf{P} \ones_m = \mathbf{a}, \;
  \mathbf{P}^\top \ones_n = \mathbf{b} \right\}
\]
で与えられる．
$P_{i,j}$ は「輸送元のビン $i$ から輸送先のビン $j$ へ送られる質量」を表す．
$\CouplingsD(\mathbf{a}, \mathbf{b})$ は $n + m$ 本の等式制約で定まる
$\R_+^{n \times m}$ の有界部分集合であるから，
凸多面体（定義~\ref{def:sem-polytope}）をなす．

\subsection{Kantorovich 問題}

\begin{definition}{Kantorovich 問題}{sem-kantorovich}
  $\X, \Y$ 上の確率測度 $\alpha \in \Mm_+^1(\X)$, $\beta \in \Mm_+^1(\Y)$ と
  可測関数 $c \colon \X \times \Y \to \R_+$ に対して，
  \textbf{Kantorovich 問題}とは
  \[
    \MK_c(\alpha, \beta) \defeq
    \inf_{\pi \in \Couplings(\alpha, \beta)}
    \int_{\X \times \Y} c(x, y) \, \d\pi(x, y)
  \]
  を求める問題である．
\end{definition}

離散カップリング $\pi = \sum_{i,j} P_{i,j} \delta_{(x_i, y_j)}$ の場合，
$\int c \, \d\pi = \sum_{i,j} C_{i,j} P_{i,j} = \inner{\mathbf{C}}{\mathbf{P}}$
となるから，\textbf{離散 Kantorovich 問題}は
\[
  \MKD_{\mathbf{C}}(\mathbf{a}, \mathbf{b}) \defeq
  \min_{\mathbf{P} \in \CouplingsD(\mathbf{a}, \mathbf{b})}
  \inner{\mathbf{C}}{\mathbf{P}}
\]
と書ける．
これは $nm$ 個の変数と $n + m$ 本の等式制約を持つ有限次元の最適化問題である．

\begin{example}{Monge 写像が存在しない場合の解消}{sem-kantorovich-resolve}
  例~\ref{ex:sem-monge-noexist} では
  $\alpha = \delta_0$，$\beta = \frac{1}{2}\delta_{-1} + \frac{1}{2}\delta_1$
  に対して Monge 写像が存在しないことを見た．
  Kantorovich 問題（定義~\ref{def:sem-kantorovich}）の枠組みで考え直す．

  離散表現は $\mathbf{a} = (1)$，$\mathbf{b} = (1/2,\; 1/2)$ であり，
  カップリング $\mathbf{P} \in \R_+^{1 \times 2}$ は
  $\mathbf{P}\ones_2 = 1$，$\mathbf{P}^\top \ones_1 = \mathbf{b}$
  を満たす必要がある．行が1つしかないため $P_{1,j} = b_j$ が強制され，
  \[
    \mathbf{P} = \begin{pmatrix} 1/2 & 1/2 \end{pmatrix}
  \]
  が唯一の実行可能カップリングである．
  コスト $c(x,y) = |x - y|$ から
  $\mathbf{C} = (|0-(-1)|,\; |0-1|) = (1,\; 1)$ であり，
  最適値は
  \[
    \MKD_{\mathbf{C}}(\mathbf{a}, \mathbf{b})
    = \inner{\mathbf{C}}{\mathbf{P}}
    = 1 \cdot \tfrac{1}{2} + 1 \cdot \tfrac{1}{2} = 1.
  \]
  Monge 問題と異なり，質量分割を許す Kantorovich の枠組みでは
  実行可能解が存在し，最適値が定まる．
\end{example}

\subsection{離散版の解の存在}

離散 Kantorovich 問題の実行可能集合 $\CouplingsD(\mathbf{a}, \mathbf{b})$ は
凸多面体（定義~\ref{def:sem-polytope}）をなす．
有界閉集合上の連続関数は最小値を達成するから，最適解が存在する．

\begin{claim}{離散 Kantorovich 問題の解の存在}{sem-discrete-kantorovich-existence}
  任意の $\mathbf{a} \in \R_{++}^n$，$\mathbf{b} \in \R_{++}^m$
  （$\sum_i a_i = \sum_j b_j = 1$）と $\mathbf{C} \in \R_+^{n \times m}$ に対して，
  離散 Kantorovich 問題は最適解
  $\mathbf{P}^* \in \CouplingsD(\mathbf{a}, \mathbf{b})$ をもつ：
  \[
    \mathbf{P}^* \in \argmin_{\mathbf{P} \in \CouplingsD(\mathbf{a}, \mathbf{b})}
    \inner{\mathbf{C}}{\mathbf{P}},
    \qquad
    \MKD_{\mathbf{C}}(\mathbf{a}, \mathbf{b}) = \inner{\mathbf{C}}{\mathbf{P}^*}.
  \]
\end{claim}

\begin{proof}
  以下の3点を示せばよい：
  (i) $\CouplingsD(\mathbf{a}, \mathbf{b}) \neq \emptyset$，
  (ii) $\CouplingsD(\mathbf{a}, \mathbf{b}) \subset \R^{n \times m}$ はコンパクト，
  (iii) 目的関数 $\mathbf{P} \mapsto \inner{\mathbf{C}}{\mathbf{P}}$ は連続．

  \textbf{(i) 非空性．}
  積測度に相当する $\mathbf{P}_0 \defeq \mathbf{a}\mathbf{b}^\top$，すなわち
  $(P_0)_{i,j} = a_i b_j$ をとると，
  $(\mathbf{P}_0 \ones_m)_i = a_i \sum_j b_j = a_i$，
  $(\mathbf{P}_0^\top \ones_n)_j = b_j \sum_i a_i = b_j$
  ゆえ $\mathbf{P}_0 \in \CouplingsD(\mathbf{a}, \mathbf{b})$．

  \textbf{(ii) コンパクト性．}
  $\CouplingsD(\mathbf{a}, \mathbf{b})$ は線形等式
  $\mathbf{P}\ones_m = \mathbf{a}$, $\mathbf{P}^\top \ones_n = \mathbf{b}$
  と非負性 $\mathbf{P} \geq 0$ で定まる $\R^{n \times m}$ の閉集合．
  さらに各成分は $0 \leq P_{i,j} \leq \min(a_i, b_j) \leq 1$ をみたすから有界．
  $\R^{n \times m}$ における有界閉集合はコンパクト．

  \textbf{(iii) 連続性．}
  $\mathbf{P} \mapsto \inner{\mathbf{C}}{\mathbf{P}} = \sum_{i,j} C_{i,j} P_{i,j}$
  は線形写像であり，有限次元線形空間上で連続．

  $\R^{n \times m}$ 上の連続関数は空でないコンパクト集合上で最小値を達成する
  （Weierstrass の最大値の定理）．したがって最適解 $\mathbf{P}^*$ が存在する．
\end{proof}

\begin{remark}{最適解の一意性は一般に成り立たない}{sem-discrete-nonunique}
  Claim~\ref{clm:sem-uniqueness} で見たとおり，
  $\mathbf{C}$ がすべて等しい等の縮退状況では，
  最適値を達成する $\mathbf{P}^*$ は一意ではない．
  実際，目的関数は線形であり実行可能集合は凸多面体であるから，
  最適解の集合は $\CouplingsD(\mathbf{a}, \mathbf{b})$ の面の1つをなす．
\end{remark}

連続版（Polish 空間設定）の解の存在は Prokhorov の定理と
コスト関数の下半連続性を用いて示されるが，本稿では離散版に焦点を当てる．
