\chapter{Optimal Transport の基礎理論}
\label{ch:seminar-01}

\begin{remark}{本章の設定}{sem-ch2-setup}
  本章では $\X, \Y$ を\textbf{完備可分な距離空間}（Polish 空間）とする．
  距離関数を $d$ と書く．$\X$ にはボレル $\sigma$-代数 $\Bb(\X)$ を入れて
  可測空間 $(\X, \Bb(\X))$ とみなし，
  $\X$ 上の確率測度全体を $\Mm_+^1(\X)$ と記す．$\Y$ についても同様．

  距離は本章で次のように使う：
  \begin{enumerate}
    \item[(i)] 距離から $\X, \Y$ の位相・ボレル $\sigma$-代数・可測関数を定める．
    \item[(ii)] Wasserstein 距離の輸送コスト
      $c(x, y) = d(x, y)^p$ に\textbf{ground 距離} $d$ を用いる．
  \end{enumerate}
  離散版では確率測度の台 $S \defeq \{x_1, \ldots, x_n\} \subseteq \X$ が
  有限集合となる．$S$ にどの距離 $d$ を入れた距離空間 $(S, d)$ も
  自動で Polish 空間になる：
  \begin{itemize}
    \item[] \textbf{完備性．} $n = 1$ なら任意の列は定値列ゆえ自明に収束する．
      以下 $n \geq 2$ とする．相異なる 2 点間の距離の最小値を
      $\delta \defeq \min_{i \neq j} d(x_i, x_j)$ とおく．
      これは有限個の正数の最小値なので $\delta > 0$．
      任意の Cauchy 列 $(y_k)_{k \geq 1} \subseteq S$ に対し，
      Cauchy 条件で $\varepsilon \defeq \delta / 2$ を取ると，
      ある $N \in \N$ が存在して
      $k, l \geq N \Longrightarrow d(y_k, y_l) < \delta / 2 < \delta$．
      一方 $S$ の相異なる 2 点の距離は $\delta$ 以上だから，
      $d(y_k, y_l) < \delta$ は $y_k = y_l$ を強制する．
      ゆえに列 $(y_k)$ は $N$ 以降すべて同一の点となり，
      その点に収束する．よって完備．
    \item[] \textbf{可分性．} $S$ 自身が有限（特に可算）集合であり，
      自身の中で稠密だから可分．
  \end{itemize}
  これは Cuturi のように $\X, \Y$ を単なる距離空間とする設定でも，
  離散版では本章の Polish 設定が自動成立することの確認である．
  本章の Polish 強化は連続側で必要となる一方，
  離散版には追加制約を課さない．
\end{remark}

\section{最適割当問題}
\label{sec:sem-assignment}

\begin{definition}{最適割当問題（optimal assignment problem）}{sem-assignment}
  $n \in \N$，置換 $\sigma \in \Perm(n)$，
  および行列 $\mathbf{C} \in \R_+^{n \times n}$ に対して，
  \[
    \min_{\sigma \in \Perm(n)} \frac{1}{n} \sum_{i=1}^{n} C_{i, \sigma(i)}
  \]
  を達成する $\sigma$ を求める問題を\textbf{最適割当問題}という．
\end{definition}

添字 $i$ から $j$ への\textbf{輸送コスト}を $C_{i,j}$ と解釈すれば，
$\sigma(i) = j$ は $i$ を $j$ に割り当てることを意味し，
目的関数 $\frac{1}{n}\sum_i C_{i, \sigma(i)}$ はその割当の平均輸送コストを表す．

\begin{example}{最適割当}{sem-assignment-cost-example}
  $n = 2$ とし，置換 $\sigma \in \Perm(2)$ と行列
  \[
    \mathbf{C} =
    \begin{pmatrix}
      1 & 4 \\
      2 & 1
    \end{pmatrix} \in \R_+^{2 \times 2}
  \]
  を考える．$\Perm(2) = \{(1,2),\, (2,1)\}$ の $2! = 2$ 通りの割当を以下に図示する
  （$C_{i,j}$ は砂山 $i$ と行き先 $j$ の距離に対応）．

\begin{claim}{最適解の存在}{sem-assignment-existence}
  任意の $n \in \N$ と任意の $\mathbf{C} \in \R_+^{n \times n}$ に対して，
  最適割当問題の最小値は達成される．
\end{claim}

\begin{proof}
  $\Perm(n)$ は $|\Perm(n)| = n!$ の有限集合なので，集合
  \[
    \left\{ \frac{1}{n}\sum_{i=1}^{n} C_{i, \sigma(i)} \;\middle|\; \sigma \in \Perm(n) \right\}
  \]
  は高々 $n!$ 個の実数からなる空でない有限集合である．
  主張~\ref{clm:sem-finite-min} より最小元を持つので，
  それを達成する $\sigma^* \in \Perm(n)$ が存在する．
\end{proof}

\begin{claim}{最適解が一意でない場合の存在}{sem-uniqueness}
  任意の $n \geq 2$ に対して，
  異なる $\sigma_1, \sigma_2 \in \Perm(n)$ がともに最小値を達成するような
  $\mathbf{C} \in \R_+^{n \times n}$ が存在する．
\end{claim}

\begin{proof}
  $C_{i,j} \defeq 1$（$i, j = 1, \ldots, n$）と定めると，任意の $\sigma \in \Perm(n)$ に対して
  $\frac{1}{n}\sum_{i=1}^{n} C_{i, \sigma(i)} = 1$．
  異なる $\sigma_1, \sigma_2 \in \Perm(n)$ を取れば，
  いずれも最小値 $1$ を達成する．
\end{proof}

\section{Monge 問題}
\label{sec:sem-monge}

\begin{definition}{Monge 問題（Monge problem）}{sem-monge}
  $\X, \Y$ 上の確率測度 $\alpha \in \Mm_+^1(\X)$, $\beta \in \Mm_+^1(\Y)$ と
  可測関数 $c \colon \X \times \Y \to \R_+$
  （$c(x, y)$ は $x$ から $y$ への\textbf{輸送コスト}）に対して，
  \textbf{Monge 問題}とは
  \[
    \inf_{T}
    \left\{
    \int_{\X} c(x, T(x)) \, \d\alpha(x)
    \;\middle|\;
    T \colon \X \to \Y \text{ は可測}, \;
    T\pushforward \alpha = \beta
    \right\}
  \]
  を求める問題である．
\end{definition}

離散測度の場合を考える．すなわち
\[
  \alpha = \sum_{i=1}^n a_i\, \delta_{x_i},
  \qquad
  \beta = \sum_{j=1}^m b_j\, \delta_{y_j}
\]
とし，ここで $x_1, \ldots, x_n \in \X$ は砂山の位置，$y_1, \ldots, y_m \in \Y$ は行き先の位置，
$\mathbf{a} = (a_1, \ldots, a_n) \in \R_+^n$，$\mathbf{b} = (b_1, \ldots, b_m) \in \R_+^m$
はそれぞれ各砂山の質量・各行き先で積み上げる量を表す
（$\alpha, \beta$ が確率測度であることから $\sum_i a_i = \sum_j b_j = 1$）．
行列 $\mathbf{C} \in \R_+^{n \times m}$ の成分を $C_{i,j} \defeq c(x_i, y_j)$ とおく．
このとき Monge 問題の目的関数は
\begin{align*}
  \int_\X c(x, T(x)) \, \d\alpha(x)
  &= \int_\X c(x, T(x)) \, \d\!\Bigl(\textstyle\sum_{i=1}^{n} a_i\, \delta_{x_i}\Bigr)(x)
    && \text{（$\alpha$ を代入）} \\
  &= \sum_{i=1}^{n} a_i \int_\X c(x, T(x)) \, \d\delta_{x_i}(x)
    && \text{（命題~\ref{prop:sem-measure-linearity}）} \\
  &= \sum_{i=1}^{n} a_i \, c(x_i, T(x_i))
    && \text{（Dirac 積分 $\int f\, \d\delta_{x_i} = f(x_i)$）}
\end{align*}
と書き換えられる．したがって\textbf{離散 Monge 問題}は
\[
  \min_{T}
  \left\{
  \sum_{i=1}^{n} a_i \, c(x_i, T(x_i))
  \;\middle|\;
  T\pushforward \alpha = \beta
  \right\}
\]
と書ける．

\paragraph{最適割当問題への帰着．}
特に $n = m$ かつ $a_i = b_j = 1/n$（すべての砂山・行き先の質量が等しい）とする．
このとき離散測度の押し出し（命題~\ref{prop:sem-dirac-pushforward} (ii)）より
\[
  T\pushforward \alpha
  = \frac{1}{n} \sum_{i=1}^{n} \delta_{T(x_i)}.
\]
これが $\beta = \frac{1}{n}\sum_{j=1}^{n} \delta_{y_j}$ に一致するには，
多重集合として
$\{T(x_1), \ldots, T(x_n)\} = \{y_1, \ldots, y_n\}$
でなければならない．したがって $T$ は $\{x_i\}$ から $\{y_j\}$ への全単射，
すなわちある置換 $\sigma \in \Perm(n)$ により
$T(x_i) = y_{\sigma(i)}$ と書け，離散 Monge 問題は最適割当問題に帰着する．

\subsection{Monge 問題の本質的困難}

\subsubsection{困難1：実行可能集合の非凸性}

$\X = \Y = \R$ とする．
このとき可測写像 $T \colon \R \to \R$ の全体は，点ごとの和・スカラー倍
$(tT_1 + (1-t)T_2)(x) \defeq t\, T_1(x) + (1-t)\, T_2(x)$
によりベクトル空間をなす．この空間の部分集合としての
Monge 問題の実行可能集合
\[
  \mathcal{T}(\alpha, \beta)
  \defeq \left\{ T \colon \X \to \Y \;\middle|\;
  T \text{ は可測},\; T\pushforward \alpha = \beta \right\}
\]
は一般に\textbf{凸でない}（凸集合の定義は定義~\ref{def:sem-convex}）．

\begin{example}{実行可能集合の非凸性}{sem-nonconvex}
  $\alpha = \beta = \tfrac{1}{2}\delta_{-1} + \tfrac{1}{2}\delta_1$ とし，
  \[
    T_1(x) = x,\qquad T_2(x) = -x,\qquad t = \tfrac{1}{2}
  \]
  をとる．$\alpha$ の定義を代入し，命題~\ref{prop:sem-dirac-pushforward} (ii) を適用すると
  \begin{align*}
    {T_1}\pushforward \alpha
    &= {T_1}\pushforward \bigl(\tfrac{1}{2}\delta_{-1} + \tfrac{1}{2}\delta_{1}\bigr)
      && \text{（$\alpha$ の定義）} \\
    &= \tfrac{1}{2}\delta_{T_1(-1)} + \tfrac{1}{2}\delta_{T_1(1)}
      && \text{（命題~\ref{prop:sem-dirac-pushforward} (ii)）} \\
    &= \tfrac{1}{2}\delta_{-1} + \tfrac{1}{2}\delta_{1} = \beta
      && \text{（$T_1(x) = x$ より）}.
  \end{align*}
  同様に
  \begin{align*}
    {T_2}\pushforward \alpha
    &= \tfrac{1}{2}\delta_{T_2(-1)} + \tfrac{1}{2}\delta_{T_2(1)}
    = \tfrac{1}{2}\delta_{1} + \tfrac{1}{2}\delta_{-1}
    = \beta
    && \text{（$T_2(x) = -x$ より）}.
  \end{align*}
  よって $T_1, T_2 \in \mathcal{T}(\alpha, \beta)$ である．

  一方，$\bar{T} = \tfrac{1}{2}T_1 + \tfrac{1}{2}T_2$ は
  $\bar{T}(x) = \tfrac{1}{2}x + \tfrac{1}{2}(-x) = 0$
  すなわち $0$ を値にとる定値写像であり，
  $\bar{T}\pushforward\alpha = \delta_0 \neq \beta$．
  よって $\bar{T} \notin \mathcal{T}(\alpha, \beta)$ であり，
  $\mathcal{T}(\alpha, \beta)$ は凸でない．
\end{example}

\subsubsection{困難2：解の非存在}

\begin{example}{Monge 写像が存在しない場合}{sem-monge-noexist}
  $\alpha = \delta_0$，$\beta = \tfrac{1}{2}\delta_{-1} + \tfrac{1}{2}\delta_1$
  を考える．

  任意の可測写像 $T \colon \X \to \Y$ に対して，
  命題~\ref{prop:sem-dirac-pushforward} (i) より
  $T\pushforward \delta_0 = \delta_{T(0)}$ である．

  したがって $T \in \mathcal{T}(\delta_0, \beta)$ となるには
  $\delta_{T(0)} = \beta$ が必要だが，
  $A = \{-1\}$ で $\delta_{T(0)}(\{-1\}) \in \{0, 1\}$ に対し
  $\beta(\{-1\}) = \tfrac{1}{2}$ なので，どんな $T(0)$ でも一致しない．
  よって条件を満たす $T$ は存在せず $\mathcal{T}(\delta_0, \beta) = \emptyset$ である（下図）．

\section{Kantorovich 緩和}
\label{sec:sem-kantorovich}

\subsection{カップリング}

\begin{definition}{カップリング（coupling）}{sem-coupling}
  $\X, \Y$ 上の確率測度 $\alpha \in \Mm_+^1(\X)$, $\beta \in \Mm_+^1(\Y)$
  に対して，$\X \times \Y$ 上の確率測度のうち周辺測度がそれぞれ
  $\alpha, \beta$ に一致するものの集合
  \[
    \Couplings(\alpha, \beta) \defeq
    \left\{ \pi \in \Mm_+^1(\X \times \Y) \;\middle|\;
    (P_\X)\pushforward \pi = \alpha, \;
    (P_\Y)\pushforward \pi = \beta \right\}
  \]
  を $\alpha$ と $\beta$ の\textbf{カップリング}の集合という．
  ここで $P_\X \colon \X \times \Y \to \X$, $P_\Y \colon \X \times \Y \to \Y$
  はそれぞれ第 1・第 2 成分への射影である．
\end{definition}

離散測度
$\alpha = \sum_i a_i \delta_{x_i}$，
$\beta = \sum_j b_j \delta_{y_j}$
の場合，カップリングは
$\pi = \sum_{i,j} P_{i,j} \delta_{(x_i, y_j)}$
の形をとる．
周辺条件 $(P_\X)\pushforward \pi = \alpha$ は
$\sum_j P_{i,j} = a_i$，すなわち $\mathbf{P}\ones_m = \mathbf{a}$ に帰着し，
$(P_\Y)\pushforward \pi = \beta$ は $\mathbf{P}^\top \ones_n = \mathbf{b}$ に帰着する．
したがって\textbf{離散カップリング}の集合は
\[
  \CouplingsD(\mathbf{a}, \mathbf{b}) \defeq
  \left\{ \mathbf{P} \in \R_+^{n \times m} \;\middle|\;
  \mathbf{P} \ones_m = \mathbf{a}, \;
  \mathbf{P}^\top \ones_n = \mathbf{b} \right\}
\]
で与えられる．
$P_{i,j}$ は「輸送元のビン $i$ から輸送先のビン $j$ へ送られる質量」を表す．
$\CouplingsD(\mathbf{a}, \mathbf{b})$ は $n + m$ 本の等式制約で定まる
$\R_+^{n \times m}$ の有界部分集合であるから，
凸多面体（定義~\ref{def:sem-polytope}）をなす．

\begin{example}{Monge 写像が存在しない場合の解消}{sem-kantorovich-resolve}
  例~\ref{ex:sem-monge-noexist} では
  $\alpha = \delta_0$，$\beta = \frac{1}{2}\delta_{-1} + \frac{1}{2}\delta_1$
  に対して Monge 写像が存在しないことを見た．
  Kantorovich の枠組みでこの問題を考えよう．

  離散表現は $\mathbf{a} = (1)$，$\mathbf{b} = (1/2,\; 1/2)$ であり，
  カップリング $\mathbf{P} \in \R_+^{1 \times 2}$ は
  $\mathbf{P}\ones_2 = 1$，$\mathbf{P}^\top \ones_1 = \mathbf{b}$
  を満たす必要がある．行が1つしかないため $P_{1,j} = b_j$ が強制され，
  \[
    \mathbf{P} = \begin{pmatrix} 1/2 & 1/2 \end{pmatrix}
  \]
  が唯一のカップリングである．
  コスト $c(x,y) = |x - y|$ から
  $\mathbf{C} = (|0-(-1)|,\; |0-1|) = (1,\; 1)$ であり，
  \[
    \inner{\mathbf{C}}{\mathbf{P}}
    = 1 \cdot \tfrac{1}{2} + 1 \cdot \tfrac{1}{2} = 1.
  \]
\end{example}

\subsection{Kantorovich 問題}

\begin{definition}{Kantorovich 問題（Kantorovich problem）}{sem-kantorovich}
  $\X, \Y$ 上の確率測度 $\alpha \in \Mm_+^1(\X)$, $\beta \in \Mm_+^1(\Y)$ と
  可測関数 $c \colon \X \times \Y \to \R_+$ に対して，
  \textbf{Kantorovich 問題}とは
  \[
    \MK_c(\alpha, \beta) \defeq
    \inf_{\pi \in \Couplings(\alpha, \beta)}
    \int_{\X \times \Y} c(x, y) \, \d\pi(x, y)
  \]
  を求める問題である．
\end{definition}

離散カップリング $\pi = \sum_{i,j} P_{i,j} \delta_{(x_i, y_j)}$ の場合，
$\int c \, \d\pi = \sum_{i,j} C_{i,j} P_{i,j} = \inner{\mathbf{C}}{\mathbf{P}}$
となるから，\textbf{離散 Kantorovich 問題}は
\[
  \MKD_{\mathbf{C}}(\mathbf{a}, \mathbf{b}) \defeq
  \min_{\mathbf{P} \in \CouplingsD(\mathbf{a}, \mathbf{b})}
  \inner{\mathbf{C}}{\mathbf{P}}
\]
と書ける．
これは $nm$ 個の変数と $n + m$ 本の等式制約を持つ有限次元の最適化問題である．
