      disintegration theorem を使うため $\X$ に Polish 性
      （定義~\ref{def:sem-polish}）を課す．
  \end{enumerate}
\end{setting}

\section{最適割当問題}
\label{sec:sem-assignment}

$n$ 人の作業者と $n$ 個の仕事があり，
作業者 $i$ を仕事 $j$ に配属するコストが $C_{i,j}$ で与えられているとする．
作業者と仕事を1対1に対応させるとき，
コストを最小にする割当を求めたい．

\begin{definition}{最適割当問題}{sem-assignment}
  コスト行列 $\mathbf{C} \in \R_+^{n \times n}$ が与えられたとき，
  \textbf{最適割当問題}（optimal assignment problem）とは，
  \[
    \min_{\sigma \in \Perm(n)} \frac{1}{n} \sum_{i=1}^{n} C_{i, \sigma(i)}
  \]
  を解く置換 $\sigma \in \Perm(n)$ を求める問題である．
  ここで置換 $\sigma \in \Perm(n)$ は一行記法
  $\sigma = (\sigma(1), \sigma(2), \ldots, \sigma(n))$ で表す．
\end{definition}

\begin{example}{最適割当}{sem-assignment-cost-example}
  3人の作業者を3つの仕事場に配属する状況を考える．
  各作業者の自宅と仕事場の位置が異なるため，
  通勤コスト $C_{i,j}$（作業者 $i$ が仕事場 $j$ に通う負担）は
  組合せによって異なる：
  \[
    \mathbf{C} =
    \begin{pmatrix}
      5 & 2 & 8 \\
      7 & 4 & 1 \\
      3 & 9 & 6
    \end{pmatrix}.
  \]
  たとえば $C_{1,2} = 2$ は
  「作業者1が仕事場2に通うコスト」を意味する．

  矢印上の数字は個別コスト $C_{i,\sigma(i)}$ である．

この例では $3! = 6$ 通りの全列挙で最適解が求まったが，
一般には $|\Perm(n)| = n!$ は急激に増大し，
$n = 70$ でも $n! > 10^{100}$ となるため全列挙は現実的でない．
ハンガリアン法やオークションアルゴリズムなど，
$O(n^3)$ で解ける効率的な手法が知られている．

\begin{remark}{最適割当問題の解の一意性}{sem-uniqueness}
  最適割当問題の最適解は\textbf{一意とは限らない}．

  \textbf{例}：$n = 2$ とし，
  作業者 $x_1, x_2$ と仕事場 $y_1, y_2$ が
  辺の長さ $1$ の正方形の頂点に配置されているとする．
  $x_1, x_2$ を対角頂点に，$y_1, y_2$ を残りの対角頂点に置くと，
  すべての作業者--仕事場間の距離が等しくなる（$= 1$）．

\section{Monge 問題}
\label{sec:sem-monge}

最適割当問題では，同数の点を均等な重みで1対1に対応させていた．
これを，個数や重みが異なる場合や，1対多の写像にも一般化したい．

「ある場所にある砂山を，最小の労力で別の場所に所定の形状で積み上げるには，
各砂粒をどこに運べばよいか？」---
1781年に Monge が提唱したこの問題は，
これらを許すより一般的な枠組みを与える．

\begin{definition}{Monge 問題}{sem-monge}
  確率測度 $\alpha \in \Mm_+^1(\X)$，$\beta \in \Mm_+^1(\Y)$ と
  ボレル可測なコスト関数 $c \colon \X \times \Y \to \R_+$ に対して，
  \textbf{Monge 問題}とは，
  \[
    \inf_{T}
    \left\{
    \int_{\X} c(x, T(x)) \, \d\alpha(x)
    \;\middle|\;
    T \colon \X \to \Y \text{ はボレル可測}, \;
    T\pushforward \alpha = \beta
    \right\}
  \]
  を求める問題である．
\end{definition}

離散測度
$\alpha = \sum_{i=1}^n a_i \delta_{x_i}$，
$\beta = \sum_{j=1}^m b_j \delta_{y_j}$
の場合を考え，$C_{i,j} \defeq c(x_i, y_j)$ と置く．
Dirac 測度に対する積分 $\int f \, \d\delta_x = f(x)$ を用いると
\[
  \int_\X c(x, T(x)) \, \d\alpha(x)
  = \sum_{i=1}^n a_i \int_\X c(x, T(x)) \, \d\delta_{x_i}(x)
  = \sum_{i=1}^n a_i \, c(x_i, T(x_i))
\]
となるから，\textbf{離散 Monge 問題}は
\[
  \min_{T}
  \left\{
  \sum_{i=1}^{n} a_i \, c(x_i, T(x_i))
  \;\middle|\;
  T\pushforward \alpha = \beta
  \right\}
\]
と書ける．
特に $n = m$ かつ $a_i = b_j = 1/n$ のとき，
$T\pushforward \alpha = \beta$ は $T$ が全単射であることを要求するから，
Monge 問題は最適割当問題に帰着する．

\subsection{Monge 問題の本質的困難}

Monge の定式化には2つの深刻な問題がある：
\begin{enumerate}
  \item \textbf{実行可能集合の非凸性}：凸最適化の手法が使えない．
  \item \textbf{解の非存在}：実行可能な写像 $T$ 自体が存在しない場合がある．
\end{enumerate}
順に見ていく．

\subsubsection{困難1：実行可能集合の非凸性}

$\X = \Y = \R$ とする．
Monge 問題の実行可能集合
\[
  \mathcal{T}(\alpha, \beta)
  \defeq \left\{ T \colon \X \to \Y \;\middle|\;
  T \text{ はボレル可測},\; T\pushforward \alpha = \beta \right\}
\]
は一般に\textbf{凸でない}．
$\mathcal{T}(\alpha,\beta)$ が凸であるためには，
任意の $T_1, T_2 \in \mathcal{T}(\alpha,\beta)$ と
任意の係数 $t \in [0,1]$ に対して，凸結合
$\bar{T} = t T_1 + (1-t)T_2$ が再び
$\bar{T}\pushforward \alpha = \beta$ を満たす必要があるが，
これは一般に成り立たない．

\begin{example}{実行可能集合の非凸性}{sem-nonconvex}
  $\alpha = \beta = \tfrac{1}{2}\delta_{-1} + \tfrac{1}{2}\delta_1$ とし，
  \[
    T_1(x) = x,\qquad T_2(x) = -x,\qquad t = \tfrac{1}{2}
  \]
  をとる．任意のボレル可測写像 $T$，任意の点 $x \in \X$，$A \in \Bb(\Y)$ に対し
  $T(x) \in A \iff x \in T^{-1}(A)$ であるから，Dirac 測度の定義より
  \[
    \delta_{T(x)}(A) = \mathbf{1}_{\{T(x) \in A\}}
    = \mathbf{1}_{\{x \in T^{-1}(A)\}}
    = \delta_x\bigl(T^{-1}(A)\bigr).
  \]

  これと押し出しの定義
  $({T_i}\pushforward\alpha)(A) = \alpha\bigl(T_i^{-1}(A)\bigr)$
  を組み合わせると，任意の $A \in \Bb(\Y)$ に対して
  \begin{align*}
    ({T_1}\pushforward \alpha)(A)
      &= \tfrac{1}{2}\delta_{-1}\bigl(T_1^{-1}(A)\bigr)
         + \tfrac{1}{2}\delta_{1}\bigl(T_1^{-1}(A)\bigr) \\
      &= \tfrac{1}{2}\delta_{T_1(-1)}(A) + \tfrac{1}{2}\delta_{T_1(1)}(A) \\
      &= \tfrac{1}{2}\delta_{-1}(A) + \tfrac{1}{2}\delta_{1}(A)
       = \beta(A),\\
    ({T_2}\pushforward \alpha)(A)
      &= \tfrac{1}{2}\delta_{-1}\bigl(T_2^{-1}(A)\bigr)
         + \tfrac{1}{2}\delta_{1}\bigl(T_2^{-1}(A)\bigr) \\
      &= \tfrac{1}{2}\delta_{T_2(-1)}(A) + \tfrac{1}{2}\delta_{T_2(1)}(A) \\
      &= \tfrac{1}{2}\delta_{1}(A) + \tfrac{1}{2}\delta_{-1}(A)
       = \beta(A)
  \end{align*}
  となるので，$T_1, T_2 \in \mathcal{T}(\alpha, \beta)$ である．

  この凸結合は
  \[
    \bar{T}(x) = \tfrac{1}{2}\, x + \tfrac{1}{2}(-x) = 0
    \qquad (\forall\, x \in \R)
  \]
  すなわち $0$ を値にとる定値写像である．定値写像は $\alpha$
  の全質量を 1 点 $0$ に集めるから $\bar{T}\pushforward\alpha = \delta_0$ となり，
  \[
    \bar{T}\pushforward\alpha = \delta_0
    \;\neq\; \tfrac{1}{2}\delta_{-1} + \tfrac{1}{2}\delta_1 = \beta.
  \]
  よって $\bar{T} \notin \mathcal{T}(\alpha, \beta)$ であり，
  凸結合は実行可能集合に戻らない．すなわち $\mathcal{T}(\alpha, \beta)$ は凸でない．
\end{example}

この例の本質は，写像から測度への対応 $T \longmapsto T\pushforward\alpha$ が
\textbf{非線形}であることにある．実際，「押し出してから平均する」と
\[
  \tfrac{1}{2}\, {T_1}\pushforward\alpha + \tfrac{1}{2}\, {T_2}\pushforward\alpha
  = \tfrac{1}{2}\beta + \tfrac{1}{2}\beta = \beta
\]
が成り立つ一方，「平均してから押し出す」と
$\bar{T}\pushforward\alpha = \delta_0$ となり両者は一致しない．
一般に
\[
  \bigl(t T_1 + (1-t)T_2\bigr)\pushforward \alpha
  \;\neq\; t \cdot {T_1}\pushforward \alpha + (1-t) \cdot {T_2}\pushforward \alpha
\]
であり，この非線形性こそが実行可能集合 $\mathcal{T}(\alpha,\beta)$ の
非凸性の根本原因である．

\subsubsection{困難2：解の非存在}

写像による輸送では，質量の\textbf{分割}が許されない．
すなわち，$x$ の質量は $T(x)$ という1点にすべて送らなければならない．
このため，輸送元の測度が輸送先の測度より「粗い」場合に問題が生じる．

\begin{example}{Monge 写像が存在しない場合}{sem-monge-noexist}
  $\alpha = \delta_0$，$\beta = \tfrac{1}{2}\delta_{-1} + \tfrac{1}{2}\delta_1$
  を考える．$\alpha$ は原点の 1 点に集中した質量 $1$，
  $\beta$ は $-1$ と $1$ の 2 箇所に半分ずつ分散した測度である．

  任意のボレル可測写像 $T \colon \X \to \Y$ に対して，
  例~\ref{ex:sem-nonconvex} で用いた等式
  $\delta_x\bigl(T^{-1}(A)\bigr) = \delta_{T(x)}(A)$ と押し出しの定義より，
  任意の $A \in \Bb(\Y)$ に対して
  \[
    (T\pushforward \delta_0)(A)
    = \delta_0\bigl(T^{-1}(A)\bigr)
    = \delta_{T(0)}(A)
  \]
  が成り立つので $T\pushforward\delta_0 = \delta_{T(0)}$ である．
  つまり $\alpha$ を $T$ で運んだ結果は必ず行き先 $T(0)$ の
  1 点に集中した測度となり，これは写像 $T$ が質量を分割できないことを
  測度の言葉で表している．

  したがって $T \in \mathcal{T}(\delta_0, \beta)$ となるには
  1 点集中測度 $\delta_{T(0)}$ が 2 点分散測度 $\beta$ に一致しなければならない．
  しかし $\delta_{T(0)}$ はどの $A \in \Bb(\Y)$ でも値が $0$ か $1$ なのに対し，
  $A = \{-1\}$ で $\beta(\{-1\}) = \tfrac{1}{2}$ となるので，
  どんな $T(0) \in \R$ をとっても両者は食い違う．
  よって条件を満たす $T$ は存在せず
  \[
    \mathcal{T}(\delta_0, \beta) = \emptyset
  \]
  である（下図）．

\section{Kantorovich 緩和}
\label{sec:sem-kantorovich}

\subsection{鍵となるアイデア}

前節で見たとおり，Monge の定式化には
実行可能集合の非凸性と解の非存在という2つの本質的困難がある．
また，最適割当問題は $n = m$ かつ一様な重みの場合しか扱えず，
Monge 問題はこれを一般化するものの，$n!$ 通りの組合せ的構造を持つ．

Kantorovich (1942) は，これらの困難を一挙に解消するアイデアを提示した．
Monge の枠組みでは，各点 $x_i$ の質量をすべて1つの行き先 $T(x_i)$ に送る
\textbf{決定論的}な輸送を考えていた．
Kantorovich はこれを，質量を複数の行き先に分割して送ることを許す
\textbf{確率的}な輸送に置き換える．
この「\textbf{質量の分割}」（mass splitting）を許す輸送計画を，
$\X \times \Y$ 上の結合測度（\textbf{カップリング}）として定式化する．

\subsection{カップリング}

\begin{definition}{カップリング}{sem-coupling}
  確率測度 $\alpha \in \Mm_+^1(\X)$ と $\beta \in \Mm_+^1(\Y)$ に対して，
  $\alpha$ と $\beta$ の\textbf{カップリング}の集合を
  \[
    \Couplings(\alpha, \beta) \defeq
    \left\{ \pi \in \Mm_+^1(\X \times \Y) \;\middle|\;
    (P_\X)\pushforward \pi = \alpha, \;
    (P_\Y)\pushforward \pi = \beta \right\}
  \]
  で定義する．ここで $P_\X \colon \X \times \Y \to \X$，
  $P_\Y \colon \X \times \Y \to \Y$ はそれぞれ第1・第2成分への射影である．
  すなわち，$\pi$ の周辺測度がそれぞれ $\alpha$, $\beta$ に一致することを要求する．
\end{definition}

離散測度
$\alpha = \sum_i a_i \delta_{x_i}$，
$\beta = \sum_j b_j \delta_{y_j}$
の場合，カップリングは
$\pi = \sum_{i,j} P_{i,j} \delta_{(x_i, y_j)}$
の形をとる．
周辺条件 $(P_\X)\pushforward \pi = \alpha$ は
$\sum_j P_{i,j} = a_i$，すなわち $\mathbf{P}\ones_m = \mathbf{a}$ に帰着し，
$(P_\Y)\pushforward \pi = \beta$ は $\mathbf{P}^\top \ones_n = \mathbf{b}$ に帰着する．
したがって\textbf{離散カップリング}の集合は
\[
  \CouplingsD(\mathbf{a}, \mathbf{b}) \defeq
  \left\{ \mathbf{P} \in \R_+^{n \times m} \;\middle|\;
  \mathbf{P} \ones_m = \mathbf{a}, \;
  \mathbf{P}^\top \ones_n = \mathbf{b} \right\}
\]
で与えられる．
$P_{i,j}$ は「輸送元のビン $i$ から輸送先のビン $j$ へ送られる質量」を表す．
$\CouplingsD(\mathbf{a}, \mathbf{b})$ は $n + m$ 本の等式制約で定まる
$\R_+^{n \times m}$ の有界部分集合であるから，
\textbf{凸多面体}（convex polytope）をなす．

\begin{example}{Monge 写像が存在しない場合の解消}{sem-kantorovich-resolve}
  例~\ref{ex:sem-monge-noexist} では
  $\alpha = \delta_0$，$\beta = \frac{1}{2}\delta_{-1} + \frac{1}{2}\delta_1$
  に対して Monge 写像が存在しないことを見た．
  Kantorovich の枠組みでこの問題を考えよう．

  離散表現は $\mathbf{a} = (1)$，$\mathbf{b} = (1/2,\; 1/2)$ であり，
  カップリング $\mathbf{P} \in \R_+^{1 \times 2}$ は
  $\mathbf{P}\ones_2 = 1$，$\mathbf{P}^\top \ones_1 = \mathbf{b}$
  を満たす必要がある．行が1つしかないため $P_{1,j} = b_j$ が強制され，
  \[
    \mathbf{P} = \begin{pmatrix} 1/2 & 1/2 \end{pmatrix}
  \]
  が唯一のカップリングである．
  コスト $c(x,y) = |x - y|$ から
  $\mathbf{C} = (|0-(-1)|,\; |0-1|) = (1,\; 1)$ であり，
  \[
    \inner{\mathbf{C}}{\mathbf{P}}
    = 1 \cdot \tfrac{1}{2} + 1 \cdot \tfrac{1}{2} = 1.
  \]
  Monge の枠組みでは不可能だった「1点の質量を2箇所に分割する」操作が，
  Kantorovich のカップリングとして自然に実現されている．
\end{example}

\subsection{Kantorovich 問題}

\begin{definition}{Kantorovich 問題}{sem-kantorovich}
  確率測度 $\alpha \in \Mm_+^1(\X)$，$\beta \in \Mm_+^1(\Y)$，
  ボレル可測なコスト関数 $c \colon \X \times \Y \to \R_+$ に対して，
  \textbf{Kantorovich 問題}とは
  \[
    \MK_c(\alpha, \beta) \defeq
    \inf_{\pi \in \Couplings(\alpha, \beta)}
    \int_{\X \times \Y} c(x, y) \, \d\pi(x, y)
  \]
  を求める問題である．
\end{definition}

離散カップリング $\pi = \sum_{i,j} P_{i,j} \delta_{(x_i, y_j)}$ の場合，
$\int c \, \d\pi = \sum_{i,j} C_{i,j} P_{i,j} = \inner{\mathbf{C}}{\mathbf{P}}$
となるから，\textbf{離散 Kantorovich 問題}は
\[
  \MKD_{\mathbf{C}}(\mathbf{a}, \mathbf{b}) \defeq
  \min_{\mathbf{P} \in \CouplingsD(\mathbf{a}, \mathbf{b})}
  \inner{\mathbf{C}}{\mathbf{P}}
\]
と書ける．
これは $nm$ 個の変数と $n + m$ 本の等式制約を持つ\textbf{線形計画問題}（LP）である．

\subsection{最適割当問題との関係}

置換行列 $\mathbf{P}_\sigma$ は $j = \sigma(i)$ のときのみ成分が $1$ なので
\[
  \inner{\mathbf{C}}{\mathbf{P}_\sigma}
  = \sum_{i,j} C_{i,j} (\mathbf{P}_\sigma)_{i,j}
  = \sum_{i=1}^{n} C_{i,\sigma(i)}
\]
が成り立つ．一様ヒストグラムのカップリング集合に組み込むため縮尺版
$\mathbf{P}_\sigma / n$ を用いると
$\mathbf{P}_\sigma / n \in \CouplingsD(\ones_n / n,\, \ones_n / n)$ であり，
\[
  \frac{1}{n} \sum_{i=1}^{n} C_{i,\sigma(i)}
  = \inner{\mathbf{C}}{\mathbf{P}_\sigma / n}
\]
を得る．したがって最適割当問題はカップリングを
$\{\mathbf{P}_\sigma / n \mid \sigma \in \Perm(n)\}$ に制限した
Kantorovich 問題であり，包含関係から
\[
  \MKD_{\mathbf{C}}(\ones_n/n, \ones_n/n)
  \leq \min_{\sigma \in \Perm(n)} \inner{\mathbf{C}}{\mathbf{P}_\sigma / n}
\]
が従う．次の命題は，この不等式で等号が成立することを示す．

\begin{proposition}{Kantorovich 緩和のタイト性}{sem-kantorovich-tight}
  $m = n$ かつ $\mathbf{a} = \mathbf{b} = \ones_n / n$ のとき，
  Kantorovich 問題の最適解として
  ある最適置換 $\sigma^* \in \Perm(n)$ に対応する
  縮尺置換行列 $\mathbf{P}_{\sigma^*} / n$ が取れる．
  すなわち，Kantorovich 緩和は最適割当問題に対して\textbf{タイト}である．
\end{proposition}

\begin{proof}
  Birkhoff--von Neumann の定理により，二重確率行列の集合
  $\CouplingsD(\ones_n,\, \ones_n)$ の端点全体は置換行列の集合
  $\{\mathbf{P}_\sigma \mid \sigma \in \Perm(n)\}$ に等しい．
  線形写像 $\mathbf{P} \mapsto \mathbf{P} / n$ は
  $\CouplingsD(\ones_n,\, \ones_n)$ から $\CouplingsD(\ones_n/n,\, \ones_n/n)$
  への凸多面体の全単射で端点を端点に写すから，
  後者の端点全体は $\{\mathbf{P}_\sigma / n \mid \sigma \in \Perm(n)\}$ に等しい．
  線形計画の基本定理により，有界凸多面体上の線形目的関数の最小値は
  端点で達成されるから，
  $\MKD_{\mathbf{C}}(\ones_n/n, \ones_n/n)$ の最適解として
  ある $\mathbf{P}_{\sigma^*} / n$ が取れる．
\end{proof}

\subsection{Kantorovich 緩和の利点}

Kantorovich の定式化は Monge 問題の困難をすべて解消する．

\begin{remark}{カップリング集合の凸性}{sem-coupling-convex}
  Monge の実行可能集合 $\mathcal{T}(\alpha, \beta)$ は非凸であったが，
  カップリング集合 $\CouplingsD(\mathbf{a}, \mathbf{b})$ は\textbf{凸}である．
  実際，$\mathbf{P}_1, \mathbf{P}_2 \in \CouplingsD(\mathbf{a}, \mathbf{b})$
  と $t \in [0,1]$ に対し，
  凸結合 $\mathbf{P} = t\mathbf{P}_1 + (1-t)\mathbf{P}_2$ は
  \[
    \mathbf{P}\ones_m = t\,\mathbf{a} + (1-t)\,\mathbf{a} = \mathbf{a}, \quad
    \mathbf{P}^\top \ones_n = t\,\mathbf{b} + (1-t)\,\mathbf{b} = \mathbf{b}
  \]
  を満たすから $\mathbf{P} \in \CouplingsD(\mathbf{a}, \mathbf{b})$ である．
  さらに目的関数 $\inner{\mathbf{C}}{\mathbf{P}}$ は $\mathbf{P}$ に関して線形であるから，
  離散 Kantorovich 問題は\textbf{凸最適化問題}（実際には線形計画問題）となる．
\end{remark}

解の存在も保証される．
非負性と周辺条件から任意の
$\mathbf{P} \in \CouplingsD(\mathbf{a}, \mathbf{b})$ は
$0 \leq P_{i,j} \leq \min(a_i, b_j) \leq 1$ を満たすので，
$\CouplingsD(\mathbf{a}, \mathbf{b})$ は $\R^{n \times m}$ の有界閉集合
（したがってコンパクト）であり，連続関数の最小値が達成される．
さらに，積 $\mathbf{a}\mathbf{b}^\top$ は
$(\mathbf{a}\mathbf{b}^\top)\ones_m
= \mathbf{a}(\mathbf{b}^\top \ones_m) = \mathbf{a}$，
$(\mathbf{a}\mathbf{b}^\top)^\top \ones_n
= \mathbf{b}(\mathbf{a}^\top \ones_n) = \mathbf{b}$
を満たすためカップリングの 1 つを与え，実行可能集合は空でない．
したがって最適解 $\mathbf{P}^*$ が\textbf{常に存在}する．

\subsection{Monge 問題の緩和としての位置づけ}

Monge 写像 $T$ に対して，
$(\Id, T) \colon \X \to \X \times \Y$，$x \mapsto (x, T(x))$
により $\pi_T = (\Id, T)\pushforward \alpha$ とおくと
$\pi_T \in \Couplings(\alpha, \beta)$ となり，
\[
  \MK_c(\alpha, \beta) \leq
  \inf_{T}
  \left\{
  \int_\X c(x, T(x)) \, \d\alpha(x)
  \;\middle|\;
  T\pushforward \alpha = \beta
  \right\}.
\]
Kantorovich 問題は Monge 問題の緩和であり，
Monge 写像による輸送は「$\pi$ が $T$ のグラフ上に集中している」カップリングの特殊な場合である．

\section{Wasserstein 距離}
\label{sec:sem-wasserstein}

Kantorovich 問題の最適値 $\MKD_{\mathbf{C}}(\mathbf{a}, \mathbf{b})$ は
ヒストグラム間の「輸送コスト」を与えるが，
コスト行列 $\mathbf{C}$ が距離行列のべき乗であるとき，
この値から自然に\textbf{距離の公理を満たす量}が定義できる．
すなわち，OT は点間の ground 距離を，分布間の距離に\textbf{持ち上げる}
（lift する）正準的な方法を提供する．

\subsection{Ground 距離}

\begin{definition}{Ground 距離行列}{sem-ground-metric}
  $\distD \in \R_+^{n \times n}$ が $\range{n}$ 上の\textbf{距離行列}であるとは，
  以下の3条件を満たすことをいう：
  \begin{enumerate}
    \item[(i)] 同一性：$\distD_{i,j} = 0 \iff i = j$，
    \item[(ii)] 対称性：$\distD_{i,j} = \distD_{j,i}$，
    \item[(iii)] 三角不等式：$\distD_{i,k} \leq \distD_{i,j} + \distD_{j,k}$
      \quad $(\forall\, i, j, k \in \range{n})$．
  \end{enumerate}
\end{definition}

この距離行列 $\distD$ を「ビン間の距離」（ground metric）と呼ぶ．
例えば $\X = \R$ 上に台を持つヒストグラムでは，
$\distD_{i,j} = |x_i - x_j|$ のように台の点間のユークリッド距離を取ることが多い．

\subsection{$p$-Wasserstein 距離の定義}

\begin{proposition}{$p$-Wasserstein 距離}{sem-wasserstein-metric}
  $n = m$ とし，$p \geq 1$ に対して
  コスト行列を $\mathbf{C} = \distD^p = (\distD_{i,j}^p)_{i,j}$ とする．
  このとき
  \[
    \WassD_p(\mathbf{a}, \mathbf{b})
    \defeq \MKD_{\distD^p}(\mathbf{a}, \mathbf{b})^{1/p}
  \]
  は $\simplex_n$ 上の\textbf{距離}を定義する．
  すなわち，$\WassD_p$ は対称かつ正定値であり，三角不等式
  \[
    \WassD_p(\mathbf{a}, \mathbf{c})
    \leq \WassD_p(\mathbf{a}, \mathbf{b})
    + \WassD_p(\mathbf{b}, \mathbf{c})
    \qquad (\forall\, \mathbf{a}, \mathbf{b}, \mathbf{c} \in \simplex_n)
  \]
  を満たす．
\end{proposition}

\subsection{距離の公理の証明}

\begin{proof}
  \textbf{同一性．}
  $\mathbf{P}^\star \defeq \diag(\mathbf{a}) \in \CouplingsD(\mathbf{a}, \mathbf{a})$
  は実行可能で $\distD_{i,i}^p = 0$ より
  $\inner{\distD^p}{\mathbf{P}^\star} = 0$ を達成するから
  $\WassD_p(\mathbf{a}, \mathbf{a}) = 0$．

  逆に $\mathbf{a} \neq \mathbf{b}$ とする．
  $\mathbf{P} \in \CouplingsD(\mathbf{a}, \mathbf{b})$ の台が対角上に限られると仮定すれば，
  $\mathbf{P} = \diag(\mathbf{v})$ と書け，周辺条件から
  $\mathbf{a} = \mathbf{P}\ones = \mathbf{v} = \mathbf{P}^\top\ones = \mathbf{b}$
  となり仮定に反する．よって $P_{i_0, j_0} > 0$ なる $i_0 \neq j_0$ が存在し，
  $\distD$ の同一性から $\distD_{i_0, j_0} > 0$．したがって
  \[
    \inner{\distD^p}{\mathbf{P}}
    = \sum_{i,j} \distD_{i,j}^p P_{i,j}
    \geq \distD_{i_0, j_0}^p P_{i_0, j_0} > 0
  \]
  が任意の $\mathbf{P} \in \CouplingsD(\mathbf{a}, \mathbf{b})$ で成り立ち，
  最小値も正なので $\WassD_p(\mathbf{a}, \mathbf{b}) > 0$．

  \medskip
  \textbf{対称性．}
  転置 $\mathbf{P} \mapsto \mathbf{P}^\top$ は
  $\CouplingsD(\mathbf{a}, \mathbf{b})$ から
  $\CouplingsD(\mathbf{b}, \mathbf{a})$ への全単射であり，
  $\distD^p$ の対称性から
  $\inner{\distD^p}{\mathbf{P}} = \inner{\distD^p}{\mathbf{P}^\top}$．
  ゆえに $\WassD_p(\mathbf{a}, \mathbf{b}) = \WassD_p(\mathbf{b}, \mathbf{a})$．

  \medskip
  \textbf{三角不等式（glued coupling の構成）．}
  $\mathbf{a}, \mathbf{b}, \mathbf{c} \in \simplex_n$ とし，
  $\mathbf{P} \in \CouplingsD(\mathbf{a}, \mathbf{b})$ を
  $(\mathbf{a}, \mathbf{b})$ 間の最適輸送，
  $\mathbf{Q} \in \CouplingsD(\mathbf{b}, \mathbf{c})$ を
  $(\mathbf{b}, \mathbf{c})$ 間の最適輸送とする．

  $\mathbf{b}$ の台を $J \defeq \{j \mid b_j > 0\}$ と書き，
  \[
    S_{i,k} \defeq \sum_{j \in J} \frac{P_{i,j}\, Q_{j,k}}{b_j}
    \qquad (i, k \in \range{n})
  \]
  で補助カップリング $\mathbf{S} \in \R_+^{n \times n}$ を定める．
  $j \notin J$ のとき周辺制約 $\sum_i P_{i,j} = b_j = 0$ と $P_{i,j} \geq 0$ から
  $P_{i,j} = 0$（$\forall\, i$），同様に $Q_{j,k} = 0$ なので，
  和は $j \in J$ に制限してよい．
  このとき $\mathbf{S} \in \CouplingsD(\mathbf{a}, \mathbf{c})$ である：
  \[
    (\mathbf{S} \ones_n)_i
    = \sum_{j \in J} \frac{P_{i,j}}{b_j} \sum_{k} Q_{j,k}
    = \sum_{j \in J} \frac{P_{i,j}}{b_j} \cdot b_j
    = \sum_{j \in J} P_{i,j}
    = \sum_{j} P_{i,j}
    = a_i,
  \]
  $\mathbf{S}^\top \ones_n = \mathbf{c}$ も同様に確認できる．

  \medskip
  \textbf{不等式の導出．}
  有限集合 $I \defeq \range{n} \times J \times \range{n}$ 上で
  重み $w_{i,j,k} \defeq P_{i,j} Q_{j,k} / b_j \geq 0$ を測度と見なし，
  $L^p(w)$ ノルム $\|f\|_{L^p(w)} \defeq
  \bigl(\sum_{(i,j,k) \in I} |f_{i,j,k}|^p w_{i,j,k}\bigr)^{1/p}$
  を用いる．
  $\mathbf{S}$ は最適とは限らないから，$\distD$ の三角不等式と合わせ
  \begin{align*}
    \WassD_p(\mathbf{a}, \mathbf{c})^p
    &= \min_{\mathbf{P}' \in \CouplingsD(\mathbf{a}, \mathbf{c})}
       \inner{\distD^p}{\mathbf{P}'}
    \leq \inner{\distD^p}{\mathbf{S}}
    = \sum_{(i,j,k) \in I} \distD_{i,k}^p\, w_{i,j,k} \\
    &\leq \sum_{(i,j,k) \in I}
       (\distD_{i,j} + \distD_{j,k})^p\, w_{i,j,k}.
  \end{align*}
  $L^p(w)$ ノルムに関する Minkowski 不等式
  $\|f + g\|_{L^p(w)} \leq \|f\|_{L^p(w)} + \|g\|_{L^p(w)}$
  を $f_{i,j,k} = \distD_{i,j}$，$g_{i,j,k} = \distD_{j,k}$ に適用すると
  \[
    \WassD_p(\mathbf{a}, \mathbf{c})
    \leq
    \Bigl(\sum_{I} \distD_{i,j}^p\, w_{i,j,k}\Bigr)^{1/p}
    + \Bigl(\sum_{I} \distD_{j,k}^p\, w_{i,j,k}\Bigr)^{1/p}.
  \]
  第 1 項は $k$ について和を取ると $\sum_k Q_{j,k}/b_j = 1$（$j \in J$）より
  $\sum_{i, j \in J} \distD_{i,j}^p P_{i,j} = \sum_{i,j} \distD_{i,j}^p P_{i,j}$
  に等しく，
  第 2 項は $i$ について和を取ると $\sum_i P_{i,j}/b_j = 1$（$j \in J$）より
  $\sum_{j, k} \distD_{j,k}^p Q_{j,k}$ に等しい．ゆえに
  \[
    \WassD_p(\mathbf{a}, \mathbf{c})
    \leq \Bigl(\inner{\distD^p}{\mathbf{P}}\Bigr)^{1/p}
    + \Bigl(\inner{\distD^p}{\mathbf{Q}}\Bigr)^{1/p}
    = \WassD_p(\mathbf{a}, \mathbf{b})
    + \WassD_p(\mathbf{b}, \mathbf{c}).
  \]
\end{proof}

証明の核心は，2 つの最適輸送 $\mathbf{P}$, $\mathbf{Q}$ を
$\mathbf{b}$ を経由して「接着」（gluing）し，$(\mathbf{a}, \mathbf{c})$ 間の
実行可能な（一般に最適ではない）カップリング $\mathbf{S}$ を構成する点にある．

\subsection{Wasserstein 距離の直観的理解}

$p$-Wasserstein 距離の重要な性質を確認する．

\paragraph{Dirac 測度間の距離．}
任意の2点 $x, y$ に対して
$\Couplings(\delta_x, \delta_y) = \{\delta_{(x,y)}\}$（1つしかない）であるから，
\[
  \Wass_p(\delta_x, \delta_y)^p = \dist(x, y)^p.
\]
すなわち $\Wass_p(\delta_x, \delta_y) = \dist(x,y)$ である．
点上の Dirac 測度間の Wasserstein 距離は，そのまま ground 距離に一致する．
これが「距離の持ち上げ」の最も基本的な例である．

\paragraph{弱距離としての性質．}
$L^2$ ノルムや KL ダイバージェンスなどの「古典的」な距離・ダイバージェンスは，
台が重ならない離散分布間では定義できないか，無限大となってしまう．
例えば $\delta_0$ と $\delta_\varepsilon$ はどちらも離散測度であり，
台が重ならないため $L^2$ ノルムは定義できず，KL ダイバージェンスは $+\infty$ となる．
一方，$\Wass_p(\delta_0, \delta_\varepsilon) = \varepsilon$ であるから，
$\varepsilon \to 0$ のとき $\Wass_p(\delta_0, \delta_\varepsilon) \to 0$ となり，
分布の「空間的なシフト」が自然に定量化される．
この性質が，OT を分布間の比較に用いる大きな動機である．

\begin{remark}{弱収束との関係}{sem-weak-convergence}
  コンパクト空間 $\X$ 上の確率測度の列 $(\alpha_k)_k$ が $\alpha$ に
  \textbf{弱収束}するとは，
  任意の連続関数 $g \in \Cc(\X)$ に対して
  $\int_\X g \, \d\alpha_k \to \int_\X g \, \d\alpha$
  が成り立つことをいう（$\alpha_k \rightharpoonup \alpha$ と記す）．
  コンパクト空間上では，$\Wass_p(\alpha_k, \alpha) \to 0$ と
  $\alpha_k \rightharpoonup \alpha$ は同値であることが知られている．
  すなわち Wasserstein 距離は弱収束を特徴づける距離である．
\end{remark}

\subsection{一般測度への拡張}

ヒストグラム上の結果は一般の測度に自然に拡張できる．
一般測度での gluing には正則条件付き確率（disintegration theorem）の存在が
必要となるため，以下では $\X$ を定義~\ref{def:sem-polish} の Polish 空間に制限する．

\begin{proposition}{測度上の $p$-Wasserstein 距離}{sem-wasserstein-measure}
  $\X = \Y$ を Polish 空間（定義~\ref{def:sem-polish}），$\dist$ を $\X$ 上の距離，
  $c(x,y) = \dist(x,y)^p$（$p \geq 1$）とする．
  \[
    \Mm_p(\X) \defeq \left\{
      \alpha \in \Mm_+^1(\X) \;\middle|\;
      \int_\X \dist(x, x_0)^p \, \d\alpha(x) < \infty
    \right\}
    \qquad (x_0 \in \X \text{ は任意の基準点})
  \]
  とおくと，$\Mm_p(\X)$ は $x_0$ の取り方によらず一意に定まり，
  \[
    \Wass_p(\alpha, \beta)
    \defeq \MK_{\dist^p}(\alpha, \beta)^{1/p}
  \]
  は $\Mm_p(\X)$ 上の距離を定義する．
\end{proposition}

\begin{proof}
  \textbf{有限性．}
  $\alpha, \beta \in \Mm_p(\X)$ のとき
  $\dist(x, y)^p \leq 2^{p-1}(\dist(x, x_0)^p + \dist(y, x_0)^p)$ と
  $\pi \in \Couplings(\alpha, \beta)$ の周辺条件から，
  任意のカップリング $\pi$ について
  $\int \dist^p \, \d\pi \leq 2^{p-1}\bigl(\int \dist(\cdot, x_0)^p \d\alpha
  + \int \dist(\cdot, x_0)^p \d\beta\bigr) < \infty$ であり
  $\Wass_p(\alpha, \beta) < \infty$．

  \medskip
  \textbf{同一性・対称性．}
  $\Wass_p(\alpha, \alpha) = 0$ は対角埋込 $(\Id, \Id)\pushforward\alpha
  \in \Couplings(\alpha, \alpha)$ によって $\int \dist^p \, \d\pi = 0$ から従う．
  逆向きは台の議論から得る．
  対称性は $(x, y) \mapsto (y, x)$ による押し出しが
  $\Couplings(\alpha, \beta)$ と $\Couplings(\beta, \alpha)$
  の全単射を与えることから従う．

  \medskip
  \textbf{三角不等式（gluing lemma）．}
  $\alpha, \beta, \gamma \in \Mm_p(\X)$ とし，
  $\pi_1 \in \Couplings(\alpha, \beta)$，
  $\pi_2 \in \Couplings(\beta, \gamma)$ を最適カップリングとする．
  Polish 空間上の Radon 測度に対する disintegration theorem
  （正則条件付き確率の存在）により，
  $\pi_1$ の第 2 成分 $\beta$ に関する正則条件付き分布
  $\{\pi_1(\cdot \mid y)\}_{y \in \X} \subseteq \Mm_+^1(\X)$ と
  $\pi_2$ の第 1 成分 $\beta$ に関する正則条件付き分布
  $\{\pi_2(\cdot \mid y)\}_{y \in \X} \subseteq \Mm_+^1(\X)$ が取れる．
  これらを用いて $\X^3$ 上の確率測度 $\mu$ を
  \[
    \mu(A \times B \times C) \defeq
    \int_B \pi_1(A \mid y)\, \pi_2(C \mid y) \, \d\beta(y)
    \qquad (A, B, C \in \Bb(\X))
  \]
  で定めれば，第 $(1,2)$ 周辺が $\pi_1$，第 $(2,3)$ 周辺が $\pi_2$ となる．
  射影 $P_{13} \colon \X^3 \to \X^2$ による押し出し
  $\pi_{13} \defeq (P_{13})\pushforward \mu \in \Couplings(\alpha, \gamma)$
  を取り，離散版と同様の $L^p(\mu)$ 空間における Minkowski 不等式と
  $\dist$ の三角不等式を用いて
  \[
    \Wass_p(\alpha, \gamma)
    \leq \Bigl(\int \dist(x, z)^p \, \d\pi_{13}\Bigr)^{1/p}
    \leq \Bigl(\int \dist^p \, \d\pi_1\Bigr)^{1/p}
    + \Bigl(\int \dist^p \, \d\pi_2\Bigr)^{1/p}
    = \Wass_p(\alpha, \beta) + \Wass_p(\beta, \gamma).
  \]
  詳細は Villani (2009, Thm 4.8) または Santambrogio (2015, §5.1) を参照．
\end{proof}
