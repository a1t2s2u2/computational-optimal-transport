\chapter{アルゴリズムの基礎}
\label{ch:seminar-02}

第~\ref{ch:seminar-01}~章で離散 Kantorovich 問題
\[
  \MKD_{\mathbf{C}}(\mathbf{a}, \mathbf{b})
  \defeq \min_{\mathbf{P} \in \CouplingsD(\mathbf{a}, \mathbf{b})}
  \inner{\mathbf{C}}{\mathbf{P}}
\]
を定式化し，最適解の存在
（主張~\ref{clm:sem-discrete-kantorovich-existence}）を確認した．
その求解は，線形計画標準形への帰着，輸送多面体の頂点構造，
そして\textbf{ネットワーク単体法}による反復解法に基づく．

%% ============================================================
%% 線形計画標準形
%% ============================================================
\section{線形計画標準形への帰着}
\label{sec:sem-lp-standard}

離散 Kantorovich 問題を線形計画の標準形に書き直す．
これによりネットワーク単体法をはじめとする一般の LP アルゴリズムが適用できる．

行列 $\mathbf{P} \in \R^{n \times m}$ を列方向に並べたベクトルを
$\mathbf{p} \in \R^{nm}$，コスト行列 $\mathbf{C}$ のベクトル化を
$\mathbf{c} \in \R^{nm}$ と書く．
周辺条件 $\mathbf{P}\ones_m = \mathbf{a}$，$\mathbf{P}^\top \ones_n = \mathbf{b}$
を制約行列 $\mathbf{A} \in \R^{(n+m) \times nm}$ でまとめると，
離散 Kantorovich 問題は標準形の線形計画
\[
  \MKD_{\mathbf{C}}(\mathbf{a}, \mathbf{b})
  = \min_{\substack{\mathbf{p} \in \R_+^{nm} \\
    \mathbf{A}\mathbf{p} = [\mathbf{a}^\top, \mathbf{b}^\top]^\top}}
  \mathbf{c}^\top \mathbf{p}
\]
と表される．

\begin{claim}{制約行列 $\mathbf{A}$ の階数}{sem-A-rank}
  $\rank \mathbf{A} = n + m - 1$．
\end{claim}

\begin{proof}
  $\mathbf{A}$ の最初の $n$ 行の和は $\ones_{nm}^\top$，
  残りの $m$ 行の和もまた $\ones_{nm}^\top$ に等しい．
  したがって $n + m$ 本の行は線形独立でなく，$\rank \mathbf{A} \leq n + m - 1$．

  逆に，最初の $n$ 行と残りの $m$ 行のうち $1$ 本を除いた行は線形独立であることが
  確認できる．したがって $\rank \mathbf{A} \geq n + m - 1$．
\end{proof}

ランク条件は次節の頂点構造解析の鍵となる：
LP 一般論により，標準形 $\{\mathbf{p} \geq 0 \mid \mathbf{A}\mathbf{p} = \mathbf{r}\}$
の頂点では非零成分の数が $\rank \mathbf{A}$ 以下に抑えられる．

\section{輸送多面体の頂点}
\label{sec:sem-transport-vertices}

線形計画の最適解は実行可能領域の\textbf{頂点}（端点）で達成される．
ネットワーク単体法はこの頂点間を移動して探索するため，
頂点の構造を理解することがアルゴリズムの設計に不可欠である．

\begin{definition}{頂点（端点）}{sem-vertex}
  凸集合 $K \subset \R^d$ の点 $\mathbf{v} \in K$ が \textbf{頂点}（vertex; 端点）
  であるとは，$\mathbf{v} = \tfrac{1}{2}(\mathbf{u} + \mathbf{w})$,
  $\mathbf{u}, \mathbf{w} \in K$ が成り立つならば $\mathbf{u} = \mathbf{w} = \mathbf{v}$
  であることをいう．
\end{definition}

\begin{definition}{二部グラフと台グラフ}{sem-bipartite-support}
  ソース節点集合 $V \defeq \{1, \ldots, n\}$ と
  ターゲット節点集合 $V' \defeq \{1', \ldots, m'\}$ に対し，
  $V$ と $V'$ の間のすべての辺
  $\{(i, j') \mid i \in \range{n},\, j \in \range{m}\}$ から成るグラフを
  \textbf{完全二部グラフ}という．
  輸送計画 $\mathbf{P} \in \CouplingsD(\mathbf{a}, \mathbf{b})$ の
  \textbf{台グラフ}を，$P_{i,j} > 0$ なる辺 $(i, j')$ の集合
  $S(\mathbf{P}) \defeq \{(i, j') \mid P_{i,j} > 0\}$ と頂点 $V \cup V'$ から成るグラフ
  \[
    G(\mathbf{P}) \defeq (V \cup V', S(\mathbf{P}))
  \]
  で定める．
\end{definition}

\begin{definition}{閉路・森・木}{sem-cycle-forest}
  グラフにおいて相異なる節点 $v_1, v_2, \ldots, v_k$ と辺
  $(v_1, v_2), (v_2, v_3), \ldots, (v_{k-1}, v_k), (v_k, v_1)$ から成る経路を
  \textbf{閉路}（cycle）という．閉路を持たないグラフを\textbf{森}（forest），
  連結な森を\textbf{木}（tree）という．
\end{definition}

\begin{proposition}{頂点の木構造}{sem-extremal-tree}
  $\mathbf{P}$ が $\CouplingsD(\mathbf{a}, \mathbf{b})$ の頂点ならば，
  台グラフ $G(\mathbf{P})$ は閉路を持たない（すなわち森である）．
  特に $\mathbf{P}$ の非零成分の数は $n + m - 1$ 以下である．
\end{proposition}

\begin{proof}
  背理法による．$\mathbf{P}$ が頂点であり，かつ $G(\mathbf{P})$ が
  閉路 $(i_1, j_1'), (i_2, j_1'), (i_2, j_2'), \ldots, (i_k, j_k'), (i_1, j_k')$ を
  もつとする．

  この閉路に沿って交互に符号 $+1, -1, +1, \ldots$ を付与し，
  対応する成分に小さな摂動 $\pm \theta$ を加える行列 $\mathbf{E} \in \R^{n \times m}$ を
  構成する（閉路上にない成分は $0$）．
  $\theta > 0$ を閉路上の $\mathbf{P}$ の最小値より小さく取れば，
  \[
    \mathbf{Q} \defeq \mathbf{P} + \mathbf{E},
    \qquad
    \mathbf{R} \defeq \mathbf{P} - \mathbf{E}
  \]
  はいずれも非負．閉路の構造から $\mathbf{E}$ の各行和・各列和は $0$ ゆえ，
  $\mathbf{Q}, \mathbf{R} \in \CouplingsD(\mathbf{a}, \mathbf{b})$．
  $\mathbf{P} = \tfrac{1}{2}(\mathbf{Q} + \mathbf{R})$
  かつ $\mathbf{Q} \neq \mathbf{P}$ ゆえ，$\mathbf{P}$ が頂点であることに矛盾．

  $k$ 個の節点をもつ閉路のないグラフの辺数は $k - 1$ 以下．
  $G(\mathbf{P})$ は $n + m$ 個の節点をもつから，辺数（$= \mathbf{P}$ の非零成分数）は
  $n + m - 1$ 以下である．
\end{proof}

\section{主双対最適性条件}
\label{sec:sem-primal-dual}

ネットワーク単体法の各反復は，主実行可能解 $\mathbf{P}$ と
\textbf{相補的}な双対変数 $(\mathbf{f}, \mathbf{g})$ の対を保つ．
この対が双対実行可能なら $\mathbf{P}$ は最適となる．以下を Claim として確認する．

\begin{claim}{主双対最適性}{sem-primal-dual-optimality}
  $\mathbf{P} \in \CouplingsD(\mathbf{a}, \mathbf{b})$ と
  $(\mathbf{f}, \mathbf{g}) \in \R^n \times \R^m$ が以下を満たすとする：
  \begin{enumerate}
    \item[(i)] \textbf{相補性}：$P_{i,j} > 0 \Longrightarrow f_i + g_j = C_{i,j}$，
    \item[(ii)] \textbf{双対実行可能性}：$f_i + g_j \leq C_{i,j}$（$\forall i, j$）．
  \end{enumerate}
  このとき $\mathbf{P}$ は離散 Kantorovich 問題の最適解であり，
  \[
    \MKD_{\mathbf{C}}(\mathbf{a}, \mathbf{b})
    = \inner{\mathbf{C}}{\mathbf{P}}
    = \inner{\mathbf{f}}{\mathbf{a}} + \inner{\mathbf{g}}{\mathbf{b}}.
  \]
\end{claim}

\begin{proof}
  任意の $\mathbf{Q} \in \CouplingsD(\mathbf{a}, \mathbf{b})$ に対し，
  双対実行可能性 (ii) と $Q_{i,j} \geq 0$ から
  \[
    \inner{\mathbf{C}}{\mathbf{Q}}
    = \sum_{i,j} C_{i,j} Q_{i,j}
    \geq \sum_{i,j} (f_i + g_j) Q_{i,j}.
  \]
  右辺を周辺条件 $\sum_j Q_{i,j} = a_i$, $\sum_i Q_{i,j} = b_j$ で整理すると
  \[
    \sum_{i,j} (f_i + g_j) Q_{i,j}
    = \sum_i f_i \sum_j Q_{i,j} + \sum_j g_j \sum_i Q_{i,j}
    = \inner{\mathbf{f}}{\mathbf{a}} + \inner{\mathbf{g}}{\mathbf{b}}.
  \]
  したがって任意の $\mathbf{Q}$ について
  $\inner{\mathbf{C}}{\mathbf{Q}} \geq \inner{\mathbf{f}}{\mathbf{a}} + \inner{\mathbf{g}}{\mathbf{b}}$．

  $\mathbf{P}$ については相補性 (i) から $P_{i,j} > 0$ の項で等号が成り立ち，
  $P_{i,j} = 0$ の項はそもそも和に寄与しない．ゆえに
  \[
    \inner{\mathbf{C}}{\mathbf{P}}
    = \sum_{i,j} (f_i + g_j) P_{i,j}
    = \inner{\mathbf{f}}{\mathbf{a}} + \inner{\mathbf{g}}{\mathbf{b}}.
  \]
  以上から
  $\inner{\mathbf{C}}{\mathbf{P}} \leq \inner{\mathbf{C}}{\mathbf{Q}}$
  が任意の $\mathbf{Q}$ で成立し，$\mathbf{P}$ は最適．
\end{proof}

\section{ネットワーク単体法}
\label{sec:sem-network-simplex}

\textbf{ネットワーク単体法}（network simplex）は最適輸送問題に特化した単体法で，
輸送多面体の頂点間を移動しながら最適解を探索する．
各反復で以下の操作を行う：
\begin{enumerate}
  \item 現在の主実行可能頂点 $\mathbf{P}$ の台グラフを含む基底木 $T$ から，
    相補性 $f_i + g_j = C_{i,j}$（$(i,j') \in T$）を満たす
    双対変数 $(\mathbf{f}, \mathbf{g})$ を構成する．
  \item 全ての $(i, j)$ で $f_i + g_j \leq C_{i,j}$ なら，
    主張~\ref{clm:sem-primal-dual-optimality} から $\mathbf{P}$ は最適．終了．
  \item 違反辺 $f_i + g_j > C_{i,j}$ を $T$ に追加して生じる閉路に沿い
    流量を更新し，流量 $0$ の辺を除去して次の頂点に移る．
\end{enumerate}
各ステップでコストは単調に減少し，
多項式時間で最適解に到達する（Orlin, 1997）．

\section{エントロピー正則化への動機付け}
\label{sec:sem-motivation-entropic}

ネットワーク単体法は離散 Kantorovich 問題の \textbf{exact} な解法を与えるが，
学習・大規模応用の観点ではいくつかの根本的な制約をもつ：

\begin{itemize}
  \item \textbf{計算量}：$O(nm)$ 個の変数に対する反復・データ構造管理は，
    $n, m$ が $10^5$ オーダーになると現実的でない．
    GPU 並列化にも本質的に向かない（離散探索的）．
  \item \textbf{微分不可能性}：最適値 $\MKD_{\mathbf{C}}(\mathbf{a}, \mathbf{b})$ は
    $\mathbf{a}, \mathbf{b}$ について区分線形にしかならず，
    滑らかな損失関数として自動微分による勾配最適化に組み込みにくい．
  \item \textbf{標本バイアス}：母集団からの有限サンプルに基づいて
    $\MKD_{\mathbf{C}}$ を推定すると，
    高次元では母集団値への収束が指数的に遅い（次元の呪い）．
\end{itemize}

これらの困難は，目的関数に \textbf{エントロピー} を加えて
\textbf{正則化}することで大きく緩和される：
正則化された最適輸送（\textbf{Sinkhorn アルゴリズム}）は
計算が GPU 並列で軽く，最適値は marginals に関して滑らか（微分可能）であり，
さらに統計的安定化の理論的基礎も提供する．
