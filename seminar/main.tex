\subsection*{一般記号}
\begin{tabular}{ll}
  $(\X, d_\X),\; (\Y, d_\Y)$ & Polish 空間（完備可分距離空間） \\
  $\N$ & 正の整数全体 $\{1, 2, 3, \ldots\}$ \\
  $\R_+$ & $[0,\infty)$ \\
  $\R_{++}$ & $(0,\infty)$ \\
  $\Bb(\X)$ & $\X$ 上のボレル $\sigma$-代数（$\Bb(\Y)$ も同様） \\
\end{tabular}

\subsection*{測度}
\begin{tabular}{ll}
  $\Mm_+^1(\X)$ & $\X$ 上の確率測度の全体 \\
  $\delta_x$ & 点 $x$ に質量1を集中させた Dirac 測度 \\
  $T\pushforward \alpha$ & ボレル可測写像 $T$ による $\alpha$ の押し出し \\
  $P_\X,\; P_\Y$ & 積空間 $\X \times \Y$ からの射影 \\
\end{tabular}

\subsection*{離散版}
\begin{tabular}{ll}
  $\range{n}$ & 整数の集合 $\{1, \ldots, n\}$ \\
  $\mathbf{a} \in \R_+^n$, $\mathbf{b} \in \R_+^m$
    & 輸送元・輸送先のヒストグラム（$\sum_i a_i = \sum_j b_j = 1$） \\
  $\mathbf{C} \in \R_+^{n \times m}$ & 行列（$C_{i,j}$：ビン $i$ から $j$ への輸送コストと解釈） \\
  $\ones_n$ & 成分がすべて $1$ の $n$ 次元ベクトル \\
  $\inner{\cdot}{\cdot}$ & フロベニウス内積 $\inner{\mathbf{A}}{\mathbf{B}} = \tr(\mathbf{A}^\top \mathbf{B})$ \\
  $\odot,\; \oslash$ & 成分ごとの積（アダマール積）・成分ごとの除算 \\
\end{tabular}

\subsection*{最適輸送}
\begin{tabular}{ll}
  $\Couplings(\alpha, \beta)$ & $\alpha$ と $\beta$ のカップリングの集合 \\
  $\CouplingsD(\mathbf{a}, \mathbf{b})$ & 離散カップリング集合 \\
  $\MK_c(\alpha, \beta)$ & Kantorovich 問題の最適値（連続版） \\
  $\MKD_{\mathbf{C}}(\mathbf{a}, \mathbf{b})$ & Kantorovich 問題の最適値（離散版） \\
  $\PotentialsD(\mathbf{C})$ & Kantorovich 双対ポテンシャルの実行可能集合 \\
\end{tabular}

\subsection*{エントロピー正則化}
\begin{tabular}{ll}
  $\Hb(\mathbf{P})$ & 離散エントロピー $-\sum_{i,j} P_{i,j}(\log P_{i,j} - 1)$ \\
  $\mathbf{K}$ & Gibbs カーネル $K_{i,j} = \exp(-C_{i,j}/\varepsilon)$ \\
  $\mathbf{u},\mathbf{v}$ & Sinkhorn スケーリングベクトル \\
  $\KLD(\mathbf{P} \| \mathbf{K})$ & KL ダイバージェンス \\
  $\MKD_{\mathbf{C}}^\varepsilon$ & エントロピー正則化された最適輸送の最適値 \\
  $\smin_\varepsilon$ & ソフト最小値 $-\varepsilon \log \sum_i e^{-z_i/\varepsilon}$ \\
\end{tabular}

\subsection*{VAE・リーマン幾何}
\begin{tabular}{ll}
  $\Zz$ & VAE の潜在空間 ($\cong \R^d$) \\
  $G(z)$ & 点 $z$ におけるリーマン計量テンソル \\
  $J_\mu(z)$ & デコーダ平均 $\mu_\theta$ のヤコビ行列 \\
  $G_W(z)$ & Wasserstein 引き戻し計量テンソル \\
\end{tabular}
