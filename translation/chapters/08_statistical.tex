\chapter{統計的ダイバージェンス}
\label{c-statistical}

本章では Wasserstein 距離の統計的性質を研究する。より具体的には、データサイエンスにおいて日常的に用いられる他の主要な距離やダイバージェンスとの比較を行う。2つの確率分布間の距離を、それらの分布からのサンプルのみにアクセス可能な場合にどのように近似できるかを定量化する。これらの主題を導入するため、第\ref{sec-phi-div}節および第\ref{sec-dual-norms}節では、それぞれ確率分布間のダイバージェンスと積分確率距離を概説する。ダイバージェンス $D$ は典型的には $D(\al,\be) \geq 0$ かつ $D(\al,\be)=0 \Leftrightarrow \al=\be$ を満たすが、対称性や三角不等式を満たす必要はない。測度に対する積分確率距離は、指定されたテスト関数の族を用いて定義される双対ノルムである。これらの量は Wasserstein 距離の健全な代替であり、第\ref{c-variational}章で扱われるように、推論問題に取り組むための損失関数として日常的に用いられている。まず第\ref{sec-non-embeddability}節では、optimal transport 距離が Hilbert 的でないこと、\ie 確率測度の適切な特徴表現上の Hilbert 距離を用いて効率的に近似できないことを示す。第\ref{sec-empirical-wass}節では、$\al$ と $\be$ からそれぞれ抽出された離散サンプル $(x_i)_i$ および $(y_j)_j$ から $D(\al,\be)$ をどのように近似するかを示す。この問題の統計的な理解は、Wasserstein 距離を機械学習で用いる際に極めて重要である。なお、本節は主に\emph{連続}集合上に台を持つ分布間の optimal transport の統計的近似に関するものである。基底空間が有限である場合の同じ問題については、\citet{sommerfeld2018inference}の研究に続く文献で注目されており、エントロピー正則化された量への拡張は \citet{bigot2017central} によって行われている。

\section{$\phi$-ダイバージェンス}
\label{sec-phi-div}

次節で Wasserstein $\Wass_1$ との構成上の類似性を持つ「弱い」ノルムを詳述する前に、測度間のいわゆるダイバージェンスの一般的な構成を説明する。これは確率分布を推定する際の損失関数として用いることができる。ダイバージェンスは、質量の輸送の概念を導入することなく、2つの入力測度の質量を\emph{点ごとに}比較する。ダイバージェンスは、2つの測度間の点ごとの比を見ることで、それらがどれほど近いかの感覚を与える汎関数である。これらは良好な解析的・計算的性質を持ち、\emph{エントロピー関数}に基づいて構築される。

\begin{defn}[エントロピー関数]
\label{def_entropy}
関数 $\phi : \RR \to \RR \cup \{\infty\}$ がエントロピー関数であるとは、$\phi$ が下半連続かつ凸であり、$\dom \phi\subset [0,\infty[$ を満たし、次の実行可能性条件を満たす場合をいう：$\dom \phi \; \cap\;  ]0, \infty[\; \neq \emptyset$。$\phi$ の無限遠における増大の速さは次で記述される：
\eq{
\phi'_\infty = \lim_{x\rightarrow +\infty} \varphi(x)/x \in \RR \cup \{\infty\} \, .
}
\end{defn}

$\phi'_\infty = \infty$ の場合、$\phi$ は任意の線形関数よりも速く増大し、$\phi$ は\emph{超線形}であるという。任意のエントロピー関数 $\phi$ は、以下のように $\phi$-ダイバージェンス（Cisz\'ar ダイバージェンス \citep{ciszar1967information,ali1966general} または $f$-ダイバージェンスとも呼ばれる）を誘導する。



\begin{defn}[$\phi$-ダイバージェンス]
\label{def_divergence}
$\phi$ をエントロピー関数とする。
$\al,\be \in \Mm(\X)$ に対し、$\frac{\d \al}{\d \be} \be + \al^{\perp}$ を $\al$ の $\be$ に関する Lebesgue 分解\footnote{Lebesgue 分解定理は、$\be$ が与えられたとき、$\al$ が2つの測度 $\al^s + \al^\perp$ の和として一意に分解されることを主張する。ここで $\al^s$ は $\be$ に関して絶対連続であり、$\al^\perp$ と $\be$ は互いに特異である。}とする。ダイバージェンス $\Divergm_\phi$ は次で定義される：
\eql{\label{eq-phi-div}
	\Divergm_\phi (\al|\be) \eqdef \int_\X \phi\left(\frac{\d \al}{\d \be} \right) \d \be
+ \phi'_\infty \al^{\perp}(\X)
}
ただし $\al,\be$ が非負の場合に限り、それ以外は $\infty$ とする。
\end{defn}%

式~\eqref{eq-phi-div} における追加項 $\phi'_\infty \al^{\perp}(\X)$ は、$\phi$ が無限遠で線形増大する場合（例えば全変動ノルムを定義する絶対値~\eqref{eq-tv-entropy} の場合）でも、$\Divergm_\phi$ が（測度の弱位相に対して）連続な汎関数を定義することを保証するために重要である。$\phi$ が超線形増大する場合、\eg 通常のエントロピー~\eqref{eq-shannon-entropy} では $\phi'_\infty=+\infty$ であり、$\al$ が $\be$ に関して密度を持たなければ $\Divergm_\phi (\al|\be) = +\infty$ となる。

離散設定において、
\eql{\label{eq-div-disc-meas}
	\al=\sum_i \a_i \de_{x_i}
	\qandq \be=\sum_i \b_i \de_{x_i}
}
が $n$ 個の点 $(x_i)_{i=1}^n \subset \X$ の同一の集合上に台を持つと仮定すると、式~\eqref{eq-phi-div} は $\simplex_n$ 上のダイバージェンスを定義する：
\eql{\label{eq-discr-diverg}
	\DivergmD_\phi(\a|\b) = \sum_{i \in \Supp(\b)} \phi\pa{ \frac{\a_i}{\b_i} } \b_i + \phi'_\infty \sum_{i \notin \Supp(\b)} \a_i,
}
ここで $\Supp(\b) \eqdef \enscond{i \in \range{n}}{ b_i \neq 0 }$ である。

次の命題の証明は \cite[Thm 2.7]{LieroMielkeSavareLong} にある。

\begin{proposition}
$\phi$ がエントロピー関数ならば、$\Divergm_\phi$ は $(\al,\be)$ に関して同時に $1$-同次、凸、かつ弱*下半連続である。
\end{proposition}

\begin{rem}[双対表現]
	$\phi$-ダイバージェンスは $\phi$ の Legendre 変換
	\eq{
		\phi^*(s) \eqdef \usup{t \in \RR} st - \phi(t)
	}
	（式~\eqref{eq-legendre} も参照）を用いて次のように表現できる：
	\eq{
		\Divergm_\phi (\al|\be) = \usup{f: \X \rightarrow \RR} \int_\X f(x) \d\al(x) - \int_\X \phi^*(f(x)) \d\be(x);
	}
	詳細は \citet{LieroMielkeSavareLong} を参照されたい。
\end{rem}

\begin{example}[Kullback--Leibler ダイバージェンス]
\label{ex_KLdiv}
Kullback--Leibler ダイバージェンス $\KL \eqdef \Divergm_{\phi_{\KL}}$ は相対エントロピーとも呼ばれ、すでに式~\eqref{eq-defn-rel-entropy} および式~\eqref{eq-kl-defn} で導入されている。これは Shannon--Boltzmann エントロピー関数 $\phi_{\KL}$ に付随するダイバージェンスであり、次で与えられる：
\eql{\label{eq-shannon-entropy}
	\phi_{\KL}(s)= \begin{cases}
		s\log(s)-s+1 & \textnormal{for } s>0 , \\
		1 & \textnormal{for } s=0 , \\
		+\infty & \textnormal{otherwise.}
		\end{cases}
}
\end{example}

\begin{remark}[Bregman ダイバージェンス]\label{rem-bregman}
離散 KL ダイバージェンス $\KLD \eqdef \DivergmD_{\phi_{\KL}}$ は、$\phi$-ダイバージェンスであると同時に Bregman ダイバージェンスでもあるという特有の性質を持つ。
%
$\RR^n$ 内の離散ベクトルに対し、滑らかな狭義凸関数 $\psi : \RR^n \rightarrow \RR$ に付随する Bregman ダイバージェンス \citep{bregman1967relaxation} は次で定義される：
\eql{\label{eq-bregman-discr}
	\VectMode{B}_\psi(\a|\b) \eqdef \psi(\a) - \psi(\b) - \dotp{\nabla \psi(\b)}{\a-\b},
}
ここで $\dotp{\cdot}{\cdot}$ は $\RR^n$ 上の標準内積である。$\VectMode{B}_\psi(\a|\b)$ は $\a$ の凸関数であり、$\psi$ の線形関数であることに注意されたい。
%
$\phi$-ダイバージェンスと同様に、Bregman ダイバージェンスは $\VectMode{B}_\psi(\a|\b) \geq 0$ を満たし、$\VectMode{B}_\psi(\a|\b)=0$ と $\a=\b$ は同値である。
%
KL ダイバージェンスは、式~\eqref{eq-discr-entropy} で定義されるエントロピーの符号を反転させた $\psi = -\HD$ に対する Bregman ダイバージェンスである。\ie $\KLD = \VectMode{B}_{-\HD}$ である。
%
Bregman ダイバージェンスは局所的には二乗ユークリッド距離である。実際、
\eq{
	\VectMode{B}_\psi(\a + \varepsilon|\a + \eta) = \dotp{ \partial^2 \psi(\a)(\varepsilon-\eta) }{\varepsilon-\eta}
	+ o(\norm{\varepsilon-\eta}^2)
}
であり、分離点の集合 $\enscond{\a}{\VectMode{B}_\psi(\a|\b) = \VectMode{B}_\psi(\a|\b')}$ は $\b$ と $\b'$ の間の超平面である。
%
これらの性質により、Bregman ダイバージェンスは一次最適化手法においてユークリッド距離の代替として適している。最もよく知られた例は、式~\eqref{eq-explicit-metricflow} の形の陽的降下ステップであるミラー勾配降下法 \citep{beck2003mirror} である。
%
Bregman ダイバージェンスは凸最適化においても重要であり、例えば Sinkhorn 反復を導出し、有限次元における収束を研究するために用いることができる。注意~\ref{rem-iterative-projection} を参照されたい。
\end{remark}

\begin{remark}[KL の双曲幾何学]
	Kullback--Leibler ダイバージェンスの幾何学を、Gauss 分布を比較する際の二次 optimal transport が定義する幾何学と対比させることは興味深い。例えば \citet{costa2015fisher} が詳述するように、Kullback--Leibler ダイバージェンスは Gauss 密度に対して閉じた形の式を持つ。一変量の場合 $\dim=1$ で $\al = \Nn(m_\al,\si_\al^2)$、$\be = \Nn(m_\be,\si_\be^2)$ とすると、
	\eql{\label{eq-kl-gaussian}
		\KL(\al|\be) = \frac{1}{2}\pa{
			\frac{\si_\al^2}{\si_\be^2}  +
			\log\pa{\frac{\si_\be^2}{\si_\al^2}}
			+ \frac{|m_\al-m_\be|}{\si_\be^2}
			 - 1
		}.
	}
	この式は、$\si_\be$ が $0$ に減少し $\be$ が Dirac 質量になるにつれて、$\al$ と $\be$ の間のダイバージェンスが無限大に発散することを示している。この意味で、KL 幾何学において特異 Gauss 分布は他のすべての Gauss 分布から無限に遠いといえる。したがって、この幾何学は特異共分散を避けたい場合に有用である。解析を簡単にするため、KL の無限小幾何学を見ることができ、これは2次の Taylor 展開を行うことで得られる：
	\eq{
		\KL( \Nn(m+\de_m,(\si+\de_\si)^2) | \Nn(m,\si^2) ) = \frac{1}{\si^2}\pa{
			\frac{1}{2}\de_m^2 + \de_\si^2
		} +
		o( \de_m^2,\de_\si^2 ).
	}
	この局所 Riemann 計量、いわゆる Fisher 計量は、$(m/\sqrt{2},\si) \in \RR \times \RR_{+,*}$ 上で表現すると、双曲 Poincar\'e 半平面の計量と正確に一致する。
	%
	この空間上の測地線は $\si=0$ の直線を中心とする半円であり、指数的な速さを持つ。\ie $\si=0$ の極限に達するのは無限時間後である。
	%
	特に、$\si_\al=\si_\be$ だが $m_\al \neq m_\al$ の場合、この双曲半平面上の $(\al,\be)$ 間の測地線は標準偏差が一定ではないことに注意されたい。

	KL の Gauss パラメータ $(m,\si)$ の空間上の双曲幾何学は、注意~\ref{rem-dist-gaussians} で記述される OT に付随するユークリッド幾何学と対比されるべきである。一変量の場合、
	\eql{\label{eq-univariate-gauss}
		\Wass_2^2( \al,\be ) = |m_\al - m_\be |^2 + |\si_\al-\si_\be|^2.
	}
	図~\ref{fig-kl-gaussians} はこれら2つの幾何学とそれぞれの測地線の視覚的比較を示している。
	%
	この興味深い比較は Jean Feydy 氏によって提案されたものである。
\end{remark}

\begin{example}[全変動]\label{exmp-tv}
全変動距離 $\TV \eqdef \Divergm_{\phi_{\TV}}$ は次に付随するダイバージェンスである：
\eql{\label{eq-tv-entropy}
	\phi_{\TV}(s)= \begin{cases}
		|s-1| & \textnormal{for } s\geq0 , \\
		+\infty & \textnormal{otherwise.}
		\end{cases}
}
これは実際には測度の全空間 $\Mm(\X)$ 上のノルムを定義する：
\eql{\label{eq-defn-tv}
	\TV(\al|\be) = \norm{\al-\be}_{\TV},
	\qwhereq
	\norm{\al}_{\TV} = |\al|(\X) = \int_\X \d|\al|(x).
}
$\al$ が $\X=\RR^\dim$ 上の密度 $\density{\al}$ を持つ場合、TV ノルムは関数の $L^1$ ノルムであり、$\norm{\al}_{\TV} = \int_\X |\density{\al}(x)| \d x = \norm{\density{\al}}_{L^1}$ となる。
%
$\al$ が式~\eqref{eq-div-disc-meas} のように離散の場合、TV ノルムは $\RR^n$ のベクトルの $\ell^1$ ノルムであり、$\norm{\al}_{\TV}=\sum_i |\a_i| = \norm{\a}_{\ell^1}$ となる。
\end{example}

\begin{rem}[強位相と弱位相]
	全変動ノルム~\eqref{eq-defn-tv} は測度の空間上のいわゆる「強」位相を定義する。
	%
	半径 $R$ のコンパクト領域 $\X$ 上では、
	\eq{
		\Wass_1(\al,\be) \leq R \norm{\al-\be}_{\TV}
	}
	が成り立つため、この強い収束の概念は Wasserstein 距離で距離化される弱収束を含意する。
	%
	しかし、逆は成り立たない。$x \rightarrow y$ であっても $\de_x$ は強収束の意味で $\de_y$ に収束しないからである（$x \neq y$ ならば $\norm{\de_x-\de_y}_{\TV}=2$ であることに注意）。
	%
	主要な利点は、$\Mm_+^1(\Xx)$（再びコンパクトな基底空間 $\X$ 上で）が弱位相に関してコンパクトであることであり、したがって確率測度の任意の列 $(\al_k)_k$ から常に収束する部分列を抽出できる。これにより、第\ref{c-variational}章で考えるような最適化問題に適した空間となる。
\end{rem}


\begin{example}[Hellinger]\label{exmp-hellinger}
	Hellinger 距離 $\Hellinger \eqdef \Divergm_{\phi_{H}}^{1/2}$ は次に付随するダイバージェンスの平方根である：
	\eq{
		\phi_{H}(s)= \begin{cases}
			|\sqrt{s}-1|^2 & \textnormal{for } s\geq0 , \\
			+\infty & \textnormal{otherwise.}
			\end{cases}
	}
	その名の通り、$\Hellinger$ は $\Mm_+(\X)$ 上の距離であり、$\norm{\cdot}_{\TV}$ と同様に強位相を距離化する。
	%
	$(\al,\be)$ が $\X=\RR^\dim$ 上の密度 $(\density{\al},\density{\be})$ を持つ場合、$\Hellinger(\al,\be) = \norms{\sqrt{\density{\al}}-\sqrt{\density{\be}}}_{L^2}$ となる。
	%
	$(\al,\be)$ が式~\eqref{eq-div-disc-meas} のように離散の場合、$\Hellinger(\al,\be) = \norms{\sqrt{\a}-\sqrt{\b}}$ となる。
	%
	$\phi_{L^p}(s)=|s^{1/p}-1|^p$ を考えると、Hellinger（$p=2$）と全変動（$p=1$）の距離が一般化され、$\Divergm_{\phi_{L^p}}^{1/p}$ は $0<p<+\infty$ に対して強収束を距離化する距離となる。
\end{example}


\begin{example}[Jensen--Shannon 距離]
	KL ダイバージェンスは対称ではなく、Bregman ダイバージェンス（局所的には二次ノルム）ではあるものの、距離の二乗ではない。一方、Jensen--Shannon 距離 $\text{JS}(\al,\be)$ は次で定義される：
	\eq{
		\text{JS}(\al,\be)^2 \eqdef \frac{1}{2}\pa{
			\KL(\al|\xi) + \KL(\be|\xi)
		}
		\qwhereq
		\xi = \frac{\al+\be}{2},
	}
	これは距離である \citep{endres2003new,osterreicher2003new}。
	%
	$\text{JS}^2$ は $\phi(s) = t\log(t)-(t+1)\log(t+1)$ に対する $\phi$-ダイバージェンスであることが示される。
	%
	$\KL$ とは鮮明な対照をなし、$\text{JS}(\al,\be)$ は常に有界であり、具体的には $0 \leq \text{JS}(\al,\be)^2 \leq \ln(2)$ を満たす。
	%
	TV ノルムや Hellinger 距離と同様に、強収束を距離化する。
\end{example}



\begin{example}[$\chi^2$]\label{exmp-chisquare}
	$\chi^2$-ダイバージェンス $\chi^2 \eqdef \Divergm_{\phi_{\chi^2}}$ は次に付随するダイバージェンスである：
	\eq{
		\phi_{\chi^2}(s)= \begin{cases}
			|s-1|^2 & \textnormal{for } s\geq0 , \\
			+\infty & \textnormal{otherwise.}
			\end{cases}
	}
	$(\al,\be)$ が式~\eqref{eq-div-disc-meas} のように離散であり、同一の台を持つ場合、
	\eq{
		\chi^2(\al|\be) = \sum_i \frac{(\a_i-\b_i)^2}{\b_i}.
	}
\end{example}

\section{積分確率距離}
\label{sec-dual-norms}


定式化~\eqref{eq-w1-cont} は双対ノルムの特殊な場合である。双対ノルムは、任意の測度を扱える「弱い」ノルムを設計するための便利な方法である。可測関数の対称凸集合 $B$ に対し、
\eql{\label{eq-dual-norm-cont}
	\norm{\al}_B \eqdef
	\umax{\f} \enscond{ \int_{\X} \f(x) \d\al(x) }{ \f \in B}.
}
と定義する。これらの双対ノルムは「積分確率距離」としばしば呼ばれる。\citep{sriperumbudur2012empirical} を参照されたい。



\begin{example}[全変動]
全変動ノルム（例~\ref{exmp-tv}）は連続関数の全空間に付随する双対ノルムである：
\eq{
	B = \enscond{f \in \Cc(\X)}{\norm{f}_\infty \leq 1}.
}
全変動距離は、双対ノルムでもある唯一の非自明なダイバージェンスである。\citep{sriperumbudur2009integral} を参照されたい。
\end{example}

\begin{rem}[弱収束の距離化]\label{rem-ipm-weak}

\subsection{$\Wass_1$ とフラットノルム}

集合 $B$ が有界ならば、$\norm{\cdot}_B$ は測度の全空間 $\Mm(\Xx)$ 上のノルムである。
%
$\Wass_1$ の場合はそうではなく、$\int_\X \d\al=0$ を満たす $\al$ に対してのみ定義される（そうでなければ $\norm{\al}_B=+\infty$）。
%
これは、例えばフラットノルムを定義するために、ポテンシャル $\f$ の値に上界を課すことで緩和できる。

\begin{example}[$\Wass_1$ ノルム]
式~\eqref{eq-w1-cont} で定義される $\Wass_1$ は双対ノルム~\eqref{eq-dual-norm-cont} の特殊な場合であり、
\eq{
	B = \enscond{f}{\Lip(f) \leq 1}
}
すなわち 1-Lipschitz 関数の集合を用いる。
\end{example}

\begin{example}[フラットノルムと Dudley 距離]
フラットノルムは次を用いて定義される：
\eql{\label{eq-set-flatnorm}
	B=\enscond{f}{\norm{\nabla \f}_\infty \leq 1 \qandq \norm{\f}_\infty \leq 1}.
}
これは測度の全空間 $\Mm(\X)$ 上で弱収束を距離化する。
%
式~\eqref{eq-w1-discr} は、制約 $\abs{\fD_k} \leq 1$ を追加することでフラットノルムの計算に拡張される。
%
フラットノルムは「Kantorovich--Rubinstein」ノルム \citep{hanin1992kantorovich} と呼ばれることもあり、イメージングにおける逆問題の適合度項として用いられている \citep{lellmann2014imaging}。
%
フラットノルムは Dudley 距離に類似しており、Dudley 距離は次を用いる：
\eq{\label{eq-set-dudley}
	B=\enscond{f}{\norm{\nabla \f}_\infty + \norm{\f}_\infty \leq 1}.
}
\end{example}

\subsection{双対 RKHS ノルムと最大平均不一致}
\label{sec-mmd}


カーネル法、より具体的には再生核 Hilbert 空間（RKHS; データサイエンスにおける応用の概観については \citep{scholkopf2002learning} を参照）の手法を用いて、測度上に「ユークリッド的」ノルム（二次汎関数を用いて構築される）を定義することも可能である。まず基本的な定義を確認する。

\begin{defn}\label{def-negativedefinitekernel}
集合 $\X \times \X$ 上で定義された対称関数 $k$（それぞれ $\varphi$）が正定値（それぞれ負定値）であるとは、任意の $n\geq0$、族 $x_1,\dots,x_n\in\Z$、およびベクトル $r\in\RR^n$ に対して次の不等式が成り立つ場合をいう：
\eql{\label{eq-dual-kern}
	\sum_{i,j=1}^n r_i r_j k(x_i,x_j)\geq 0,
	\quad
	\left(\text{resp.}\quad\sum_{i,j=1}^n r_i r_j \varphi(x_i,x_j)\leq 0\right).
}
正定値性が式~\eqref{eq-dual-kern} において零平均ベクトル $r$（\ie $\dotp{r}{\ones_n}=0$ を満たすもの）に対してのみ成り立つ場合、カーネルは条件付き正定値であるという。
\end{defn}


\if 0 %%%% REMOVED
Using a positive definite kernel $k$, one can now define a Hilbertian norm $\norm{\cdot}_\Krkhs$ on functions
\eq{
	\norm{f}_\Krkhs^2 \eqdef \int_{\X \times \X} \Krkhs(x,y) f(x) f(y) \d \rho(x) \d \rho(y)
}
where $\rho$ is some reference measure, typically Lebesgue's measure on $\X=\RR^\dim$. This norm can then be used to define a subset $B$ of functions with bounded RKHS norm,
\eql{\label{eq-dual-rkhs}
	B = \enscond{\f}{ \norm{\f}_\Krkhs \leq 1}
}
%
that is plugged into notation~\eqref{eq-dual-norm-cont}. The dual RKHS norm for measures that results from this construction is often referred to as the ``maximum mean discrepancy'' (MMD) (see~\citet{gretton2007kernel}), and can be revisited through the prism of Kernel Mean Embeddings (see~\citet{muandet2017kernel} for a review).
\fi


$k$ が条件付き正定値の場合、次のノルムを定義する：
\eql{\label{eq-kernel-dual}
	\norm{\al}^2_{\Krkhs} \eqdef \int_{\X \times \X} \Krkhs(x,y) \d \al(x) \d \al(y).
}
これらのノルムは「最大平均不一致」（MMD）としばしば呼ばれ（\citep{gretton2007kernel} を参照）、形状解析では「カーネルノルム」とも呼ばれている \citep{glaunes2004diffeomorphic}。
%
この式~\eqref{eq-kernel-dual} は、$\X$ 上の法則 $\al$ に従う2つの独立な確率ベクトル $(X,X')$ を導入することで次のように言い換えられる：
\eq{
	\norm{\al}^2_{\Krkhs} = \EE_{X,X'}( \Krkhs(X,X') ).
}
$\norm{\cdot}^2_{\Krkhs}$ は $k$ に付随する RKHS の単位球 $B$ に関する式~\eqref{eq-dual-norm-cont} の意味での双対ノルムであることが示される。RKHS 関数空間の詳細については \citep{berlinet03reproducing,Hofmann2008,scholkopf2002learning} を参照されたい。

\begin{rem}[普遍カーネル]
注意~\ref{rem-ipm-weak} によれば、MMD ノルム $\norm{\cdot}_k$ が弱収束を距離化するのは、双対球 $B$ が張る空間が連続関数の空間 $\Cc(\Xx)$ 内で一様ノルム $\norm{\cdot}_\infty$ に関して稠密である場合である。これは、$\sum_{i=1}^n a_i k(x_i,\cdot)$ の形の有限和（$n$ と点 $(x_i)_i$ の任意の選択に対して）が $\Cc(\Xx)$ 内で一様ノルム $\norm{\cdot}_\infty$ に関して稠密であることを意味する。$\X=\RR^d$ 上の並進不変カーネル $\Krkhs(x,y)=\Krkhs_0(x-y)$ に対しては、これは非消滅 Fourier 変換 $\hat \Krkhs_0(\om) >0$ を持つことと同値である。
\end{rem}

$\al$ が式~\eqref{eq-pair-discr} の形の離散測度である特殊な場合には、次の単純な式が得られる：
\eq{
	\norm{\al}_{\Krkhs}^2 = \sum_{i=1}^n \sum_{i'=1}^n \a_i\a_{i'} \KrkhsD_{i,i'}  = \dotp{\KrkhsD\a}{\a}
	\qwhereq
	\KrkhsD_{i,i'} \eqdef \Krkhs(x_i,x_{i'}).
}
特に、$\al=\sum_{i=1}^n \a_i \de_{x_i}$ および $\be=\sum_{i=1}^n \b_i \de_{x_i}$ が同一の点集合上に台を持つ場合、$\norm{\al-\be}_{\Krkhs}^2 = \dotp{\KrkhsD(\a-\b)}{\a-\b}$ となるため、$\norm{\cdot}_{\Krkhs}$ は単体 $\simplex_n$ 上のユークリッドノルム（$\KrkhsD$ が正定値ならば真のノルム、$\KrkhsD$ が半正定値ならば退化したノルム）である。
%
式~\eqref{eq-pair-discr} の形の2つの離散測度間の不一致を計算するには、次を用いることができる：
\eql{\label{eq-mmd-discr}
	\norm{\al-\be}_{\Krkhs}^2 =
		\sum_{i,i'} \a_i \a_{i'} \Krkhs(x_i,x_{i'})	+
		\sum_{j,j'} \b_j \b_{j'} \Krkhs(y_j,y_{j'}) - 2
		\sum_{i,j} \a_i \b_j \Krkhs(x_i,y_j).
}

\begin{example}[Gauss RKHS]\label{exmp-gaussian-kernel}
最も広く使われるカーネルの1つは Gauss カーネル $\Krkhs(x,y) = e^{-\frac{\norm{x-y}^2}{2 \si^2}}$ であり、$\Xx=\RR^d$ 上の正定値普遍カーネルである。
%
Gauss カーネルの魅力的な特徴は、1次元カーネルの積として分離可能であることであり、正則格子上で作業する際に計算を容易にする（注意~\ref{rem-separable} も参照）。
しかし、Gauss カーネルを用いる際に生じる重要な問題は、バンド幅パラメータ $\si$ を選択する必要があることである。このバンド幅は、比較される測度内の観測間の「典型的なスケール」に合致すべきである。測度がマルチスケールな特徴を持つ場合（ある領域は非常に密であり、他は非常にまばらである場合）、Gauss カーネルは適合せず、次に詳述する「スケールフリー」なカーネルを検討すべきである。このようなスケールフリーカーネルの問題は、それらが大域的であり（遅い多項式減衰を持つ）、コンパクト台による近似が不可能であるため、計算コストが典型的により高くなることである。
%

\end{example}

\begin{example}[$H^{-1}(\RR^\dim)$]\label{exp-sobolev-neg1}
もう1つの重要な双対ノルムは $H^{-1}(\RR^\dim)$ であり、これは $L^2(\RR^\dim)$ に属する導関数を持つ関数の Sobolev 空間 $H^1(\RR^\dim)$ の（超関数上の）双対である。
%
これはプライマル RKHS ノルム $\norm{\nabla \f}_{L^2(\RR^\dim)}^2$ を用いて定義される。$\dim=1$ でない限り特異測度（\emph{例えば} Dirac）に対しては定義されない。これは Sobolev 空間 $H^1(\RR^\dim)$ の関数が一般に連続でないためである。
%
この $H^{-1}$ ノルム（零平均で密度を持つ測度の空間上で定義される）はダイバージェンス形式で定式化することもできる：
\eql{\label{eq-dual-sobolev-div}
	\norm{\al-\be}_{H^{-1}(\RR^\dim)}^2 =
	\umin{\flow} \enscond{ \int_{\RR^\dim} \norm{\flow(x)}_2^2 \d x }{  \diverg(\flow)=\al-\be },
}
これは式~\eqref{eq-w1-cont-div} と対比されるべきであり、そこではここで用いられている $L^2$ ノルムの代わりにベクトル場 $\flow$ の $L^1$ ノルムが用いられていた。
%
この Sobolev 双対ノルムの「重み付き」版は、
\eq{
	\norm{\rho}_{H^{-1}(\al)}^2 =
	\umin{\diverg(\flow)=\rho} \int_{\RR^\dim} \norm{\flow(x)}_2^2 \d \al(x),
}
であり、Wasserstein $\Wass_2$ ノルムの自然な「線形化」と解釈できる。Benamou--Brenier 動的定式化が無限小的に次のように解釈できるという意味でである：
\eql{\label{eq-wass-asymp-sob}
	\Wass_2(\al,\al+\varepsilon \rho) = \varepsilon \norm{\rho}_{H^{-1}(\al)} + o(\varepsilon).
}
汎関数 $\Wass_2(\al,\be)$ と $\norm{\al-\be}_{H^{-1}(\al)}$ は同値であることが示される \citep{peyre2011comparison}。
%
問題は、$\norm{\al-\be}_{H^{-1}(\al)}$ がノルムではないこと（$\al$ による重み付けのため）であり、$(\al,\be)$ が密度を持たない限り一般にこれを $\norm{\al-\be}_{H^{-1}(\RR^\dim)}$ で置き換えることはできない。
%
この場合、$\al$ と $\be$ が同一の台上に $a>0$ で下から、$b<+\infty$ で上から有界な密度を持つならば、
\eql{\label{eq-compar-w2-sob}
	b^{-1/2} \norm{\al-\be}_{H^{-1}(\RR^\dim)} \leq W_2(\al,\be) \leq a^{-1/2} \norm{\al-\be}_{H^{-1}(\RR^\dim)};
}
が成り立つ。\citep[Theo. 5.34]{SantambrogioBook} を参照し、最良定数については \citep{peyre2011comparison} を参照されたい。
\end{example}

\begin{example}[負指数 Sobolev 空間]\label{exp-sobolev-neg}
	任意の負指数の Sobolev 空間 $H^{-r}(\RR^\dim)$ を考えることでこの構成を一般化できる。これは、（超関数の意味で）$r$ 階の導関数が $L^2(\RR^\dim)$ に属する関数の Sobolev 空間 $H^{r}(\RR^\dim)$ の双対である。
	%
	弱収束を距離化するためには、$H^{r}(\RR^\dim)$ の関数が連続である必要があり、これは $r>\dim/2$ の場合に成り立つ。次元 $\dim$ が増加するにつれて、より高い正則性を考える必要がある。
	%
	任意の $\al$（必ずしも整数とは限らない）に対して、これらの空間は Fourier 変換を用いて定義され、Fourier 変換 $\hat \al(\om)$（ここでは Lebesgue 測度 $\d\om$ に関する密度として記述）を持つ測度 $\al$ に対し、
	\eq{
		\norm{\al}_{H^{-r}(\RR^\dim)}^2 \eqdef \int_{\RR^\dim} \norm{\om}^{-2r} |\hat \al(\om)|^{2}\d\om.
	}
	これは畳み込みカーネル $k(x,y)=k_0(x-y)$、$\hat k_0(\om) = \pm \norm{\om}^{-2r}$ を持つ双対 RKHS ノルムに対応する。逆 Fourier 変換を取ると、（定数倍を除いて）次が得られる：
	 \eql{\label{eq-sob-kern}
	 	\foralls x \in \RR^\dim, \quad
		k_0(x) =
		\choice{
			\frac{1}{\norm{x}^{d-2r}} \qifq r < d/2,  \\
			-\norm{x}^{2r-d} \qifq r>d/2.
		}
	}
\end{example}




\begin{example}[エネルギー距離]\label{exp-energy-dist}
距離 $\dist$ に付随するエネルギー距離（$\dim=1$ の場合は Cramer 距離）\citep{szekely2004testing} は次で定義される：
\eql{\label{eq-defn-ed}
	\norm{\al-\be}_{\text{ED}(\X,\dist^p)} \eqdef \norm{\al-\be}_{\Krkhs_{\text{ED}}}
	\qwhereq
	\Krkhs_{\text{ED}}(x,y) = -\dist(x,y)^p
}
ここで $0 < p < 2$ である。
%
$\dist$ が負定値（定義~\ref{def-negativedefinitekernel} を参照）ならば、これは測度上の有効な MMD ノルムである。典型的な例はユークリッド距離 $\dist(x,y)=\norm{x-y}$ である。
%
$\X=\RR^\dim$、$\dist(x,y)=\norm{\cdot}$ の場合、式~\eqref{eq-sob-kern} を用いると、エネルギー距離は Sobolev ノルムであることがわかる：
\eq{
	\norm{\cdot}_{\text{ED}(\RR^\dim,\norm{\cdot}^p)} = \norm{\cdot}_{H^{-\frac{d+p}{2}}(\RR^\dim)}.
}
エネルギー距離が Gauss カーネル（例~\ref{exmp-gaussian-kernel}）のようなより一般的なカーネルに対して持つ主要な利点は、スケールフリーであり、バンド幅パラメータ $\si$ に依存しないことである。より正確には、$\X=\RR^\dim$ 上で因子 $s>0$ による拡大 $f_s(x)=sx$ を用いると、次のスケーリング挙動が成り立つ：
\eq{
	\norm{f_{s\sharp}(\al-\be)}_{\text{ED}(\RR^\dim,\norm{\cdot}^p)} = s^{\frac{p}{2}} \norm{\al-\be}_{\text{ED}(\RR^\dim,\norm{\cdot}^p)},
}
一方、Wasserstein 距離は完全な線形スケーリングを示す：
\eq{
	\Wass_p( f_{s\sharp}\al,f_{s\sharp}\be) ) = s \Wass_p(\al,\be) ).
}
ただし、エネルギー距離ではパラメータ $p$ は $0 < p < 2$ を満たす必要があり、$p=2$ では平均間の距離に退化する：
\eq{
	\norm{\al-\be}_{\text{ED}(\RR^\dim,\norm{\cdot}^2)}
	=
	\norm{\int_{\RR^\dim} x (\d\al(x)-\d\be(x))  },
}
したがって、もはやノルムではない。このことは、エネルギー距離では $f_{s\sharp}$ の下で Wasserstein 距離と同じ線形スケーリングを得ることができないことを示している。
\end{example}

\section{Wasserstein 空間は Hilbert 的でない}\label{sec-non-embeddability}

第\ref{sec:specialcases}節で先に概説した Wasserstein 幾何学の特殊な場合のいくつかは、optimal transport 距離が閉じた形で計算できる場合があることを示してきた。また、そのような場合に optimal transport 距離が確率測度間の \emph{Hilbert 的}距離であること、すなわち以下に定義されるように、入力測度の空間から Hilbert 空間への写像 $\tphi$ が存在することも示している。

\begin{defn}\label{def-Hilbertmetric}
集合 $\Z\times \Z$ 上で定義された距離 $d$ が Hilbert 的であるとは、Hilbert 空間 $\RKHS$ と写像 $\tphi:\Z\rightarrow \RKHS$ が存在して、$\Z$ の任意のペア $z,z'$ に対し $d(z,z')=\|\tphi(z)-\tphi(z')\|_\RKHS$ が成り立つ場合をいう。
\end{defn}

例えば、注意~\ref{rem-1d-ot-generic} は、Wasserstein 距離が一変量分布間の Hilbert ノルムであることを示しており、$\tphi$ を測度にその一般化分位関数を対応させる写像と定義するだけでよい。注意~\ref{rem-dist-gaussians} は、一変量 Gauss 分布に対して、本章の式~\eqref{eq-univariate-gauss} に書かれているように、2つの一変量 Gauss 分布間の Wasserstein 距離がそれらの平均と標準偏差の間のユークリッド距離に他ならないことを示している。

Hilbert 的距離はデータ解析の文脈で用いる際に多くの好ましい性質を持つ \citep{dattorro2010convex}。第一に、それらは放射基底関数カーネルとして容易に定式化できる：任意の Hilbert 的距離 $\dist$ に対し、\citep[Cor. 3.3.3, Prop. 3.2.7]{berg84harmonic} に示されるように、$e^{-\dist^p/t}$ は任意の $0\leq p\leq 2$ と任意の正のスカラー $t$ に対して正定値カーネルである。Gauss カーネル（$p=2$）と Laplace カーネル（$p=1$）は、通常のユークリッド距離を用いたこの結果の単純な応用である。カーネル法の分野全体 \citep{Hofmann2008} は、凸学習アルゴリズムを正定値カーネル行列上で定義するために、カーネル関数の正定値性に基づいている。Hilbert 空間に住む点は、多次元尺度構成法 \citep{borg2005modern} のような単純な手法を用いて、低い歪み率で効率的に低次元に埋め込むこともできる \citep{Johnson84}, \citep[\S V.6.2]{barvinok2002course}。

Hilbert 的距離がこのような性質を持つため、上述のものよりも一般的な設定、特に $\X$ の次元が2以上の場合に、Wasserstein 距離が Hilbert 的であり続けることを期待するかもしれない。これは次の同値性を用いて反証できる。

\begin{prop}\label{prop-negative-definite} 距離 $d$ が Hilbert 的であることと $d^2$ が負定値であることは同値である。
\end{prop}
\begin{proof} 距離が Hilbert 的ならば、$d^2$ は自明に負定値である。実際、$\Z$ 内の $n$ 個の点が与えられたとき、和 $\sum r_i r_j d^2(z_i,z_j)$ は $\sum r_i r_j \|\tphi(z_i)-\tphi(z_j)\|_\RKHS^2$ と書き換えられ、$\sum r_i=0$ であることを利用して展開すると $-2\sum r_i r_j \dotp{\tphi(z_i)}{\tphi(z_j)}_\RKHS$ となり、Hilbert 内積の定義により非正である。逆に、$d^2$ が負定値ならば、$d$ が Hilbert 的であることは \citet{schoenberg38} による重要な結果（\cite[p. 82, Prop. 3.2]{berg84harmonic}）から従う。
\end{proof}

したがって、二乗 Wasserstein 距離が負定値でないことを示せば、次の命題に述べられるように、それが Hilbert 的でないことが示される。

\begin{prop}\label{prop-negative-definite-negative} $\X=\RR^\dim$（$\dim\geq 2$）で ground コストが $\dist(x,y)=\norm{x-y}_2$ と設定されている場合、$p$-Wasserstein 距離は $p=1,2$ に対して Hilbert 的でない。
\end{prop}

\begin{proof} $\dim=2$ に対して結果を示せば十分である。その次元での反例は任意のより高い次元での反例を得るのに十分だからである。次のように定義される4つのベクトル $x^1, x^2,x^3,x^4\in\RR^{2}$ 上に台を持つ測度を用いた非確率的反例を与える：$x^{1}=[0,0], x^{2}=[1,0], x^{3}=[0,1], x^{4}=[1,1]$。4次元の単体上の刻み幅 $1/4$ の正則格子上のすべての点を考える。そのような点は単体内に $35=\bigl(\!\bigl(\substack{4\\4}\bigl)\!\bigl)=\binom{4+4-1}{4}$ 個ある。各確率ベクトル $\a^i$ は、$j\leq 4$ に対して $\a^i_j$ が集合 $\{0,\tfrac{1}{4},\tfrac{1}{2},\tfrac{3}{4},1\}$ に属し、$\sum_{j=1}^4\a^i_j=1$ を満たすような格子上の点である。与えられた $p$ に対し、これらのヒストグラム間の $35\times 35$ のペアワイズ Wasserstein 距離行列 $\D_p$ を計算できる。$\D_p$ が負定値でないのは、その要素ごとの二乗 $\D_p^2$ に対して $\J \D_p^2 \J$ が正の固有値を持つ場合であり、ここで $\J$ はセンタリング行列 $\J=\Identity_n-\tfrac{1}{n}\ones_{n,n}$ である。これは図~\ref{fig-counterexample} に示されている通りである。
\end{proof}

\begin{figure}[h!]

\includegraphics[width=.6\linewidth]{counterexample_nsdness/maxeig_vs_p}
\caption{\label{fig-counterexample}

\subsection{埋め込みと歪み}

重要な研究群が、Hilbert 的埋め込みを用いた Wasserstein 距離の近似の困難さを定量化している。
%
$\X=\RR^\dim$（$\dim\geq 2$）の場合、$\ell_2$ 空間への測度の埋め込みは必然的に重大な歪みを伴うことが示されている（\citet{naor-2005,andoni2015snowflake}）。

$0 < p \leq 1$ に対する $p$-Wasserstein 空間を $\ell_1$ に準等長的に埋め込むことは可能であるが（\citep{indyk,andoni2008earth,do2011sublinear} を参照）、距離間の同値定数は次元 $\dim$ とともに急速に増大する。また、$p=1$ の場合、埋め込みは離散測度に対してのみ成り立つ（\ie 埋め込み定数はスパイク間の最小距離に依存する）ことに注意されたい。
%
密接に関連する埋め込み手法として、$\Wass_1$ を Lipschitz 関数 $f$ の双対として特徴づけること（第\ref{sec-w1-eucl}節を参照）を用い、Lipschitz 制約 $\norm{\nabla f}_1 \leq 1$ をウェーブレット係数上の重み付き $\ell_1$ 球で近似する方法がある。\citep{shirdhonkar2008approximate} を参照されたい。
%
このウェーブレット係数の重み付き $\ell_1$ 球は、負指数のいわゆる Besov 空間を定義する \citep{leeb2016holder}。
%
これらの埋め込み結果は、ダイアディック分割を用いて得られる Wasserstein 距離の上界とも類似している。\citep[Prop. 1]{weed2017sharp} および \citep{fournier2015rate} を参照されたい。
%
これも $\ell_1$ への準等長的埋め込み（この埋め込みは再スケールされたウェーブレット係数によって与えられる）を提供し、入力測度が一様格子上で離散化されている場合にこの埋め込みが線形時間で近似的に計算できるという利点を持つ。ウェーブレットの詳細については \citep{mallat2008wavelet} を参照されたい。マルチスケール埋め込みを用いて Wasserstein 的距離を計算するという発想は、コンピュータビジョンにおいて広く用いられてきた。例えば \citep{ling2006diffusion,grauman2005pyramid,CuturiNIPS2006,lazebnik2006beyond} を参照されたい。

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\subsection{Optimal Transport の負定値/正定値な変種}

後の第\ref{sec-sliced}節で、Wasserstein 距離の\emph{スライス}近似（本質的には、直線上に射影された測度のランダム push-forward に対して計算される1次元方向の輸送距離の和）が、負定値関数の和として負定値であることを示す \citep[\S 3.1.11]{berg84harmonic}。この結果は正定値カーネルを定義するために用いることができる \citep{kolouri2016sliced}。正定値カーネルを回復する別の方法は、optimal transport 問題を（すべての可能な輸送テーブルにわたる）最小値ではなくソフトミン問題として定式化することであり、これはエントロピー正則化を導入するために \citet{kosowsky1994invisible} によって提案された。そのソフトミンは、その負の指数関数（母関数とも呼ばれる）が正定値であるような項を定義する \citep{cuturi2012positivity}。

\section{OT、MMD および $\phi$-ダイバージェンスの経験的推定量}
\label{sec-empirical-wass}

応用的な設定において、2つの入力測度 $(\al,\be) \in \Mm_+^1(\X)^2$ が与えられたとき、$\al$ からのサンプル $(x_i)_{i=1}^n$ と $\be$ からのサンプル $(y_j)_{j=1}^m$ のみを用いて（通常は未知の）ダイバージェンス $D(\al,\be)$ を近似することは、重要な統計的問題である。これらのサンプルは、それぞれの分布から独立同分布に従うと仮定する。

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\subsection{OT と MMD の経験的推定量}

Wasserstein 距離 $\Wass_p$（式~\ref{eq-defn-wass-dist} を参照）と MMD ノルム（第\ref{sec-dual-norms}節を参照）の両方に対し、分布間の未知の距離の直接的な推定量は、経験測度間で直接計算することであり、理想的には後者から前者への収束速度を制御できることが期待される：
\eq{
	D(\al,\be) \approx D(\hat \al_n,\hat \be_m) \qwhereq
	\choice{
		\hat \al_n \eqdef \frac{1}{n}\sum_i \de_{x_i},\\
		\hat \be_m \eqdef \frac{1}{m}\sum_j \de_{y_j}.
	}
}
ここで $\hat \al_n$ と $\hat \be_m$ はともに確率測度であるため、$D(\hat \al_n,\hat \be_m)$ は確率変数であることに注意されたい。
%
簡単のため、$\X$ はコンパクトであると仮定する（非有界領域を扱うには入力測度のモーメントに関する追加の制約が必要である）。

弱収束を距離化する（定義~\ref{dfn-weak-conv} を参照）双対距離に対し、弱収束 $\hat\al_n \rightarrow \al$ があるので、$n \rightarrow +\infty$ で $D(\hat \al_n,\hat \be_n) \rightarrow D(\al,\be)$ が成り立つ。
%
しかし、重要な問題は $D(\hat \al_n,\hat \be_n)$ の $D(\al,\be)$ への収束速度であり、この速度はしばしば $D$ の「標本複雑度」と呼ばれる。

$D(\al,\be) = \norm{\cdot}_{\TV}$ の場合、TV ノルムは弱収束を距離化しないため、$\norms{\hat \al_n - \hat \be_n}_{\TV}$ は一致推定量ではない。すなわち $\norm{\al-\be}_{\TV}$ に収束しない。実際、確率 1 で $\norms{\hat \al_n - \hat \be_n}_{\TV}=2$ となる。2つの離散測度の台が重ならないためである。他の $\phi$-ダイバージェンスでも同様の問題が生じ、経験分布間のダイバージェンスを用いて推定することはできない。


%%%%
\paragraph{OT の収束速度}

$\X=\RR^\dim$ で有界領域上に台を持つ測度に対し、$\dim>2$ かつ $1 \leq p < +\infty$ の場合、\citep{dudley1969speed} によって次が示されている：
\eq{
	\EE( |\Wass_p(\hat \al_n,\hat \be_n)-\Wass_p(\al,\be)| ) = O(n^{-\frac{1}{\dim}}),
}
ここで期待値 $\EE$ はランダムサンプル $(x_i,y_i)_i$ に関して取られる。2つの測度の一方が Lebesgue 測度に関して密度を持つ場合、この速度は $\RR^\dim$ において最良である。この結果は被覆数の概念を用いて一般的な距離空間に対して証明され \citep{dudley1969speed}、後に特に $\X=\RR^d$ に対して精密化された \citep{dereich2013constructive,fournier2015rate}。
この速度は、測度が低次元の部分領域上に台を持つ場合に精密化できる。\citet{weed2017sharp} は、速度が実際には台の固有次元に依存することを示している。\citeauthor{weed2017sharp} は、離散近似された測度（\emph{例えば}小さな分散を持つ Gauss 混合）のような非漸近的振る舞いも研究している。
%
$\Wass_p(\hat \al_n,\hat \be_n)$ の平均 $\Wass_p(\al,\be)$ 周りの集中不等式を証明することも可能である。\citep{bolley2007quantitative,boissard2011simple,weed2017sharp} を参照されたい。

%%%
\paragraph{MMD の収束速度}

式~\eqref{eq-kernel-dual} で定義される RKHS ノルムの双対である弱ノルム $\norm{\cdot}^2_{\Krkhs}$（MMD とも呼ばれる）に対しては、Wasserstein 距離とは対照的に、標本複雑度は周囲の次元に依存しない：
\eq{
	\EE( |\norms{\hat \al_n-\hat \be_n}_{\Krkhs} - \norm{\al-\be}_{\Krkhs} | ) = O(n^{-\frac{1}{2}});
}
\citep{sriperumbudur2012empirical} を参照されたい。
%
図~\ref{fig-sample-complexity} は Wasserstein 距離と MMD 距離の標本複雑度の数値的比較を示している。
%
ただし、$\norms{\hat \al_n-\hat \be_n}_{\Krkhs}^2$ は $\norm{\al-\be}_{\Krkhs}^2$ のわずかに偏った推定量であることに注意されたい。不偏推定量を定義し、例えばそのような損失を最小化する際に SGD を用いることができるようにするためには、次の不偏推定量を用いるべきである：
\begin{align*}
	\text{MMD}_{\Krkhs}(\hat \al_n,\hat \be_n)^2 & \eqdef
		\frac{1}{n(n-1)} \sum_{i,i'} \Krkhs(x_i,x_{i'})	+
		\frac{1}{n(n-1)}\sum_{j,j'} \Krkhs(y_j,y_{j'}) \\
		 & \qquad - 2
		\frac{1}{n^2}\sum_{i,j} \Krkhs(x_i,y_j),
\end{align*}
これは式~\eqref{eq-mmd-discr} と比較されるべきである。$\EE(\text{MMD}_{\Krkhs}(\hat \al_n,\hat \be_n)^2) = \norm{\al-\be}_{\Krkhs}^2$ を満たす。\citep{gretton2012kernel} を参照されたい。

\subsection{$\phi$-ダイバージェンスの経験的推定量}

式~\eqref{def_divergence} で定義される $\Divergm_\phi(\al|\be)$ を、$\Divergm_\phi(\hat \al_n|\hat \be_n)$ を用いて離散サンプルから近似することはできない。実際、この量は $+\infty$ であるか（例えば $\KL$ ダイバージェンスの場合）、または $n \rightarrow +\infty$ で $\Divergm_\phi(\al|\be)$ に収束しない（例えば $\TV$ ノルムの場合）。
%
代わりに、離散経験測度を何らかの形で平滑化し密度に置き換える密度推定量を用いる必要がある。\citep{silverman1986density} を参照されたい。
%
ユークリッド空間 $\Xx=\RR^\dim$ において、滑らかな窓関数とバンド幅 $\si>0$ を用いて $h_\si=h(\cdot/\si)$ と置くと、$\al$ の密度推定量はこのカーネルに対する畳み込みを用いて定義される：
\eql{\label{eq-kernel-density-estimate}
	\hat \mu_n \star h_\si = \frac{1}{n}\sum_i h_\si(\cdot-x_i).
}
%
$\phi$ ダイバージェンスは次のように近似できる：
\eq{
	\Divergm_\phi^\si(\hat \al_n |\hat \be_n)
	\eqdef
	\frac{1}{n} \sum_{j=1}^n
	\phi\pa{
		\frac{
			\sum_i h_\si(y_j-x_i)
		}{
			\sum_{j'} h_\si(y_j-y_{j'}),
		}
	}
}
ここで $\si$ はサンプル数 $n$ と次元 $\dim$ に適応させるべきである。
%
固定バンド幅 $\si$ の選択を回避し、代わりに最近傍の数 $k$ を選択するノンパラメトリック推定量を考案することも可能である。これらの手法は典型的には最近傍間の距離 \citep{loftsgaarden1965nonparametric} を利用するものであり、これは局所的なサンプリング密度に合わせてバンド幅 $\si$ を局所的に適応させることに類似している。$\De_k(x)$ を $x \in \RR^\dim$ と $(x_i)_{i=1}^n$ の中のその $k$ 番目の最近傍との間の距離とすると、密度推定量は次で定義される：
\eql{\label{eq-knn-density-estimate}
	\rho_{\hat \al_n}^k(x) \eqdef \frac{k/n}{ |B_\dim| \De_k(x)^r },
}
ここで $|B_\dim|$ は $\RR^\dim$ 内の単位球の体積である。
%
式~\eqref{eq-kernel-density-estimate} において幅 $\si$ の領域内に落ちるサンプルの数を何らかの形で「数える」代わりに、この式~\eqref{eq-knn-density-estimate} は $k$ 個のサンプルを包含するために必要な半径を推定する。図~\ref{fig-density-estimation} は推定量~\eqref{eq-kernel-density-estimate} と~\eqref{eq-knn-density-estimate} を比較している。
%
典型的な応用例はエントロピー汎関数に対する式~\eqref{eq-discr-entropy} に詳述されており、これは Lebesgue 測度に関する KL ダイバージェンスである。詳細については \citep{BerishaHero} を参照されたい。

\section{エントロピー正則化：OT と MMD の間}
\label{sec-entropy-ot-mmd}

命題~\ref{prop-sinkhorn-div} に従い、Sinkhorn ダイバージェンスが次で定義されることを想起する：
\eq{
	\SINKHORNP_\C^\varepsilon(\a,\b) \eqdef
	\dotp{\P^\star}{\C} =
	\dotp{e^{\frac{\fD^\star}{\varepsilon}}}{(\K\odot \C)e^{\frac{\gD^\star}{\varepsilon}}},
}
ここで $\P^\star$ は式~\eqref{eq-regularized-discr} の解であり、$(\fD^\star,\gD^\star)$ は式~\eqref{eq-dual-formulation} の解である。
%
$\C_{i,j}=\dist(x_i,x_j)^p$（$\X$ 上のある距離 $\dist$ に対して）と仮定し、式~\eqref{eq-pair-discr} の形の2つの離散確率分布に対し、これは正則化 Wasserstein コストを定義する：
\eq{
	\Wass_{p,\varepsilon}(\al,\be)^p \eqdef \SINKHORNP_\C^\varepsilon(\a,\b).
}
この定義は任意の入力分布（必ずしも離散とは限らない）に次のように一般化される：
\eq{
	\Wass_{p,\varepsilon}(\al,\be)^p \eqdef \int_{\X \times \X} \dist(x,y)^p \d\pi^\star(x,y),
}
ここで $\pi^\star$ は式~\eqref{eq-entropic-generic} の解である。

正則化によって導入されるバイアスを打ち消すため（特に $\Wass_{p,\varepsilon}(\al,\al) \neq 0$ であるため）、補正された正則化ダイバージェンスを導入する：
\eq{
	\tilde\Wass_{p,\varepsilon}(\al,\be)^p \eqdef
	2 \Wass_{p,\varepsilon}(\al,\be)^p
	-
	\Wass_{p,\varepsilon}(\al,\al)^p
	-
	\Wass_{p,\varepsilon}(\be,\be)^p.
}
$e^{-c/\epsilon}$ が正定値カーネルならば、関連する補正ダイバージェンス（$\SINKHORNP_\C^\varepsilon$ の代わりに $\MKD_\C^\varepsilon$ を用いて得られる）が正であることが \citep{feydy2018interpolating} で証明されている。
%
正則化 optimal transport を用いた他の再正規化スキームを定義することも可能であり、例えば \citet{amari2017information} によって提案されている。

	次が成り立つ：
	\eq{
		\tilde\Wass_{p,\varepsilon}(\al,\be)  \overset{\varepsilon \rightarrow 0}{\longrightarrow} 2 \Wass_{p}(\al,\be)
		\qandq
		\tilde\Wass_{p,\varepsilon}(\al,\be)^p \overset{\varepsilon \rightarrow +\infty}{\longrightarrow} \norm{\al-\be}_{\text{ED}(\X,\dist)}^2,
	}
	ここで $\norm{\cdot}_{\text{ED}(\X,\dist)}$ は式~\eqref{eq-defn-ed} で定義される。
\end{prop}
